### Real hayot misoli:

Tasavvur qiling, siz kitobxona mudiri sizlar:

## Fayllar bilan ishlash

### 1. Fayl yaratish va yozish:

```javascript
const fs = require('fs');

// Sinxron (Sync) - kutib turadi
fs.writeFileSync('salom.txt', 'Assalom aleykum, Node.js!');
console.log('✅ Fayl yaratildi!');

// Asinxron (Async) - parallel ishlaydi
fs.writeFile('salom-async.txt', 'Assalom aleykum, Async!', (err) => {
  if (err) {
    console.error('❌ Xatolik:', err.message);
  } else {
    console.log('✅ Async fayl yaratildi!');
  }
});

// Modern async/await
const util = require('util');
const writeFileAsync = util.promisify(fs.writeFile);

async function createFile() {
  try {
    await writeFileAsync('modern.txt', 'Modern async/await!');
    console.log('✅ Modern fayl yaratildi!');
  } catch (error) {
    console.error('❌ Xatolik:', error.message);
  }
}

createFile();
```

## Papkalar (Directories) bilan ishlash

### 1. Papka yaratish:

```javascript
const fs = require('fs');
const path = require('path');

// Oddiy papka yaratish
if (!fs.existsSync('test-folder')) {
  fs.mkdirSync('test-folder');

}

// Nested papkalar yaratish

  fs.mkdirSync(nestedPath, { recursive: true });
  console.log('✅ Nested papkalar yaratildi:', nestedPath);
} catch (error) {
  console.error('❌ Xatolik:', error.message);
}

// Async versiya
async function createDirectories() {
  try {
    await fsPromises.mkdir('async-folder', { recursive: true });
    console.log('✅ Async papka yaratildi');

    // Bir nechta papka
    const folders = ['images', 'documents', 'videos'];

    for (const folder of folders) {
      await fsPromises.mkdir(path.join('async-folder', folder), { recursive: true });
      console.log(`✅ Papka yaratildi: ${folder}`);
    }

  } catch (error) {
    console.error('❌ Papka yaratishda xatolik:', error.message);
  }
}

createDirectories();
```

### 3. Papka ichidan qidirish (Recursive):

```javascript
const fs = require('fs');
const path = require('path');

async function findFiles(dirPath, extension) {
  const foundFiles = [];

  async function searchRecursive(currentPath) {
    try {
      const items = await fsPromises.readdir(currentPath);

      for (const item of items) {
        const itemPath = path.join(currentPath, item);
        const stats = await fsPromises.stat(itemPath);

        if (stats.isDirectory()) {
          // Rekursiv qidirish
          await searchRecursive(itemPath);
        } else if (path.extname(item) === extension) {
          // Qidirilayotgan fayl topildi
          foundFiles.push({
            name: item,
            path: itemPath,
            size: stats.size,
            modified: stats.mtime
          });
        }
      }
    } catch (error) {
      console.error(`❌ Xatolik ${currentPath} da:`, error.message);
    }
  }

  await searchRecursive(dirPath);
  return foundFiles;
}

// JavaScript fayllarini qidirish
async function searchJS() {
  console.log('🔍 JavaScript fayllarini qidiryapman...\n');

  const jsFiles = await findFiles('.', '.js');

  if (jsFiles.length === 0) {
    console.log('❌ JavaScript fayllar topilmadi');
    return;
  }

  console.log(`✅ ${jsFiles.length} ta JavaScript fayl topildi:\n`);

  jsFiles.forEach(file => {
    console.log(`📄 ${file.name}`);
    console.log(`   📍 Yo'l: ${file.path}`);
    console.log(`   📏 Hajm: ${(file.size / 1024).toFixed(2)} KB`);
    console.log(`   📅 O'zgartirilgan: ${file.modified.toLocaleString('uz-UZ')}`);
    console.log('');
  });
}

searchJS();
```

## Sync vs Async - Qachon qaysi birini ishlatish?

### Sinxron (Sync) operatsiyalar:

```javascript
console.log('1. Boshlang\'ich');

// Bu yerda dastur to'xtaydi!
const content = fs.readFileSync('large-file.txt', 'utf8');
console.log('2. Fayl o\'qildi');

console.log('3. Dastur davom etadi');

// Natija:
// 1. Boshlang'ich
// 2. Fayl o'qildi (5 soniyadan keyin)
// 3. Dastur davom etadi
```

**Afzalliklar:**
- ✅ Oddiy kod
- ✅ Ketma-ketlik kafolatlanadi
- ✅ Error handiling oson

**Kamchiliklar:**
- ❌ Blocking - boshqa ishlar to'xtaydi
- ❌ Slow performance
- ❌ Server uchun yomon

### Asinxron (Async) operatsiyalar:

```javascript
console.log('1. Boshlang\'ich');

// Bu yerda dastur to'xtamaydi!
fs.readFile('large-file.txt', 'utf8', (err, content) => {
  console.log('3. Fayl o\'qildi');
});

console.log('2. Dastur davom etadi');

// Natija:
// 1. Boshlang'ich
// 2. Dastur davom etadi (darhol)
// 3. Fayl o'qildi (5 soniyadan keyin)
```

**Afzalliklar:**
- ✅ Non-blocking - boshqa ishlar davom etadi
- ✅ High performance
- ✅ Server uchun ideal

**Kamchiliklar:**
- ❌ Callback hell
- ❌ Error handling murakkab
- ❌ Timing issues

### Modern Promise/Async-Await:

```javascript
const fsPromises = require('fs').promises;

async function modernFileOperations() {
  console.log('1. Boshlang\'ich');

  try {
    // Non-blocking, lekin kutadi
    const content = await fsPromises.readFile('large-file.txt', 'utf8');
    console.log('2. Fayl o\'qildi');

    // Ketma-ket operatsiyalar
    await fsPromises.writeFile('backup.txt', content);
    console.log('3. Backup yaratildi');

  } catch (error) {
    console.error('❌ Xatolik:', error.message);
  }

  console.log('4. Dastur tugadi');
}

modernFileOperations();
console.log('5. Bu darhol ko\'rsatiladi');

// Natija:
// 5. Bu darhol ko'rsatiladi
// 1. Boshlang'ich
// 2. Fayl o'qildi (5 soniyadan keyin)
// 3. Backup yaratildi
// 4. Dastur tugadi
```

```javascript
// src/logger.js
const chalk = require('chalk');
const moment = require('moment');
const fs = require('fs').promises;
const path = require('path');

class Logger {
  constructor() {
    this.logFile = path.join(__dirname, '../data/operations.json');
    this.ensureLogDir();
  }

  async ensureLogDir() {
    try {
      await fs.mkdir(path.dirname(this.logFile), { recursive: true });
    } catch (error) {
      // Directory already exists
    }
  }

  async log(level, message, extra = {}) {
    const timestamp = moment().format('YYYY-MM-DD HH:mm:ss');

    // Console'ga chiqarish
    const colors = {
      info: chalk.blue,
      success: chalk.green,
      warning: chalk.yellow,
      error: chalk.red
    };

    const icons = {
      info: 'ℹ️',
      success: '✅',
      warning: '⚠️',
      error: '❌'
    };

    console.log(colors[level](`[${timestamp}] ${icons[level]} ${message}`));

    // Faylga yozish
    try {
      let logs = [];
      try {
        const data = await fs.readFile(this.logFile, 'utf8');
        logs = JSON.parse(data);
      } catch {
        // Fayl yo'q yoki bo'sh
      }

      logs.push({
        timestamp,
        level,
        message,
        ...extra
      });

      // Faqat oxirgi 1000 ta log saqlash
      if (logs.length > 1000) {
        logs = logs.slice(-1000);
      }

      await fs.writeFile(this.logFile, JSON.stringify(logs, null, 2));
    } catch (error) {
      console.error('❌ Log yozishda xatolik:', error.message);
    }
  }

  info(message, extra) { return this.log('info', message, extra); }
  success(message, extra) { return this.log('success', message, extra); }
  warning(message, extra) { return this.log('warning', message, extra); }
  error(message, extra) { return this.log('error', message, extra); }

  async getLogs(limit = 50) {
    try {
      const data = await fs.readFile(this.logFile, 'utf8');
      const logs = JSON.parse(data);
      return logs.slice(-limit).reverse();
    } catch {
      return [];
    }
  }
}

module.exports = Logger;
```

### 3. Utilities:

```javascript
// src/utils.js
const path = require('path');

class Utils {
  static formatSize(bytes) {
    const units = ['B', 'KB', 'MB', 'GB', 'TB'];
    let i = 0;
    while (bytes >= 1024 && i < units.length - 1) {
      bytes /= 1024;
      i++;
    }
    return `${bytes.toFixed(2)} ${units[i]}`;
  }

  static formatDate(date) {
    return new Date(date).toLocaleString('uz-UZ', {
      year: 'numeric',
      month: '2-digit',
      day: '2-digit',
      hour: '2-digit',
      minute: '2-digit',
      second: '2-digit'
    });
  }

  static isValidFilename(filename) {
    // Taqiqlangan belgilar
    const invalidChars = /[<>:"/\\|?*]/;
    return !invalidChars.test(filename) && filename.trim().length > 0;
  }

  static getFileType(filePath) {
    const ext = path.extname(filePath).toLowerCase();

    const types = {
      // Matn fayllar
      '.txt': 'text',
      '.md': 'text',
      '.json': 'text',
      '.js': 'text',
      '.html': 'text',
      '.css': 'text',

      // Rasm fayllar
      '.jpg': 'image',
      '.jpeg': 'image',
      '.png': 'image',
      '.gif': 'image',
      '.svg': 'image',

      // Video fayllar
      '.mp4': 'video',
      '.avi': 'video',
      '.mov': 'video',
      '.mkv': 'video',

      // Audio fayllar
      '.mp3': 'audio',
      '.wav': 'audio',
      '.flac': 'audio',

      // Hujjat fayllar
      '.pdf': 'document',
      '.doc': 'document',
      '.docx': 'document',
      '.xls': 'document',
      '.ppt': 'document'
    };

    return types[ext] || 'unknown';
  }

  static getFileIcon(filePath, isDirectory = false) {
    if (isDirectory) return '📁';

    const type = this.getFileType(filePath);
    const icons = {
      text: '📄',
      image: '🖼️',
      video: '🎬',
      audio: '🎵',
      document: '📋',
      unknown: '📄'
    };

    return icons[type] || '📄';
  }
}

module.exports = Utils;
```

### 4. File Manager class:

```javascript
// src/fileManager.js
const fs = require('fs').promises;
const fsSync = require('fs');
const path = require('path');
const Logger = require('./logger');
const Utils = require('./utils');

class FileManager {
  constructor() {
    this.logger = new Logger();
    this.currentPath = process.cwd();
  }

  // Joriy papkani o'zgartirish
  async changeDirectory(newPath) {
    try {
      const absolutePath = path.resolve(this.currentPath, newPath);
      const stats = await fs.stat(absolutePath);

      if (!stats.isDirectory()) {
        throw new Error('Bu papka emas');
      }

      this.currentPath = absolutePath;
      await this.logger.success(`Papka o'zgartirildi: ${this.currentPath}`);
      return this.currentPath;
    } catch (error) {
      await this.logger.error(`Papka o'zgartirishda xatolik: ${error.message}`);
      throw error;
    }
  }

  // Joriy papka ichidagilarni ko'rish
  async listDirectory(dirPath = this.currentPath) {
    try {
      const items = await fs.readdir(dirPath);
      const result = [];

      for (const item of items) {
        const itemPath = path.join(dirPath, item);
        const stats = await fs.stat(itemPath);

        result.push({
          name: item,
          path: itemPath,
          isDirectory: stats.isDirectory(),
          size: stats.size,
          modified: stats.mtime,
          created: stats.birthtime,
          icon: Utils.getFileIcon(itemPath, stats.isDirectory()),
          type: stats.isDirectory() ? 'directory' : Utils.getFileType(itemPath)
        });
      }

      // Papkalar birinchi, keyin fayllar
      result.sort((a, b) => {
        if (a.isDirectory && !b.isDirectory) return -1;
        if (!a.isDirectory && b.isDirectory) return 1;
        return a.name.localeCompare(b.name);
      });

      await this.logger.info(`Directory listed: ${dirPath}`, { count: result.length });
      return result;
    } catch (error) {
      await this.logger.error(`Directory listing xatolik: ${error.message}`);
      throw error;
    }
  }

  // Fayl yaratish
  async createFile(filename, content = '') {
    try {
      if (!Utils.isValidFilename(filename)) {
        throw new Error('Noto\'g\'ri fayl nomi');
      }

      const filePath = path.join(this.currentPath, filename);

      if (fsSync.existsSync(filePath)) {
        throw new Error('Fayl allaqachon mavjud');
      }

      await fs.writeFile(filePath, content, 'utf8');
      await this.logger.success(`Fayl yaratildi: ${filename}`);
      return filePath;
    } catch (error) {
      await this.logger.error(`Fayl yaratishda xatolik: ${error.message}`);
      throw error;
    }
  }

  // Papka yaratish
  async createDirectory(dirname) {
    try {
      if (!Utils.isValidFilename(dirname)) {
        throw new Error('Noto\'g\'ri papka nomi');
      }

      const dirPath = path.join(this.currentPath, dirname);
      await fs.mkdir(dirPath, { recursive: true });
      await this.logger.success(`Papka yaratildi: ${dirname}`);
      return dirPath;
    } catch (error) {
      await this.logger.error(`Papka yaratishda xatolik: ${error.message}`);
      throw error;
    }
  }

  // Fayl o'qish
  async readFile(filename) {
    try {
      const filePath = path.resolve(this.currentPath, filename);
      const content = await fs.readFile(filePath, 'utf8');
      await this.logger.info(`Fayl o'qildi: ${filename}`);
      return content;
    } catch (error) {
      await this.logger.error(`Fayl o'qishda xatolik: ${error.message}`);
      throw error;
    }
  }

  // Fayl yozish
  async writeFile(filename, content) {
    try {
      const filePath = path.resolve(this.currentPath, filename);
      await fs.writeFile(filePath, content, 'utf8');
      await this.logger.success(`Fayl yangilandi: ${filename}`);
      return filePath;
    } catch (error) {
      await this.logger.error(`Fayl yozishda xatolik: ${error.message}`);
      throw error;
    }
  }

  // Fayl/papka o'chirish
  async delete(itemName) {
    try {
      const itemPath = path.resolve(this.currentPath, itemName);
      const stats = await fs.stat(itemPath);

      if (stats.isDirectory()) {
        await fs.rmdir(itemPath, { recursive: true });
        await this.logger.success(`Papka o'chirildi: ${itemName}`);
      } else {
        await fs.unlink(itemPath);
        await this.logger.success(`Fayl o'chirildi: ${itemName}`);
      }

      return true;
    } catch (error) {
      await this.logger.error(`O'chirishda xatolik: ${error.message}`);
      throw error;
    }
  }

  // Nusxa ko'chirish
  async copy(sourceName, destName) {
    try {
      const sourcePath = path.resolve(this.currentPath, sourceName);
      const destPath = path.resolve(this.currentPath, destName);

      await fs.copyFile(sourcePath, destPath);
      await this.logger.success(`Nusxa ko'chirildi: ${sourceName} → ${destName}`);
      return destPath;
    } catch (error) {
      await this.logger.error(`Nusxa ko'chirishda xatolik: ${error.message}`);
      throw error;
    }
  }

  // Qayta nomlash
  async rename(oldName, newName) {
    try {
      if (!Utils.isValidFilename(newName)) {
        throw new Error('Noto\'g\'ri nom');
      }

      const oldPath = path.resolve(this.currentPath, oldName);
      const newPath = path.resolve(this.currentPath, newName);

      await fs.rename(oldPath, newPath);
      await this.logger.success(`Qayta nomlandi: ${oldName} → ${newName}`);
      return newPath;
    } catch (error) {
      await this.logger.error(`Qayta nomlashda xatolik: ${error.message}`);
      throw error;
    }
  }

  // Qidirish
  async search(pattern, searchType = 'name') {
    try {
      const items = await this.listDirectory();
      const results = [];

      const searchPattern = pattern.toLowerCase();

      for (const item of items) {
        let match = false;

        switch (searchType) {
          case 'name':
            match = item.name.toLowerCase().includes(searchPattern);
            break;
          case 'extension':
            match = path.extname(item.name).toLowerCase() === searchPattern;
            break;
          case 'type':
            match = item.type === searchPattern;
            break;
        }

        if (match) {
          results.push(item);
        }
      }

      await this.logger.info(`Qidiruv: "${pattern}" (${searchType})`, { found: results.length });
      return results;
    } catch (error) {
      await this.logger.error(`Qidirishda xatolik: ${error.message}`);
      throw error;
    }
  }

  // Fayl/papka ma'lumotlari
  async getInfo(itemName) {
    try {
      const itemPath = path.resolve(this.currentPath, itemName);
      const stats = await fs.stat(itemPath);

      const info = {
        name: itemName,
        path: itemPath,
        type: stats.isDirectory() ? 'directory' : 'file',
        size: Utils.formatSize(stats.size),
        sizeBytes: stats.size,
        created: Utils.formatDate(stats.birthtime),
        modified: Utils.formatDate(stats.mtime),
        accessed: Utils.formatDate(stats.atime),
        isDirectory: stats.isDirectory(),
        isFile: stats.isFile(),
        fileType: stats.isFile() ? Utils.getFileType(itemPath) : null
      };

      await this.logger.info(`Ma'lumot olindi: ${itemName}`);
      return info;
    } catch (error) {
      await this.logger.error(`Ma'lumot olishda xatolik: ${error.message}`);
      throw error;
    }
  }

  // Disk ma'lumotlari
  getDiskInfo() {
    return {
      currentPath: this.currentPath,
      homePath: require('os').homedir(),
      tempPath: require('os').tmpdir(),
      platform: process.platform,
      pathSeparator: path.sep
    };
  }
}

module.exports = FileManager;
```

```javascript
      console.log(chalk.red(`\n❌ Xatolik: ${error.message}`));
    }
```
