## Debugging Fundamentals

**Debugging** - kodda xatoliklarni topish, tushunish va tuzatish jarayoni.

### 1. Console debugging:

```javascript
// Basic console methods
console.log('Basic log');                    // Oddiy log
console.info('Information message');         // Ma'lumot
console.warn('Warning message');             // Ogohlantirish
console.error('Error message');              // Xatolik
console.debug('Debug information');          // Debug (NODE_ENV=development'da ko'rinadi)

// Advanced console methods
const user = { id: 1, name: 'Alibek', email: 'alibek@example.com' };

console.table([user]);                       // Jadval ko'rinishida
console.dir(user, { depth: null });          // Object struktura
console.time('operation');                   // Timer start
console.timeEnd('operation');                // Timer end
console.trace('Call stack trace');           // Call stack
console.count('function calls');             // Counter
console.group('Grouped logs');               // Log grouping
console.groupEnd();

// Conditional logging
console.assert(user.id > 0, 'User ID must be positive');

// Custom logging function
function debugLog(message, data = null, level = 'info') {
  if (process.env.NODE_ENV === 'development') {
    const timestamp = new Date().toISOString();
    const prefix = `[${timestamp}] [${level.toUpperCase()}]`;

    console[level](`${prefix} ${message}`);

    if (data) {
      console[level]('Data:', JSON.stringify(data, null, 2));
    }
  }
}

// Usage
debugLog('User authentication started', { userId: 123 });
debugLog('Database query slow', { duration: '2.5s' }, 'warn');
```

```javascript
const express = require('express');
const app = express();

app.post('/api/calculate', (req, res) => {
  const { a, b, operation } = req.body;

  debugger; // Breakpoint - dastur bu yerda to'xtaydi

  switch (operation) {
    case 'add':
      debugger; // Debugging point
      result = a + b;

      result = a - b;
      break;
    case 'multiply':
      result = a * b;
      break;

        debugger; // Division by zero check
        return res.status(400).json({ error: 'Division by zero' });
      }
      result = a / b;
      break;
    default:
      debugger; // Invalid operation
      return res.status(400).json({ error: 'Invalid operation' });
  }

  res.json({ result, operation, inputs: { a, b } });
});

app.listen(3000);
```

## Error Handling va Logging

### 1. Professional error tracking:

```javascript
// utils/errorHandler.js
class AppError extends Error {
  constructor(message, statusCode, code = null, isOperational = true) {
    super(message);

    this.statusCode = statusCode;
    this.code = code;
    this.isOperational = isOperational;
    this.timestamp = new Date().toISOString();

    Error.captureStackTrace(this, this.constructor);
  }
}

class ErrorHandler {
  static handle(error, req = null) {
    // Log error
    console.error('💥 Error occurred:', {
      message: error.message,
      stack: error.stack,
      statusCode: error.statusCode,
      code: error.code,
      timestamp: error.timestamp,
      request: req ? {
        method: req.method,
        url: req.url,
        ip: req.ip,
        userAgent: req.get('User-Agent')
      } : null
    });

    // Send to external service (Sentry, LogRocket, etc.)
    if (process.env.NODE_ENV === 'production') {
      this.sendToExternalService(error, req);
    }
  }

  static sendToExternalService(error, req) {
    // Sentry, Bugsnag, or other error tracking service
    console.log('📡 Sending to error tracking service...');
  }

  static createExpressErrorHandler() {
    return (error, req, res, next) => {
      this.handle(error, req);

      const statusCode = error.statusCode || 500;
      const message = error.isOperational ? error.message : 'Internal Server Error';

      res.status(statusCode).json({
        success: false,
        error: {
          message,
          code: error.code,
          statusCode,
          timestamp: new Date().toISOString()
        },
        ...(process.env.NODE_ENV === 'development' && {
          stack: error.stack
        })
      });
    };
  }
}

// Async error catcher
const catchAsync = (fn) => {
  return (req, res, next) => {
    Promise.resolve(fn(req, res, next)).catch(next);
  };
};

module.exports = {
  AppError,
  ErrorHandler,
  catchAsync
};
```

```javascript
// utils/calculator.js
class Calculator {
  add(a, b) {
    if (typeof a !== 'number' || typeof b !== 'number') {
      throw new Error('Arguments must be numbers');
    }
    return a + b;
  }

  subtract(a, b) {
    if (typeof a !== 'number' || typeof b !== 'number') {
      throw new Error('Arguments must be numbers');
    }
    return a - b;
  }

  multiply(a, b) {
    if (typeof a !== 'number' || typeof b !== 'number') {
      throw new Error('Arguments must be numbers');
    }
    return a * b;
  }

  divide(a, b) {
    if (typeof a !== 'number' || typeof b !== 'number') {
      throw new Error('Arguments must be numbers');
    }

    if (b === 0) {
      throw new Error('Cannot divide by zero');
    }

    return a / b;
  }

  static factorial(n) {
    if (!Number.isInteger(n) || n < 0) {
      throw new Error('Input must be a non-negative integer');
    }

    if (n === 0 || n === 1) return 1;

    let result = 1;
    for (let i = 2; i <= n; i++) {
      result *= i;
    }

    return result;
  }
}

module.exports = Calculator;
```

### 3. Async function testing:

```javascript
// utils/asyncOperations.js
const fs = require('fs').promises;
const axios = require('axios');

class AsyncOperations {
  static async readFileContent(filePath) {
    try {
      const content = await fs.readFile(filePath, 'utf8');
      return content.trim();
    } catch (error) {
      throw new Error(`Failed to read file: ${error.message}`);
    }
  }

  static async fetchUserData(userId) {
    if (!userId || userId <= 0) {
      throw new Error('Valid user ID is required');
    }

    try {
      const response = await axios.get(`https://jsonplaceholder.typicode.com/users/${userId}`);
      return {
        id: response.data.id,
        name: response.data.name,
        email: response.data.email,
        username: response.data.username
      };
    } catch (error) {
      if (error.response && error.response.status === 404) {
        throw new Error('User not found');
      }
      throw new Error(`Failed to fetch user data: ${error.message}`);
    }
  }

  static async processDataWithDelay(data, delayMs = 1000) {
    await new Promise(resolve => setTimeout(resolve, delayMs));

    if (!Array.isArray(data)) {
      throw new Error('Data must be an array');
    }

    return data
      .filter(item => item != null)
      .map(item => String(item).toUpperCase());
  }
}

module.exports = AsyncOperations;
```

```javascript
  res.json({ status: 'OK', timestamp: new Date().toISOString() });

  const { name, email } = req.body;

  if (!name || !email) {
    throw new AppError('Name va email majburiy', 400, 'VALIDATION_ERROR');
  }

  if (!email.includes('@')) {

    name,
    email,
    createdAt: new Date().toISOString()
  };

  res.status(201).json({
    success: true,
    data: user
  });
});

app.get('/api/users/:id', (req, res) => {
  const userId = parseInt(req.params.id);

  if (isNaN(userId) || userId <= 0) {
    throw new AppError('Noto\'g\'ri user ID', 400, 'INVALID_ID');
  }

  // Simulate user lookup
  if (userId === 999) {
    throw new AppError('User topilmadi', 404, 'USER_NOT_FOUND');
  }

  const user = {
    id: userId,
    name: `User ${userId}`,
    email: `user${userId}@example.com`
  };

  res.json({
    success: true,
    data: user
  });
});

// Error handling
app.use(ErrorHandler.createExpressErrorHandler());
```

```javascript
// __tests__/integration/database.test.js
const { Pool } = require('pg');
const app = require('../../app');
const request = require('supertest');

describe('Database Integration Tests', () => {
  let pool;

  beforeAll(async () => {
    // Test database connection
    pool = new Pool({
      connectionString: process.env.TEST_DATABASE_URL,
      ssl: false
    });

    // Setup test tables
    await pool.query(`
      CREATE TABLE IF NOT EXISTS users (
        id SERIAL PRIMARY KEY,
        name VARCHAR(100) NOT NULL,
        email VARCHAR(100) UNIQUE NOT NULL,
        created_at TIMESTAMP DEFAULT NOW()
      );
    `);
  });

  beforeEach(async () => {
    // Clean up before each test
    await pool.query('DELETE FROM users');

    // Insert test data
    await pool.query(`
      INSERT INTO users (name, email) VALUES
      ('Alibek', 'alibek@test.com'),
      ('Malika', 'malika@test.com');
    `);
  });

  afterAll(async () => {
    // Cleanup
    await pool.query('DROP TABLE IF EXISTS users');
    await pool.end();
  });

  describe('User CRUD operations', () => {
    test('should fetch all users from database', async () => {
      const response = await request(app)
        .get('/api/users')
        .expect(200);

      expect(response.body.data).toHaveLength(2);
      expect(response.body.data[0]).toHaveProperty('name', 'Alibek');
    });

    test('should create user and save to database', async () => {
      const userData = {
        name: 'Jasur',
        email: 'jasur@test.com'
      };

      const response = await request(app)
        .post('/api/users')
        .send(userData)
        .expect(201);

      // Verify in database
      const result = await pool.query('SELECT * FROM users WHERE email = $1', [userData.email]);
      expect(result.rows).toHaveLength(1);
      expect(result.rows[0].name).toBe(userData.name);
    });

    test('should handle unique constraint violation', async () => {
      const duplicateUser = {
        name: 'Duplicate',
        email: 'alibek@test.com' // Already exists
      };

      const response = await request(app)
        .post('/api/users')
        .send(duplicateUser)
        .expect(409); // Conflict

      expect(response.body.error.code).toBe('DUPLICATE_EMAIL');
    });
  });

  describe('Transaction handling', () => {
    test('should rollback transaction on error', async () => {
      const client = await pool.connect();

      try {
        await client.query('BEGIN');

        await client.query('INSERT INTO users (name, email) VALUES ($1, $2)',
          ['Test User', 'test@example.com']);

        // Simulate error
        await client.query('INSERT INTO users (name, email) VALUES ($1, $2)',
          ['Another User', 'test@example.com']); // Duplicate email

        await client.query('COMMIT');
      } catch (error) {
        await client.query('ROLLBACK');

        // Verify rollback worked
        const result = await client.query('SELECT * FROM users WHERE email = $1', ['test@example.com']);
        expect(result.rows).toHaveLength(0);
      } finally {
        client.release();
      }
    });
  });
});
```

```javascript
// __tests__/performance/benchmarks.test.js
const app = require('../../app');
const request = require('supertest');

describe('Performance Benchmarks', () => {
  describe('API Response Times', () => {
    test('GET /health should respond within 50ms', async () => {
      const start = Date.now();

      await request(app)
        .get('/health')
        .expect(200);

      const duration = Date.now() - start;
      expect(duration).toBeLessThan(50);
    });

    test('POST /api/users should respond within 200ms', async () => {
      const userData = {
        name: 'Performance Test',
        email: 'perf@test.com'
      };

      const start = Date.now();

      await request(app)
        .post('/api/users')
        .send(userData)
        .expect(201);

      const duration = Date.now() - start;
      expect(duration).toBeLessThan(200);
    });
  });

  describe('Concurrent Requests', () => {
    test('should handle 10 concurrent requests', async () => {
      const requests = Array.from({ length: 10 }, (_, i) =>
        request(app)
          .post('/api/users')
          .send({
            name: `User ${i}`,
            email: `user${i}@concurrent.test`
          })
      );

      const responses = await Promise.all(requests);

      responses.forEach((response, i) => {
        expect(response.status).toBe(201);
        expect(response.body.data.email).toBe(`user${i}@concurrent.test`);
      });
    }, 10000); // 10 second timeout
  });

  describe('Memory Usage', () => {
    test('should not have significant memory leaks', async () => {
      const initialMemory = process.memoryUsage().heapUsed;

      // Simulate workload
      for (let i = 0; i < 100; i++) {
        await request(app).get('/health');
      }

      // Force garbage collection if available
      if (global.gc) {
        global.gc();
      }

      const finalMemory = process.memoryUsage().heapUsed;
      const memoryIncrease = finalMemory - initialMemory;

      console.log(`Memory increase: ${(memoryIncrease / 1024 / 1024).toFixed(2)} MB`);

      // Should not increase by more than 10MB
      expect(memoryIncrease).toBeLessThan(10 * 1024 * 1024);
    });
  });
});
```
