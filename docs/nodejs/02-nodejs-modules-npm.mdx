## CommonJS - require/exports tizimi

Node.js **CommonJS** formatidan foydalanadi.

### 1. Oddiy export:

```javascript
// math.js
function add(a, b) {
  return a + b;
}

function subtract(a, b) {
  return a - b;
}

function multiply(a, b) {
  return a * b;
}

// Bitta function export qilish
module.exports = add;
```

```javascript
// app.js
const add = require('./math');

console.log(add(5, 3)); // 8
// subtract va multiply yo'q! ❌
```

### 2. Object export:

```javascript
// math.js
function add(a, b) {
  return a + b;
}

function subtract(a, b) {
  return a - b;
}

function multiply(a, b) {
  return a * b;
}

// Bir nechta functionlarni export qilish
module.exports = {
  add,      // add: add qisqartmasi
  subtract,
  multiply
};

// Yoki to'liq:
/*
module.exports = {
  add: add,
  subtract: subtract,
  multiply: multiply
};
*/
```

```javascript
// app.js
const math = require('./math');

console.log(math.add(5, 3));      // 8
console.log(math.subtract(10, 4)); // 6
console.log(math.multiply(6, 7));  // 42

// Yoki destructuring:
const { add, subtract } = require('./math');

console.log(add(5, 3));      // 8
console.log(subtract(10, 4)); // 6
```

### 3. exports shorthand:

```javascript
// calculator.js
exports.add = (a, b) => a + b;
exports.subtract = (a, b) => a - b;
exports.multiply = (a, b) => a * b;
exports.divide = (a, b) => a / b;

// Bu quyidagi bilan bir xil:
// module.exports.add = (a, b) => a + b;
```

### 4. Aralash export:

```javascript
// user.js
class User {
  constructor(name, email) {
    this.name = name;
    this.email = email;
  }

  getInfo() {
    return `${this.name} (${this.email})`;
  }
}

const ADMIN_ROLE = 'admin';
const USER_ROLE = 'user';

function createUser(name, email) {
  return new User(name, email);
}

module.exports = {
  User,           // Class
  ADMIN_ROLE,     // Constant
  USER_ROLE,      // Constant
  createUser      // Function
};
```

```javascript
// app.js
const { User, ADMIN_ROLE, createUser } = require('./user');

const user1 = new User('Alibek', 'alibek@email.com');
const user2 = createUser('Malika', 'malika@email.com');

console.log(user1.getInfo()); // Alibek (alibek@email.com)
console.log(ADMIN_ROLE);      // admin
```

```javascript
// modules/fileManager.js
const fs = require('fs');
const path = require('path');

const DATA_FILE = path.join(__dirname, '../data/employees.json');

class FileManager {
  static readEmployees() {
    try {
      if (!fs.existsSync(DATA_FILE)) {
        return [];
      }
      const data = fs.readFileSync(DATA_FILE, 'utf8');
      return JSON.parse(data);
    } catch (error) {
      console.error('Faylni o\'qishda xatolik:', error.message);
      return [];
    }
  }

  static saveEmployees(employees) {
    try {
      // data papkasini yaratish (agar yo'q bo'lsa)
      const dataDir = path.dirname(DATA_FILE);
      if (!fs.existsSync(dataDir)) {
        fs.mkdirSync(dataDir, { recursive: true });
      }

      const data = JSON.stringify(employees, null, 2);
      fs.writeFileSync(DATA_FILE, data, 'utf8');
      return true;
    } catch (error) {
      console.error('Faylni saqlashda xatolik:', error.message);
      return false;
    }
  }
}

module.exports = FileManager;
```

### 5. Employee moduli:

```javascript
// modules/employee.js
const FileManager = require('./fileManager');
const Logger = require('./logger');

class Employee {
  constructor(id, name, position, salary, hireDate, department) {
    this.id = id;
    this.name = name;
    this.position = position;
    this.salary = salary;
    this.hireDate = hireDate;
    this.department = department;
  }

  // Barcha xodimlarni olish
  static getAllEmployees() {
    return FileManager.readEmployees();
  }

  // ID orqali xodimni topish
  static findById(id) {
    const employees = FileManager.readEmployees();
    return employees.find(emp => emp.id === parseInt(id));
  }

  // Yangi xodim qo'shish
  static addEmployee(employeeData) {
    const employees = FileManager.readEmployees();

    // Yangi ID yaratish
    const newId = employees.length > 0
      ? Math.max(...employees.map(emp => emp.id)) + 1
      : 1;

    const newEmployee = new Employee(
      newId,
      employeeData.name,
      employeeData.position,
      employeeData.salary,
      employeeData.hireDate,
      employeeData.department
    );

    employees.push(newEmployee);

    if (FileManager.saveEmployees(employees)) {
      Logger.success(`Xodim qo'shildi: ${employeeData.name}`);
      return newEmployee;
    } else {
      Logger.error('Xodimni qo\'shishda xatolik');
      return null;
    }
  }

  // Xodimni yangilash
  static updateEmployee(id, updates) {
    const employees = FileManager.readEmployees();
    const index = employees.findIndex(emp => emp.id === parseInt(id));

    if (index === -1) {
      Logger.error('Xodim topilmadi');
      return null;
    }

    // Ma'lumotni yangilash
    employees[index] = { ...employees[index], ...updates };

    if (FileManager.saveEmployees(employees)) {
      Logger.success(`Xodim yangilandi: ${employees[index].name}`);
      return employees[index];
    } else {
      Logger.error('Xodimni yangilashda xatolik');
      return null;
    }
  }

  // Xodimni o'chirish
  static deleteEmployee(id) {
    const employees = FileManager.readEmployees();
    const index = employees.findIndex(emp => emp.id === parseInt(id));

    if (index === -1) {
      Logger.error('Xodim topilmadi');
      return false;
    }

    const deletedEmployee = employees[index];
    employees.splice(index, 1);

    if (FileManager.saveEmployees(employees)) {
      Logger.success(`Xodim o'chirildi: ${deletedEmployee.name}`);
      return true;
    } else {
      Logger.error('Xodimni o\'chirishda xatolik');
      return false;
    }
  }

  // Statistika
  static getStatistics() {
    const employees = FileManager.readEmployees();

    if (employees.length === 0) {
      return {
        total: 0,
        departments: {},
        avgSalary: 0
      };
    }

    const departments = {};
    let totalSalary = 0;

    employees.forEach(emp => {
      // Department bo'yicha guruhlash
      if (!departments[emp.department]) {
        departments[emp.department] = 0;
      }
      departments[emp.department]++;

      // Maoshni hisoblash
      totalSalary += emp.salary;
    });

    return {
      total: employees.length,
      departments: departments,
      avgSalary: Math.round(totalSalary / employees.length)
    };
  }
}

module.exports = Employee;
```

```javascript
// app.js
const inquirer = require('inquirer');
const Employee = require('./modules/employee');
const Logger = require('./modules/logger');
const chalk = require('chalk');

async function mainMenu() {
  console.clear();
  console.log(chalk.blue.bold('🏢 EMPLOYEE MANAGEMENT SYSTEM\n'));

  const choices = [
    '📋 Barcha xodimlarni ko\'rish',
    '➕ Yangi xodim qo\'shish',
    '🔍 Xodimni qidirish',
    '✏️  Xodimni yangilash',
    '🗑️  Xodimni o\'chirish',
    '📊 Statistika',
    '❌ Chiqish'
  ];

  const { action } = await inquirer.prompt([
    {
      type: 'list',
      name: 'action',
      message: 'Amalni tanlang:',
      choices: choices
    }
  ]);

  switch (action) {
    case choices[0]: await showAllEmployees(); break;
    case choices[1]: await addNewEmployee(); break;
    case choices[2]: await searchEmployee(); break;
    case choices[3]: await updateEmployee(); break;
    case choices[4]: await deleteEmployee(); break;
    case choices[5]: await showStatistics(); break;
    case choices[6]:
      Logger.info('Dastur tugadi. Xayr!');
      process.exit(0);
  }

  // Menu'ga qaytish
  console.log('\nIltimos, Enter ni bosing...');
  await inquirer.prompt([{ name: 'continue', message: '' }]);
  return mainMenu();
}

async function showAllEmployees() {
  const employees = Employee.getAllEmployees();

  if (employees.length === 0) {

    return;
  }

  console.log(chalk.green.bold('\n📋 XODIMLAR RO\'YXATI:\n'));

  employees.forEach(emp => {

    console.log(chalk.yellow(`👤 Ism: ${emp.name}`));

    console.log(chalk.cyan(`🏢 Bo'lim: ${emp.department}`));
    console.log(chalk.green(`💰 Maosh: ${emp.salary.toLocaleString('uz-UZ')} so'm`));
    console.log(chalk.gray(`📅 Ishga kirgan: ${emp.hireDate}`));
    console.log(chalk.gray('─'.repeat(40)));
  });
}

async function addNewEmployee() {
  console.log(chalk.green.bold('\n➕ YANGI XODIM QO\'SHISH:\n'));

  const answers = await inquirer.prompt([
    {
      type: 'input',
      name: 'name',
      message: '👤 Xodim ismi:',
      validate: input => input.length > 0 || 'Ism kiritish majburiy!'
    },
    {
      type: 'input',
      name: 'position',
      message: '💼 Lavozimi:',
      validate: input => input.length > 0 || 'Lavozim kiritish majburiy!'
    },
    {
      type: 'input',
      name: 'department',
      message: '🏢 Bo\'lim:',
      validate: input => input.length > 0 || 'Bo\'lim kiritish majburiy!'
    },
    {
      type: 'input',
      name: 'salary',
      message: '💰 Maosh (so\'m):',
      validate: input => {
        const salary = parseInt(input);
        return (!isNaN(salary) && salary > 0) || 'To\'g\'ri maosh kiriting!';
      },
      filter: input => parseInt(input)
    },
    {
      type: 'input',
      name: 'hireDate',
      message: '📅 Ishga kirgan san (YYYY-MM-DD):',
      default: new Date().toISOString().split('T')[0],
      validate: input => {
        const date = new Date(input);
        return !isNaN(date.getTime()) || 'To\'g\'ri sana kiriting!';
      }
    }
  ]);

  Employee.addEmployee(answers);
}

async function searchEmployee() {
  const { searchId } = await inquirer.prompt([
    {
      type: 'input',
      name: 'searchId',
      message: '🔍 Xodim ID sini kiriting:',
      validate: input => !isNaN(parseInt(input)) || 'To\'g\'ri ID kiriting!'
    }
  ]);

  const employee = Employee.findById(searchId);

  if (employee) {
    console.log(chalk.green.bold('\n✅ XODIM TOPILDI:\n'));
    console.log(chalk.yellow(`👤 Ism: ${employee.name}`));
    console.log(chalk.blue(`💼 Lavozim: ${employee.position}`));
    console.log(chalk.cyan(`🏢 Bo'lim: ${employee.department}`));
    console.log(chalk.green(`💰 Maosh: ${employee.salary.toLocaleString('uz-UZ')} so'm`));
    console.log(chalk.gray(`📅 Ishga kirgan: ${employee.hireDate}`));
  } else {
    Logger.error('Xodim topilmadi');
  }
}

async function updateEmployee() {
  const { updateId } = await inquirer.prompt([
    {
      type: 'input',
      name: 'updateId',
      message: '✏️  Yangilamoqchi bo\'lgan xodim ID si:',
      validate: input => !isNaN(parseInt(input)) || 'To\'g\'ri ID kiriting!'
    }
  ]);

  const employee = Employee.findById(updateId);

  if (!employee) {
    Logger.error('Xodim topilmadi');
    return;
  }

  console.log(chalk.green(`\nHozirgi ma'lumotlar: ${employee.name} - ${employee.position}\n`));

  const updates = await inquirer.prompt([
    {
      type: 'input',
      name: 'name',
      message: 'Yangi ism (bo\'sh qoldirsangiz o\'zgartirilmaydi):',
      default: employee.name
    },
    {
      type: 'input',
      name: 'position',
      message: 'Yangi lavozim:',
      default: employee.position
    },
    {
      type: 'input',
      name: 'department',
      message: 'Yangi bo\'lim:',
      default: employee.department
    },
    {
      type: 'input',
      name: 'salary',
      message: 'Yangi maosh:',
      default: employee.salary,
      filter: input => parseInt(input)
    }
  ]);

  Employee.updateEmployee(updateId, updates);
}

async function deleteEmployee() {
  const { deleteId } = await inquirer.prompt([
    {
      type: 'input',
      name: 'deleteId',
      message: '🗑️  O\'chirmoqchi bo\'lgan xodim ID si:',
      validate: input => !isNaN(parseInt(input)) || 'To\'g\'ri ID kiriting!'
    }
  ]);

  const employee = Employee.findById(deleteId);

  if (!employee) {
    Logger.error('Xodim topilmadi');
    return;
  }

  const { confirm } = await inquirer.prompt([
    {
      type: 'confirm',
      name: 'confirm',
      message: `${employee.name} ni o'chirishni tasdiqlaysizmi?`,
      default: false
    }
  ]);

  if (confirm) {
    Employee.deleteEmployee(deleteId);
  } else {
    Logger.info('O\'chirish bekor qilindi');
  }
}

async function showStatistics() {
  const stats = Employee.getStatistics();

  console.log(chalk.blue.bold('\n📊 STATISTIKA:\n'));

  if (stats.total === 0) {
    Logger.warning('Ma\'lumotlar yo\'q');
    return;
  }

  console.log(chalk.green(`👥 Jami xodimlar: ${stats.total}`));
  console.log(chalk.yellow(`💰 O'rtacha maosh: ${stats.avgSalary.toLocaleString('uz-UZ')} so'm`));

  console.log(chalk.blue('\n🏢 Bo\'limlar bo\'yicha:'));
  Object.entries(stats.departments).forEach(([dept, count]) => {
    console.log(chalk.cyan(`   ${dept}: ${count} xodim`));
  });
}
```
