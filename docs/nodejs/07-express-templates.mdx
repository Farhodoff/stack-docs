# Express Templates va Dynamic HTML

Assalom aleykum! Oldingi darsda Express middleware'lari bilan ishladik. Endi **template engine'lar** bilan dynamic HTML yaratishni o'rganamiz - real web saytlar qanday ishlashiningasosi!

## O'qish rejasi

Bu darsda quyidagilarni o'rganasiz:
- 🎨 **Template Engine nima** - static vs dynamic content
- 📝 **EJS** - Embedded JavaScript templates
- 🔧 **Handlebars** - Logic-less templates
- 🏗️ **Template inheritance** - layout va partials
- 📊 **Data passing** - variable'lar va loops
- 💼 **Real loyiha** - Blog Website bilan SSR

---

## Template Engine nima va nega kerak?

**Template Engine** - HTML'ni JavaScript ma'lumotlari bilan birlashtirib, dynamic content yaratish uchun tool.

### Static vs Dynamic content:

**Static HTML:**
```html
<!-- Doimo bir xil -->
<h1>Salom!</h1>
<p>Bu static content</p>
```

**Dynamic HTML (Template bilan):**
```html
<!-- Har safar boshqacha -->
<h1>Salom, <%= user.name %>!</h1>
<p>Bugun <%= new Date().toLocaleDateString() %></p>
<% if (user.isVIP) { %>
  <span class="vip-badge">VIP Member</span>
<% } %>
```

### Real hayot misoli:

Tasavvur qiling, **Instagram** saytini:

- **Static**: Har bir user uchun alohida HTML fayl yaratish (impossible!)
- **Dynamic**: 1 ta template + user ma'lumotlari = Personal page

```javascript
// Template + Data = Dynamic HTML
const userData = {
  name: 'Alibek',
  followers: 1250,
  posts: [
    { image: 'photo1.jpg', likes: 45 },
    { image: 'photo2.jpg', likes: 67 }
  ]
};

// Template Engine:
// template + userData = Alibek'ning personal Instagram page
```

## EJS (Embedded JavaScript)

**EJS** - eng mashhur va oson template engine. JavaScript kodini to'g'ridan-to'g'ri HTML ichida yozish mumkin.

### 1. EJS o'rnatish va setup:

```bash
npm install ejs
```

```javascript
// server.js
const express = require('express');
const path = require('path');

const app = express();

// Template engine setup
```

## Handlebars Template Engine

**Handlebars** - "logic-less" template engine. JS kod HTML ichida minimal bo'ladi.

### 1. Handlebars setup:

```bash
npm install express-handlebars
```

```javascript
// server-handlebars.js
const express = require('express');
const exphbs = require('express-handlebars');
const path = require('path');

const app = express();

// Handlebars setup
const hbs = exphbs.create({
  defaultLayout: 'main',
  extname: '.hbs',
  layoutsDir: path.join(__dirname, 'views/layouts/'),
  partialsDir: path.join(__dirname, 'views/partials/'),
  helpers: {
    // Custom helpers
    capitalize: function(str) {
      return str.charAt(0).toUpperCase() + str.slice(1);
    },
    formatDate: function(date) {
      return new Date(date).toLocaleDateString('uz-UZ');
    },
    eq: function(a, b) {
      return a === b;
    },
    gt: function(a, b) {
      return a > b;
    }
  }
});

app.engine('hbs', hbs.engine);
app.set('view engine', 'hbs');
app.set('views', './views');

// Static files
app.use(express.static('public'));

// Sample data

// Routes

  res.render('index', {
    posts: blogData.posts,
    site: blogData.site,

  res.render('post', {

    title: post.title
  });
});

app.listen(3000, () => {
  console.log('🎺 Handlebars Server: http://localhost:3000');
});
```

### 2. Template ma'lumotlarini formatlash:

```html
<!-- Helper functions in template -->
<%
  function formatNumber(num) {
    if (num >= 1000000) {
      return (num / 1000000).toFixed(1) + 'M';
    } else if (num >= 1000) {
      return (num / 1000).toFixed(1) + 'K';
    }
    return num.toString();
  }

  function timeAgo(date) {
    const now = new Date();
    const postDate = new Date(date);
    const diffMs = now - postDate;
    const diffDays = Math.floor(diffMs / (1000 * 60 * 60 * 24));

    if (diffDays === 0) return 'Bugun';
    if (diffDays === 1) return 'Kecha';
    if (diffDays < 7) return diffDays + ' kun oldin';
    if (diffDays < 30) return Math.floor(diffDays / 7) + ' hafta oldin';
    return Math.floor(diffDays / 30) + ' oy oldin';
  }

  function truncateText(text, length = 100) {
    if (text.length <= length) return text;
    return text.substr(0, length) + '...';
  }
%>

<!-- Usage in template -->
<% posts.forEach(function(post) { %>
  <article class="post">
    <h3><%= post.title %></h3>
    <p><%= truncateText(post.content, 150) %></p>
    <div class="meta">
      <span>👁️ <%= formatNumber(post.views) %></span>
      <span>📅 <%= timeAgo(post.createdAt) %></span>
      <span>❤️ <%= formatNumber(post.likes) %></span>
    </div>
  </article>
<% }) %>
```

```javascript
// helpers/handlebars-helpers.js
const moment = require('moment');

module.exports = {
  // Date formatting
  formatDate: (date, format = 'DD.MM.YYYY') => {
    return moment(date).format(format);
  },

  // Relative time
  timeAgo: (date) => {
    return moment(date).fromNow();
  },

  // Capitalize first letter
  capitalize: (str) => {
    if (!str || typeof str !== 'string') return '';
    return str.charAt(0).toUpperCase() + str.slice(1);
  },

  // Truncate text
  truncate: (text, length = 100) => {
    if (!text || text.length <= length) return text;
    return text.substring(0, length) + '...';
  },

  // Format numbers
  formatNumber: (num) => {
    if (num >= 1000000) {
      return (num / 1000000).toFixed(1) + 'M';
    } else if (num >= 1000) {
      return (num / 1000).toFixed(1) + 'K';
    }
    return num.toString();
  },

  // Equality check
  eq: (a, b) => a === b,

  // Greater than
  gt: (a, b) => a > b,

  // Addition
  add: (a, b) => a + b,

  // JSON stringify for JavaScript
  json: (obj) => JSON.stringify(obj),

  // Generate range for pagination
  range: (start, end) => {
    const result = [];
    for (let i = start; i <= end; i++) {
      result.push(i);
    }
    return result;
  }
};
```

```javascript
// server.js
const express = require('express');

const blogData = require('./data/blog-data');

app.engine('hbs', engine({
  defaultLayout: 'main',
  extname: '.hbs',
  layoutsDir: path.join(__dirname, 'views/layouts/'),
  partialsDir: path.join(__dirname, 'views/partials/'),
  helpers: helpers

app.set('views', './views');

// Middleware
app.use(express.static('public'));
app.use(express.json());

// Routes
app.get('/', (req, res) => {
  const featuredPosts = blogData.getFeaturedPosts();

  res.render('index', {
    title: 'Bosh sahifa',
    posts: blogData.posts,
    featuredPosts: featuredPosts,
    categories: blogData.categories,
    tags: blogData.tags,
    site: blogData.siteInfo
  });
});

app.get('/post/:slug', (req, res) => {
  const post = blogData.getPostBySlug(req.params.slug);

  if (!post) {
    return res.status(404).render('404', {
      title: 'Sahifa topilmadi',
      site: blogData.siteInfo
    });
  }

  // Increment views (demo uchun)
  post.views++;

  // Related posts (bir xil category)
  const relatedPosts = blogData.posts
    .filter(p => p.category === post.category && p.id !== post.id)
    .slice(0, 3);

  res.render('post', {
    title: post.title,
    post: post,
    relatedPosts: relatedPosts,
    site: blogData.siteInfo
  });
});

app.get('/category/:category', (req, res) => {
  const categoryName = req.params.category;
  const posts = blogData.getPostsByCategory(categoryName);

  res.render('category', {
    title: `${categoryName} postlari`,
    posts: posts,
    categoryName: categoryName,
    site: blogData.siteInfo
  });
});

app.get('/tag/:tag', (req, res) => {
  const tagName = req.params.tag;
  const posts = blogData.getPostsByTag(tagName);

  res.render('tag', {
    title: `#${tagName} postlari`,
    posts: posts,
    tagName: tagName,
    site: blogData.siteInfo
  });
});

app.get('/about', (req, res) => {
  res.render('about', {
    title: 'Biz haqimizda',
    site: blogData.siteInfo
  });
});

// 404 handler
app.use('*', (req, res) => {
  res.status(404).render('404', {
    title: 'Sahifa topilmadi',
    site: blogData.siteInfo
  });
});

const PORT = process.env.PORT || 3000;
```
