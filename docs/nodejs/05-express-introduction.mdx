## Express.js nima va nega kerak?

**Express.js** - Node.js uchun minimal va flexible web application framework. Bu "built-in HTTP modulining ustki qavati" deb o'ylashingiz mumkin.

### Built-in HTTP vs Express.js:

**Oldingi darsda built-in HTTP bilan:**
```javascript
// 50+ qator kod
const server = http.createServer((req, res) => {
  const parsedUrl = url.parse(req.url, true);
  const pathname = parsedUrl.pathname;
  const method = req.method;

  // Manual routing
  if (pathname === '/api/users' && method === 'GET') {
    res.statusCode = 200;
    res.setHeader('Content-Type', 'application/json');
    res.end(JSON.stringify({ users: [] }));
  } else if (pathname === '/api/users' && method === 'POST') {
    // POST body parsing
    let body = '';
    req.on('data', chunk => body += chunk);
    req.on('end', () => {
      // Manual JSON parsing...
    });
  }
  // Va hokazo...
});
```

**Express.js bilan:**
```javascript
// 5 qator kod
const express = require('express');
const app = express();

app.use(express.json()); // Automatic JSON parsing

app.get('/api/users', (req, res) => {
  res.json({ users: [] });
});

app.post('/api/users', (req, res) => {
  console.log(req.body); // Parsed JSON ready!
  res.json({ success: true });
});

app.listen(3000, () => console.log('Server running!'));
```

- ✅ **Kam kod** - 10x kamroq kod yozish

## Express.js routing tizimi

**Route** - HTTP method + URL path + Handler function.

### 1. Basic routes:

```javascript
const express = require('express');
const app = express();

app.use(express.json());

// GET route
app.get('/', (req, res) => {
  res.json({ message: 'Bosh sahifa' });
});

// POST route
app.post('/users', (req, res) => {
  console.log('Yangi user:', req.body);
  res.json({ message: 'User yaratildi', data: req.body });
});

// PUT route
app.put('/users/:id', (req, res) => {
  const userId = req.params.id;
  console.log(`User ${userId} yangilanyapti:`, req.body);
  res.json({ message: `User ${userId} yangilandi` });
});

// DELETE route
app.delete('/users/:id', (req, res) => {
  const userId = req.params.id;
  res.json({ message: `User ${userId} o'chirildi` });
});

app.listen(3000, () => {
  console.log('🧪 Test qiling:');
  console.log('GET    http://localhost:3000/');
  console.log('POST   http://localhost:3000/users');
  console.log('PUT    http://localhost:3000/users/123');
  console.log('DELETE http://localhost:3000/users/123');
});
```

### 3. Query parameters:

```javascript
const express = require('express');
const app = express();

app.get('/search', (req, res) => {
  const {
    q,          // query string
    limit = 10, // default value
    page = 1,
    sort = 'name'
  } = req.query;

  res.json({
    search: {
      query: q,
      limit: parseInt(limit),
      page: parseInt(page),
      sort: sort
    },
    url: req.url,
    allQuery: req.query
  });
});

// Real search logic misoli
const products = [
  { id: 1, name: 'Laptop', price: 1200, category: 'electronics' },
  { id: 2, name: 'Phone', price: 800, category: 'electronics' },
  { id: 3, name: 'Book', price: 50, category: 'education' },
  { id: 4, name: 'Chair', price: 150, category: 'furniture' }
];

app.get('/products', (req, res) => {
  let result = [...products];

  // Search by name
  if (req.query.search) {
    const searchTerm = req.query.search.toLowerCase();
    result = result.filter(p =>
      p.name.toLowerCase().includes(searchTerm)
    );
  }

  // Filter by category
  if (req.query.category) {
    result = result.filter(p => p.category === req.query.category);
  }

  // Filter by price range
  if (req.query.minPrice) {
    result = result.filter(p => p.price >= parseInt(req.query.minPrice));
  }

  if (req.query.maxPrice) {
    result = result.filter(p => p.price <= parseInt(req.query.maxPrice));
  }

  // Sort
  if (req.query.sort) {
    const sortField = req.query.sort;
    result.sort((a, b) => {
      if (typeof a[sortField] === 'number') {
        return a[sortField] - b[sortField];
      }
      return a[sortField].localeCompare(b[sortField]);
    });
  }

  // Pagination
  const limit = parseInt(req.query.limit) || 10;
  const page = parseInt(req.query.page) || 1;
  const startIndex = (page - 1) * limit;
  const endIndex = startIndex + limit;

  const paginatedResult = result.slice(startIndex, endIndex);

  res.json({
    success: true,
    data: paginatedResult,
    pagination: {
      page,
      limit,
      total: result.length,
      totalPages: Math.ceil(result.length / limit)
    },
    filters: req.query
  });
});

app.listen(3000, () => {
  console.log('🧪 Query test'lari:');
  console.log('http://localhost:3000/search?q=nodejs&limit=5&page=2');
  console.log('http://localhost:3000/products?search=phone&category=electronics');
  console.log('http://localhost:3000/products?minPrice=100&maxPrice=1000&sort=price');
});
```

## Middleware tushunchasi

**Middleware** - request va response o'rtasida ishlaydigan function'lar. pipeline kabi ishlaydi:

```
Request → Middleware 1 → Middleware 2 → Route Handler → Response
```

### 1. Built-in middleware:

```javascript
const express = require('express');
const app = express();

// JSON body parser
app.use(express.json());

// URL-encoded data parser (forms)
app.use(express.urlencoded({ extended: true }));

// Static files
app.use(express.static('public'));

// Routes
app.get('/', (req, res) => {
  res.json({ message: 'Middleware'lar ishladi!' });
});

app.post('/submit', (req, res) => {
  res.json({
    message: 'Form ma\'lumoti olindi',
    body: req.body
  });
});

app.listen(3000);
```

### 2. Custom middleware:

```javascript
const express = require('express');
const app = express();

// Logger middleware
const customLogger = (req, res, next) => {
  const timestamp = new Date().toISOString();
  console.log(`📡 ${timestamp} - ${req.method} ${req.url}`);

  // IP address
  const ip = req.headers['x-forwarded-for'] || req.connection.remoteAddress;
  console.log(`📍 IP: ${ip}`);

  // User agent
  console.log(`🖥️ User-Agent: ${req.headers['user-agent']}`);

  next(); // Keyingi middleware'ga o'tish!
};

// Authentication middleware
const authenticateToken = (req, res, next) => {
  const authHeader = req.headers['authorization'];
  const token = authHeader && authHeader.split(' ')[1]; // Bearer TOKEN

  if (!token) {
    return res.status(401).json({
      success: false,
      message: 'Access token kerak!'
    });
  }

  // Token validation (simplified)
  if (token === 'valid-token') {
    req.user = { id: 1, name: 'Alibek', role: 'admin' };
    next();
  } else {
    res.status(403).json({
      success: false,
      message: 'Noto\'g\'ri token!'
    });
  }
};

// Request timing middleware
const requestTimer = (req, res, next) => {
  req.startTime = Date.now();

  // Response tugagach timing ni hisoblash
  res.on('finish', () => {
    const duration = Date.now() - req.startTime;
    console.log(`⏰ Request ${req.url} - ${duration}ms`);
  });

  next();
};

// Global middleware'lar
app.use(express.json());
app.use(customLogger);
app.use(requestTimer);

// Public route
app.get('/', (req, res) => {
  res.json({ message: 'Public route' });
});

// Protected route
app.get('/dashboard', authenticateToken, (req, res) => {
  res.json({
    message: 'Dashboard data',
    user: req.user
  });
});

app.listen(3000, () => {
  console.log('🧪 Middleware test'lari:');
  console.log('Public:    http://localhost:3000/');
  console.log('Protected: http://localhost:3000/dashboard');
  console.log('Header:    Authorization: Bearer valid-token');
});
```

### 3. Error handling middleware:

```javascript
const express = require('express');
const app = express();

app.use(express.json());

// Routes
app.get('/', (req, res) => {
  res.json({ message: 'OK' });
});

app.get('/error-test', (req, res, next) => {
  // Xatolikni simulate qilish
  const error = new Error('Bu test xatoligi!');
  error.status = 500;
  next(error); // Error middleware'ga jo'natish
});

app.get('/async-error', async (req, res, next) => {
  try {
    // Async operation simulate
    await new Promise((resolve, reject) => {
      setTimeout(() => reject(new Error('Async xatolik!')), 1000);
    });
  } catch (error) {
    next(error);
  }
});

// 404 handler (oxirgi route'dan keyin)
app.use('*', (req, res) => {
  res.status(404).json({
    success: false,
    message: 'Route topilmadi',
    path: req.originalUrl
  });
});

// Error handling middleware (eng oxirda)
app.use((error, req, res, next) => {
  console.error('💥 Error:', error.message);

  // Error stack'ni development'da ko'rsatish
  const isDevelopment = process.env.NODE_ENV === 'development';

  res.status(error.status || 500).json({
    success: false,
    message: error.message,
    ...(isDevelopment && { stack: error.stack })
  });
});

app.listen(3000, () => {
  console.log('🧪 Error handling test'lari:');
  console.log('http://localhost:3000/error-test');
  console.log('http://localhost:3000/async-error');
  console.log('http://localhost:3000/not-found-route');
});
```

## Built-in HTTP'dan Express'ga migration

Oldingi darsda yaratgan Blog API'ni Express'ga o'tkazamiz:

### 1. Express version - Posts routes:

```javascript
// routes/posts.js
const express = require('express');
const router = express.Router();

// In-memory database (demo uchun)
let posts = [
  {
    id: 1,
    title: 'Node.js bilan web server',
    content: 'Node.js bilan web server yaratish...',
    author: 'Alibek',
    tags: ['nodejs', 'backend'],
    createdAt: '2026-03-29T10:00:00Z',
    views: 0
  }
];

let nextId = 2;

// GET /api/posts - Barcha postlar
router.get('/', (req, res) => {
  const { q, limit = 10, page = 1 } = req.query;

  let result = [...posts];

  // Search
  if (q) {
    const searchTerm = q.toLowerCase();
    result = result.filter(post =>
      post.title.toLowerCase().includes(searchTerm) ||
      post.content.toLowerCase().includes(searchTerm) ||
      post.tags.some(tag => tag.toLowerCase().includes(searchTerm))
    );
  }

  // Pagination
  const limitNum = parseInt(limit);
  const pageNum = parseInt(page);
  const startIndex = (pageNum - 1) * limitNum;
  const endIndex = startIndex + limitNum;
  const paginatedPosts = result.slice(startIndex, endIndex);

  res.json({
    success: true,
    data: paginatedPosts,
    pagination: {
      page: pageNum,
      limit: limitNum,
      total: result.length,
      totalPages: Math.ceil(result.length / limitNum)
    }
  });
});

// GET /api/posts/:id - Bitta post
router.get('/:id', (req, res) => {
  const postId = parseInt(req.params.id);
  const post = posts.find(p => p.id === postId);

  if (!post) {
    return res.status(404).json({
      success: false,
      message: 'Post topilmadi'
    });
  }

  // Views increment
  post.views++;

  res.json({
    success: true,
    data: post
  });
});

// POST /api/posts - Yangi post
router.post('/', (req, res) => {
  const { title, content, author, tags } = req.body;

  // Validation
  if (!title || !content || !author) {
    return res.status(400).json({
      success: false,
      message: 'Title, content va author majburiy!'
    });
  }

  const newPost = {
    id: nextId++,
    title,
    content,
    author,
    tags: tags || [],
    createdAt: new Date().toISOString(),
    views: 0
  };

  posts.push(newPost);

  res.status(201).json({
    success: true,
    message: 'Post yaratildi',
    data: newPost
  });
});

// PUT /api/posts/:id - Post yangilash
router.put('/:id', (req, res) => {
  const postId = parseInt(req.params.id);
  const postIndex = posts.findIndex(p => p.id === postId);

  if (postIndex === -1) {
    return res.status(404).json({
      success: false,
      message: 'Post topilmadi'
    });
  }

  posts[postIndex] = {
    ...posts[postIndex],
    ...req.body,
    id: postId, // ID o'zgarmaydi
    updatedAt: new Date().toISOString()
  };

  res.json({
    success: true,
    message: 'Post yangilandi',
    data: posts[postIndex]
  });
});

// DELETE /api/posts/:id - Post o'chirish
router.delete('/:id', (req, res) => {
  const postId = parseInt(req.params.id);
  const postIndex = posts.findIndex(p => p.id === postId);

  if (postIndex === -1) {
    return res.status(404).json({
      success: false,
      message: 'Post topilmadi'
    });
  }

  const deletedPost = posts[postIndex];
  posts.splice(postIndex, 1);

  res.json({
    success: true,
    message: 'Post o\'chirildi',
    data: deletedPost
  });
});

module.exports = router;
```

```javascript
// middleware/validation.js
const validateProduct = (req, res, next) => {
  const { name, description, price, category, stock } = req.body;

  const errors = [];

  if (!name || name.trim().length < 2) {
    errors.push('Product name kamida 2 ta belgi bo\'lishi kerak');
  }

  if (!description || description.trim().length < 10) {
    errors.push('Description kamida 10 ta belgi bo\'lishi kerak');
  }

  if (!price || price < 0) {
    errors.push('Price 0dan katta bo\'lishi kerak');
  }

  if (!category) {
    errors.push('Category majburiy');
  }

  if (stock === undefined || stock < 0) {
    errors.push('Stock 0 yoki katta bo\'lishi kerak');
  }

  if (errors.length > 0) {
    return res.status(400).json({
      success: false,
      message: 'Validation xatoliklari',
      errors
    });
  }

  next();
};

const validateOrder = (req, res, next) => {
  const { customerName, customerEmail, items } = req.body;

  const errors = [];

  if (!customerName || customerName.trim().length < 2) {
    errors.push('Customer name majburiy');
  }

  if (!customerEmail || !/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(customerEmail)) {
    errors.push('To\'g\'ri email kiriting');
  }

  if (!items || !Array.isArray(items) || items.length === 0) {
    errors.push('Order items majburiy');
  }

  if (errors.length > 0) {
    return res.status(400).json({
      success: false,
      message: 'Validation xatoliklari',
      errors
    });
  }

  next();
};

module.exports = {
  validateProduct,
  validateOrder
};
```

```javascript
// controllers/productController.js
const { products, categories, getNextProductId } = require('../models/data');

const getAllProducts = (req, res) => {
  const {
    category,
    search,
    minPrice,
    maxPrice,
    sort = 'name',
    limit = 10,
    page = 1
  } = req.query;

  let result = [...products];

  // Filter by category
  if (category) {
    result = result.filter(p => p.category === parseInt(category));
  }

  // Search by name or description
  if (search) {
    const searchTerm = search.toLowerCase();
    result = result.filter(p =>
      p.name.toLowerCase().includes(searchTerm) ||
      p.description.toLowerCase().includes(searchTerm)
    );
  }

  // Price range filter
  if (minPrice) {
    result = result.filter(p => p.price >= parseFloat(minPrice));
  }

  if (maxPrice) {
    result = result.filter(p => p.price <= parseFloat(maxPrice));
  }

  // Sorting
  result.sort((a, b) => {
    switch (sort) {
      case 'price_asc':
        return a.price - b.price;
      case 'price_desc':
        return b.price - a.price;
      case 'name':
        return a.name.localeCompare(b.name);
      case 'newest':
        return new Date(b.createdAt) - new Date(a.createdAt);
      default:
        return a.name.localeCompare(b.name);
    }
  });

  // Pagination
  const limitNum = parseInt(limit);
  const pageNum = parseInt(page);
  const startIndex = (pageNum - 1) * limitNum;
  const endIndex = startIndex + limitNum;
  const paginatedProducts = result.slice(startIndex, endIndex);

  // Category ma'lumotlarini qo'shish
  const productsWithCategory = paginatedProducts.map(product => ({
    ...product,
    categoryName: categories.find(c => c.id === product.category)?.name
  }));

  res.json({
    success: true,
    data: productsWithCategory,
    pagination: {
      page: pageNum,
      limit: limitNum,
      total: result.length,
      totalPages: Math.ceil(result.length / limitNum)
    },
    filters: { category, search, minPrice, maxPrice, sort }
  });
};

const getProductById = (req, res) => {
  const productId = parseInt(req.params.id);
  const product = products.find(p => p.id === productId);

  if (!product) {
    return res.status(404).json({
      success: false,
      message: 'Product topilmadi'
    });
  }

  // Category ma'lumoti qo'shish
  const categoryData = categories.find(c => c.id === product.category);

  res.json({
    success: true,
    data: {
      ...product,
      categoryName: categoryData?.name,
      categoryDescription: categoryData?.description
    }
  });
};

const createProduct = (req, res) => {
  const { name, description, price, category, stock, imageUrl } = req.body;

  // Category mavjudligini tekshirish
  const categoryExists = categories.find(c => c.id === parseInt(category));
  if (!categoryExists) {
    return res.status(400).json({
      success: false,
      message: 'Category topilmadi'
    });
  }

  const newProduct = {
    id: getNextProductId(),
    name: name.trim(),
    description: description.trim(),
    price: parseFloat(price),
    category: parseInt(category),
    stock: parseInt(stock),
    imageUrl: imageUrl || '/images/default-product.jpg',
    createdAt: new Date().toISOString()
  };

  products.push(newProduct);

  res.status(201).json({
    success: true,
    message: 'Product yaratildi',
    data: newProduct
  });
};

const updateProduct = (req, res) => {
  const productId = parseInt(req.params.id);
  const productIndex = products.findIndex(p => p.id === productId);

  if (productIndex === -1) {
    return res.status(404).json({
      success: false,
      message: 'Product topilmadi'
    });
  }

  // Category check (agar category yangilanayotgan bo'lsa)
  if (req.body.category) {
    const categoryExists = categories.find(c => c.id === parseInt(req.body.category));
    if (!categoryExists) {
      return res.status(400).json({
        success: false,
        message: 'Category topilmadi'
      });
    }
  }

  products[productIndex] = {
    ...products[productIndex],
    ...req.body,
    id: productId,
    updatedAt: new Date().toISOString()
  };

  res.json({
    success: true,
    message: 'Product yangilandi',
    data: products[productIndex]
  });
};

const deleteProduct = (req, res) => {
  const productId = parseInt(req.params.id);
  const productIndex = products.findIndex(p => p.id === productId);

  if (productIndex === -1) {
    return res.status(404).json({
      success: false,
      message: 'Product topilmadi'
    });
  }

  const deletedProduct = products[productIndex];
  products.splice(productIndex, 1);

  res.json({
    success: true,
    message: 'Product o\'chirildi',
    data: deletedProduct
  });
};

module.exports = {
  getAllProducts,
  getProductById,
  createProduct,
  updateProduct,
  deleteProduct
};
```
