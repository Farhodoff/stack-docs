```javascript
// src/models/index.js
const fs = require('fs').promises;
const path = require('path');
const { v4: uuidv4 } = require('uuid');

class Database {
  constructor() {
    this.dataPath = path.join(__dirname, '../../data');
    this.ensureDataDirectory();
  }

  async ensureDataDirectory() {
    try {
      await fs.mkdir(this.dataPath, { recursive: true });
    } catch (error) {
      // Directory might already exist
    }
  }

  async read(collection) {
    try {
      const filePath = path.join(this.dataPath, `${collection}.json`);
      const data = await fs.readFile(filePath, 'utf8');
      return JSON.parse(data);
    } catch (error) {
      if (error.code === 'ENOENT') {
        return [];
      }
      throw error;
    }
  }

  async write(collection, data) {
    const filePath = path.join(this.dataPath, `${collection}.json`);
    await fs.writeFile(filePath, JSON.stringify(data, null, 2), 'utf8');
  }

  async findById(collection, id) {
    const data = await this.read(collection);
    return data.find(item => item.id === id);
  }

  async findOne(collection, query) {
    const data = await this.read(collection);
    return data.find(item => {
      return Object.keys(query).every(key => item[key] === query[key]);
    });
  }

  async findMany(collection, query = {}, options = {}) {
    let data = await this.read(collection);

    // Filter
    if (Object.keys(query).length > 0) {
      data = data.filter(item => {
        return Object.keys(query).every(key => {
          if (Array.isArray(query[key])) {
            return query[key].includes(item[key]);
          }
          return item[key] === query[key];
        });
      });
    }

    // Sort
    if (options.sort) {
      const [field, direction] = options.sort.split(':');
      data.sort((a, b) => {
        if (direction === 'desc') {
          return new Date(b[field]) - new Date(a[field]);
        }
        return new Date(a[field]) - new Date(b[field]);
      });
    }

    // Pagination
    if (options.limit) {
      const start = (options.page - 1) * options.limit || 0;
      data = data.slice(start, start + options.limit);
    }

    return data;
  }

  async create(collection, item) {
    const data = await this.read(collection);
    const newItem = {
      id: uuidv4(),
      createdAt: new Date().toISOString(),
      updatedAt: new Date().toISOString(),
      ...item
    };

    data.push(newItem);
    await this.write(collection, data);
    return newItem;
  }

  async update(collection, id, updates) {
    const data = await this.read(collection);
    const index = data.findIndex(item => item.id === id);

    if (index === -1) {
      throw new Error(`Item with id ${id} not found in ${collection}`);
    }

    data[index] = {
      ...data[index],
      ...updates,
      id, // Preserve original ID
      updatedAt: new Date().toISOString()
    };

    await this.write(collection, data);
    return data[index];
  }

  async delete(collection, id) {
    const data = await this.read(collection);
    const index = data.findIndex(item => item.id === id);

    if (index === -1) {
      throw new Error(`Item with id ${id} not found in ${collection}`);
    }

    const deletedItem = data.splice(index, 1)[0];
    await this.write(collection, data);
    return deletedItem;
  }

  async count(collection, query = {}) {
    const data = await this.findMany(collection, query);
    return data.length;
  }
}

module.exports = new Database();
```

// src/models/User.js
const bcrypt = require('bcrypt');
const database = require('./index');

class User {
  static collection = 'users';

  static async create(userData) {
    const { name, email, password, role = 'user' } = userData;

    // Check if user already exists
    const existingUser = await database.findOne(this.collection, { email });
    if (existingUser) {
      throw new Error('User with this email already exists');
    }

    // Hash password
    const saltRounds = 12;
    const hashedPassword = await bcrypt.hash(password, saltRounds);

    const user = await database.create(this.collection, {
      name,
      email: email.toLowerCase(),
      password: hashedPassword,
      role,
      isActive: true,
      lastLogin: null,
      avatar: null,
      bio: null,
      social: {
        twitter: null,
        github: null,
        website: null
      },
      preferences: {
        emailNotifications: true,
        theme: 'light'
      }
    });

    // Remove password from returned object
    delete user.password;
    return user;
  }

  static async findById(id) {
    const user = await database.findById(this.collection, id);
    if (user) {
      delete user.password;
    }
    return user;
  }

  static async findByEmail(email) {
    return database.findOne(this.collection, { email: email.toLowerCase() });
  }

  static async findAll(options = {}) {
    const users = await database.findMany(this.collection, {}, options);
    return users.map(user => {
      delete user.password;
      return user;
    });
  }

  static async updateById(id, updates) {
    // Don't allow updating password through this method
    delete updates.password;
    delete updates.id;

    const user = await database.update(this.collection, id, updates);
    delete user.password;
    return user;
  }

  static async updatePassword(id, newPassword) {
    const saltRounds = 12;
    const hashedPassword = await bcrypt.hash(newPassword, saltRounds);

    await database.update(this.collection, id, { password: hashedPassword });
  }

  static async deleteById(id) {
    return database.delete(this.collection, id);
  }

  static async verifyPassword(user, password) {
    return bcrypt.compare(password, user.password);
  }

  static async updateLastLogin(id) {
    return database.update(this.collection, id, { lastLogin: new Date().toISOString() });
  }

  static async getStats() {
    const users = await database.read(this.collection);
    const now = new Date();
    const sevenDaysAgo = new Date(now.getTime() - 7 * 24 * 60 * 60 * 1000);

    return {
      total: users.length,
      active: users.filter(user => user.isActive).length,
      admins: users.filter(user => user.role === 'admin').length,
      newThisWeek: users.filter(user => new Date(user.createdAt) > sevenDaysAgo).length
    };
  }
}

module.exports = User;

// src/models/Post.js
const database = require('./index');
const slug = require('slug');
const { marked } = require('marked');

class Post {
  static collection = 'posts';

  static async create(postData) {
    const { title, content, excerpt, authorId, categoryId, tags = [], status = 'draft', featured = false } = postData;

    // Generate slug from title
    const postSlug = slug(title, { lower: true });

    // Check if slug already exists
    let finalSlug = postSlug;
    let counter = 1;
    while (await database.findOne(this.collection, { slug: finalSlug })) {
      finalSlug = `${postSlug}-${counter}`;
      counter++;
    }

    const post = await database.create(this.collection, {
      title,
      slug: finalSlug,
      content,
      contentHtml: marked(content), // Convert markdown to HTML
      excerpt: excerpt || this.generateExcerpt(content),
      authorId,
      categoryId,
      tags,
      status, // draft, published, archived
      featured,
      views: 0,
      likes: 0,
      commentsCount: 0,
      featuredImage: null,
      seo: {
        metaTitle: title,
        metaDescription: excerpt,
        keywords: tags.join(', ')
      },
      publishedAt: status === 'published' ? new Date().toISOString() : null
    });

    return post;
  }

  static generateExcerpt(content, maxLength = 160) {
    // Remove markdown and get plain text
    const plainText = content
      .replace(/#{1,6}\s+/g, '') // Remove headers
      .replace(/\*\*(.*?)\*\*/g, '$1') // Remove bold
      .replace(/\*(.*?)\*/g, '$1') // Remove italic
      .replace(/\[([^\]]+)\]\([^)]+\)/g, '$1') // Remove links
      .replace(/`([^`]+)`/g, '$1'); // Remove code

    if (plainText.length <= maxLength) {
      return plainText;
    }

    return plainText.substring(0, maxLength).replace(/\s+\w*$/, '') + '...';
  }

  static async findById(id) {
    return database.findById(this.collection, id);
  }

  static async findBySlug(slug) {
    return database.findOne(this.collection, { slug });
  }

  static async findAll(options = {}) {
    const query = {};

    if (options.status) {
      query.status = options.status;
    }

    if (options.categoryId) {
      query.categoryId = options.categoryId;
    }

    if (options.authorId) {
      query.authorId = options.authorId;
    }

    const posts = await database.findMany(this.collection, query, {
      sort: 'createdAt:desc',
      ...options
    });

    return posts;
  }

  static async findPublished(options = {}) {
    return this.findAll({ ...options, status: 'published' });
  }

  static async findFeatured(limit = 3) {
    const posts = await database.findMany(this.collection,
      { status: 'published', featured: true },
      { sort: 'publishedAt:desc', limit }
    );
    return posts;
  }

  static async searchPosts(searchTerm, options = {}) {
    const allPosts = await database.findMany(this.collection, { status: 'published' });

    const searchResults = allPosts.filter(post => {
      const searchIn = `${post.title} ${post.excerpt} ${post.content} ${post.tags.join(' ')}`.toLowerCase();
      return searchIn.includes(searchTerm.toLowerCase());
    });

    // Sort by relevance (title matches first)
    searchResults.sort((a, b) => {
      const aInTitle = a.title.toLowerCase().includes(searchTerm.toLowerCase());
      const bInTitle = b.title.toLowerCase().includes(searchTerm.toLowerCase());

      if (aInTitle && !bInTitle) return -1;
      if (!aInTitle && bInTitle) return 1;

      return new Date(b.publishedAt) - new Date(a.publishedAt);
    });

    return searchResults;
  }

  static async updateById(id, updates) {
    delete updates.id;
    delete updates.createdAt;
    delete updates.views; // Views updated separately

    // Regenerate HTML if content changed
    if (updates.content) {
      updates.contentHtml = marked(updates.content);
    }

    // Update slug if title changed
    if (updates.title) {
      const currentPost = await database.findById(this.collection, id);
      if (currentPost && updates.title !== currentPost.title) {
        let newSlug = slug(updates.title, { lower: true });

        // Check for slug conflicts
        let finalSlug = newSlug;
        let counter = 1;
        const existingPost = await database.findOne(this.collection, { slug: finalSlug });

        while (existingPost && existingPost.id !== id) {
          finalSlug = `${newSlug}-${counter}`;
          counter++;
          const conflictPost = await database.findOne(this.collection, { slug: finalSlug });
          if (!conflictPost || conflictPost.id === id) break;
        }

        updates.slug = finalSlug;
      }
    }

    // Update publishedAt if status changed to published
    if (updates.status === 'published') {
      const currentPost = await database.findById(this.collection, id);
      if (currentPost && currentPost.status !== 'published') {
        updates.publishedAt = new Date().toISOString();
      }
    }

    return database.update(this.collection, id, updates);
  }

  static async deleteById(id) {
    return database.delete(this.collection, id);
  }

  static async incrementViews(id) {
    const post = await database.findById(this.collection, id);
    if (post) {
      return database.update(this.collection, id, { views: post.views + 1 });
    }
    return null;
  }

  static async updateCommentsCount(id) {
    const Comment = require('./Comment');
    const count = await Comment.countByPost(id);
    return database.update(this.collection, id, { commentsCount: count });
  }

  static async getStats() {
    const posts = await database.read(this.collection);
    const now = new Date();
    const sevenDaysAgo = new Date(now.getTime() - 7 * 24 * 60 * 60 * 1000);

    return {
      total: posts.length,
      published: posts.filter(post => post.status === 'published').length,
      draft: posts.filter(post => post.status === 'draft').length,
      featured: posts.filter(post => post.featured).length,
      newThisWeek: posts.filter(post => new Date(post.createdAt) > sevenDaysAgo).length,
      totalViews: posts.reduce((sum, post) => sum + (post.views || 0), 0)
    };
  }

  static async getPopular(limit = 5) {
    const posts = await database.findMany(this.collection,
      { status: 'published' },
      { sort: 'views:desc', limit }
    );
    return posts;
  }

  static async getByTag(tag) {
    const allPosts = await database.findMany(this.collection, { status: 'published' });
    return allPosts.filter(post => post.tags.includes(tag));
  }
}

module.exports = Post;

// src/models/Comment.js
const database = require('./index');

class Comment {
  static collection = 'comments';

  static async create(commentData) {
    const { content, authorName, authorEmail, postId, parentId = null } = commentData;

    return database.create(this.collection, {
      content,
      authorName,
      authorEmail,
      postId,
      parentId,
      status: 'pending', // pending, approved, rejected
      likes: 0,
      replies: []
    });
  }

  static async findByPost(postId, status = 'approved') {
    const comments = await database.findMany(this.collection,
      { postId, status },
      { sort: 'createdAt:asc' }
    );

    // Build comment tree
    const commentTree = [];
    const commentMap = new Map();

    comments.forEach(comment => {
      commentMap.set(comment.id, { ...comment, replies: [] });
    });

    comments.forEach(comment => {
      if (comment.parentId && commentMap.has(comment.parentId)) {
        commentMap.get(comment.parentId).replies.push(commentMap.get(comment.id));
      } else {
        commentTree.push(commentMap.get(comment.id));
      }
    });

    return commentTree;
  }

  static async updateStatus(id, status) {
    return database.update(this.collection, id, { status });
  }

  static async countByPost(postId) {
    return database.count(this.collection, { postId, status: 'approved' });
  }

  static async findAll(options = {}) {
    return database.findMany(this.collection, {}, {
      sort: 'createdAt:desc',
      ...options
    });
  }

  static async deleteById(id) {
    return database.delete(this.collection, id);
  }
}

module.exports = Comment;

// src/services/authService.js
const jwt = require('jsonwebtoken');
const User = require('../models/User');
const config = require('../config/environment');

class AuthService {
  static generateTokens(user) {
    const payload = {
      id: user.id,
      email: user.email,
      role: user.role,
      name: user.name
    };

    const accessToken = jwt.sign(payload, config.jwt.secret, {
      expiresIn: config.jwt.expiresIn,
      algorithm: 'HS256'
    });

    const refreshToken = jwt.sign(
      { id: user.id, tokenVersion: Date.now() },
      config.jwt.refreshSecret,
      { expiresIn: '7d', algorithm: 'HS256' }
    );

    return { accessToken, refreshToken };
  }

  static verifyToken(token) {
    try {
      return jwt.verify(token, config.jwt.secret);
    } catch (error) {
      throw new Error('Invalid token');
    }
  }

  static verifyRefreshToken(token) {
    try {
      return jwt.verify(token, config.jwt.refreshSecret);
    } catch (error) {
      throw new Error('Invalid refresh token');
    }
  }

  static async login(email, password) {
    // Find user with password
    const user = await User.findByEmail(email);

    if (!user) {
      throw new Error('Invalid email or password');
    }

    if (!user.isActive) {
      throw new Error('Account is deactivated');
    }

    // Verify password
    const isValidPassword = await User.verifyPassword(user, password);

    if (!isValidPassword) {
      throw new Error('Invalid email or password');
    }

    // Update last login
    await User.updateLastLogin(user.id);

    // Generate tokens
    const tokens = this.generateTokens(user);

    // Remove password from user object
    delete user.password;

    return {
      user,
      ...tokens
    };
  }

  static async register(userData) {
    const { name, email, password, confirmPassword } = userData;

    if (password !== confirmPassword) {
      throw new Error('Passwords do not match');
    }

    if (password.length < 8) {
      throw new Error('Password must be at least 8 characters long');
    }

    try {
      const user = await User.create({ name, email, password });
      const tokens = this.generateTokens(user);

      return {
        user,
        ...tokens
      };
    } catch (error) {
      if (error.message.includes('already exists')) {
        throw new Error('Email is already registered');
      }
      throw error;
    }
  }

  static async refreshAccessToken(refreshToken) {
    const decoded = this.verifyRefreshToken(refreshToken);
    const user = await User.findById(decoded.id);

    if (!user) {
      throw new Error('User not found');
    }

    const tokens = this.generateTokens(user);
    return tokens;
  }

  static async changePassword(userId, currentPassword, newPassword) {
    const user = await User.findByEmail(await User.findById(userId).email);

    if (!user) {
      throw new Error('User not found');
    }

    const isValidPassword = await User.verifyPassword(user, currentPassword);

    if (!isValidPassword) {
      throw new Error('Current password is incorrect');
    }

    if (newPassword.length < 8) {
      throw new Error('New password must be at least 8 characters long');
    }

    await User.updatePassword(userId, newPassword);
  }
}

module.exports = AuthService;

// src/middleware/auth.js
const AuthService = require('../services/authService');
const User = require('../models/User');

const authenticateToken = async (req, res, next) => {
  try {
    const authHeader = req.headers.authorization;
    const token = authHeader && authHeader.split(' ')[1]; // Bearer TOKEN

    if (!token) {
      return res.status(401).json({
        success: false,
        message: 'Access token required',
        code: 'TOKEN_MISSING'
      });
    }

    const decoded = AuthService.verifyToken(token);
    const user = await User.findById(decoded.id);

    if (!user) {
      return res.status(401).json({
        success: false,
        message: 'User not found',
        code: 'USER_NOT_FOUND'
      });
    }

    if (!user.isActive) {
      return res.status(401).json({
        success: false,
        message: 'Account is deactivated',
        code: 'ACCOUNT_DEACTIVATED'
      });
    }

    req.user = user;
    req.token = decoded;
    next();
  } catch (error) {
    return res.status(403).json({
      success: false,
      message: 'Invalid or expired token',
      code: 'TOKEN_INVALID'
    });
  }
};

const requireRole = (roles) => {
  return (req, res, next) => {
    if (!req.user) {
      return res.status(401).json({
        success: false,
        message: 'Authentication required',
        code: 'AUTH_REQUIRED'
      });
    }

    if (!roles.includes(req.user.role)) {
      return res.status(403).json({
        success: false,
        message: 'Insufficient permissions',
        code: 'INSUFFICIENT_PERMISSIONS',
        required: roles,
        current: req.user.role
      });
    }

    next();
  };
};

const optionalAuth = async (req, res, next) => {
  try {
    const authHeader = req.headers.authorization;
    const token = authHeader && authHeader.split(' ')[1];

    if (token) {
      const decoded = AuthService.verifyToken(token);
      const user = await User.findById(decoded.id);

      if (user && user.isActive) {
        req.user = user;
        req.token = decoded;
      }
    }
  } catch (error) {
    // Ignore auth errors for optional auth
  }

  next();
};

module.exports = {
  authenticateToken,
  requireRole,
  optionalAuth,
  requireAuth: authenticateToken,
  requireAdmin: requireRole(['admin']),
  requireAuthor: requireRole(['admin', 'author'])
};
