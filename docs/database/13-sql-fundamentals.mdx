## 2. CRUD Operations - Ma'lumot Boshqaruvi 🔍

### SELECT - Ma'lumotni O'qish

```sql
-- 1. Oddiy SELECT
SELECT * FROM users;

SELECT id, username, email, first_name || ' ' || last_name as full_name
FROM users
WHERE is_active = true;

-- 2. WHERE bilan filterlash
SELECT name, price, stock_quantity
FROM products
WHERE price BETWEEN 100 AND 500
  AND stock_quantity > 20
  AND status = 'active';

-- 3. LIKE operatori bilan qidirish
SELECT name, description
FROM products
WHERE name ILIKE '%iphone%'
   OR description ILIKE '%smartphone%';

-- 4. IN va NOT IN operatorlari
SELECT name, price, category_id
FROM products
WHERE category_id IN (1, 6, 7)  -- Electronics, Smartphones, Laptops
  AND status NOT IN ('inactive', 'out_of_stock');

-- 5. NULL qiymatlar bilan ishlash
SELECT name, price, sale_price,
       CASE
         WHEN sale_price IS NOT NULL THEN sale_price
         ELSE price
       END as effective_price
FROM products
WHERE sale_price IS NOT NULL;  -- Chegirmadagi mahsulotlar

-- 6. ORDER BY va LIMIT
SELECT name, price, stock_quantity
FROM products
WHERE status = 'active'
ORDER BY price DESC, stock_quantity ASC
LIMIT 10 OFFSET 0;  -- Birinchi 10 ta

-- 7. DISTINCT - takrorlanmaydigan qiymatlar
SELECT DISTINCT category_id, status
FROM products
ORDER BY category_id;

-- 8. Complex filtering
SELECT p.name, p.price, c.name as category_name
FROM products p
JOIN categories c ON p.category_id = c.id
WHERE p.price > (
    SELECT AVG(price) FROM products WHERE status = 'active'
)
ORDER BY p.price DESC;
```

```sql
INSERT INTO categories (name, slug, description)
VALUES ('Electronics', 'electronics', 'Updated electronics description')
ON CONFLICT (slug)
DO UPDATE SET
    description = EXCLUDED.description,
    updated_at = NOW();

-- 5. INSERT ... SELECT (ma'lumotni ko'chirish)
-- Archived products jadvalini yaratish
CREATE TABLE archived_products AS
SELECT * FROM products WHERE 1=0;  -- Struktura'ni ko'chirish, ma'lumot emas

INSERT INTO archived_products
SELECT * FROM products
WHERE status = 'inactive'
  AND updated_at < NOW() - INTERVAL '6 months';

-- 6. Bulk INSERT with error handling
DO $$
DECLARE
    product_record RECORD;
BEGIN
    FOR product_record IN
        SELECT unnest(ARRAY['Product 1', 'Product 2', 'Product 3']) as name,
               unnest(ARRAY['prod-1', 'prod-2', 'prod-3']) as slug,
               unnest(ARRAY[29.99, 39.99, 49.99]) as price
    LOOP
        BEGIN
            INSERT INTO products (name, slug, price, cost_price, sku, stock_quantity, category_id)
            VALUES (
                product_record.name,
                product_record.slug,
                product_record.price,
                product_record.price * 0.7,
                'BULK-' || upper(product_record.slug),
                10,
                1
            );
        EXCEPTION WHEN unique_violation THEN
            -- Ignore duplicate errors
            RAISE NOTICE 'Product % already exists', product_record.slug;
        END;
    END LOOP;
END $$;
```

### UPDATE - Ma'lumotni Yangilash

```sql
-- 1. Oddiy UPDATE
UPDATE products
SET price = 999.99, updated_at = NOW()
WHERE id = 1;

-- 2. Shartli UPDATE
UPDATE products
SET status = 'out_of_stock'
WHERE stock_quantity = 0;

-- 3. Mathematical operations
UPDATE products
SET price = price * 1.1  -- 10% narx oshirish
WHERE category_id = 1
  AND status = 'active';

-- 4. JOIN bilan UPDATE
UPDATE products p
SET low_stock_threshold = 10
FROM categories c
WHERE p.category_id = c.id
  AND c.name = 'Electronics';

-- 5. Subquery bilan UPDATE
UPDATE products
SET featured = true
WHERE id IN (
    SELECT DISTINCT product_id
    FROM product_reviews
    WHERE rating >= 4
      AND is_approved = true
    GROUP BY product_id
    HAVING COUNT(*) >= 3
);

-- 6. CASE statement bilan conditional update
UPDATE products
SET status = CASE
    WHEN stock_quantity = 0 THEN 'out_of_stock'
    WHEN stock_quantity <= low_stock_threshold THEN 'low_stock'
    ELSE 'active'
END,
updated_at = NOW()
WHERE status != 'inactive';

-- 7. JSON field yangilash
UPDATE orders
SET shipping_address = shipping_address || '{"delivery_notes": "Leave at door"}'
WHERE id = 1;

-- 8. UPDATE ... RETURNING
UPDATE products
SET sale_price = price * 0.8  -- 20% chegirma
WHERE featured = true
  AND sale_price IS NULL
RETURNING id, name, price, sale_price;
```

### DELETE - Ma'lumotni O'chirish

```sql
-- 1. Oddiy DELETE
DELETE FROM product_reviews
WHERE rating < 2 AND is_approved = false;

-- 2. Shartli DELETE
DELETE FROM orders
WHERE status = 'cancelled'
  AND created_at < NOW() - INTERVAL '1 year';

-- 3. JOIN bilan DELETE
DELETE FROM products p
USING categories c
WHERE p.category_id = c.id
  AND c.name = 'Discontinued Category'
  AND p.status = 'inactive';

-- 4. Subquery bilan DELETE
DELETE FROM users
WHERE id NOT IN (
    SELECT DISTINCT user_id
    FROM orders
    WHERE user_id IS NOT NULL
      AND created_at > NOW() - INTERVAL '2 years'
);

-- 5. TRUNCATE (barcha ma'lumotni o'chirish)
TRUNCATE TABLE archived_products RESTART IDENTITY CASCADE;

-- 6. Soft DELETE (ma'lumotni o'chirmasdan inactive qilish)
-- Soft delete uchun yangi column qo'shish
ALTER TABLE products ADD COLUMN deleted_at TIMESTAMPTZ;

-- Soft delete function
CREATE OR REPLACE FUNCTION soft_delete_product(product_id INTEGER)
RETURNS VOID AS $$
BEGIN
    UPDATE products
    SET deleted_at = NOW(),
        status = 'inactive',
        updated_at = NOW()
    WHERE id = product_id
      AND deleted_at IS NULL;
END;
$$ LANGUAGE plpgsql;

-- Ishlatish
SELECT soft_delete_product(10);

-- 7. CASCADE DELETE misollari
-- Buyurtmani va barcha items'larni o'chirish
DELETE FROM orders WHERE id = 5;  -- order_items avtomatik o'chiriladi (CASCADE)
```

## 3. JOIN Operations - Jadvallarni Bog'lash 🔗

### INNER JOIN

```sql
-- 1. Mahsulotlar va ularning kategoriyalari
SELECT p.name as product_name,
       p.price,
       c.name as category_name,
       p.stock_quantity
FROM products p
INNER JOIN categories c ON p.category_id = c.id
WHERE p.status = 'active'
ORDER BY c.name, p.name;

-- 2. Foydalanuvchilar va ularning buyurtmalari
SELECT u.first_name || ' ' || u.last_name as customer_name,
       u.email,
       o.order_number,
       o.total_amount,
       o.status,
       o.ordered_at
FROM users u
INNER JOIN orders o ON u.id = o.user_id
WHERE o.status IN ('delivered', 'shipped')
ORDER BY o.ordered_at DESC;

-- 3. Complex INNER JOIN (3+ jadval)
SELECT u.username,
       o.order_number,
       p.name as product_name,
       oi.quantity,
       oi.unit_price,
       oi.total_price
FROM users u
INNER JOIN orders o ON u.id = o.user_id
INNER JOIN order_items oi ON o.id = oi.order_id
INNER JOIN products p ON oi.product_id = p.id
WHERE o.status = 'delivered'
ORDER BY u.username, o.ordered_at;
```

### LEFT JOIN (LEFT OUTER JOIN)

```sql
-- 1. Barcha mahsulotlar va ularning sharhlari (sharhi bo'lmagan ham)
SELECT p.name,
       p.price,
       pr.rating,
       pr.title,
       pr.comment,
       u.username as reviewer
FROM products p
LEFT JOIN product_reviews pr ON p.id = pr.product_id AND pr.is_approved = true
LEFT JOIN users u ON pr.user_id = u.id
WHERE p.status = 'active'
ORDER BY p.name, pr.created_at DESC;

-- 2. Kategoriyalar va ularning mahsulotlari soni
SELECT c.name as category_name,
       COUNT(p.id) as products_count,
       COUNT(CASE WHEN p.status = 'active' THEN 1 END) as active_count,
       AVG(p.price) as average_price
FROM categories c
LEFT JOIN products p ON c.id = p.category_id
WHERE c.is_active = true
GROUP BY c.id, c.name
ORDER BY products_count DESC;

-- 3. Foydalanuvchilar va ularning buyurtmalari soni
SELECT u.username,
       u.email,
       COUNT(o.id) as total_orders,
       SUM(o.total_amount) as total_spent,
       MAX(o.ordered_at) as last_order_date
FROM users u
LEFT JOIN orders o ON u.id = o.user_id AND o.status != 'cancelled'
GROUP BY u.id, u.username, u.email
HAVING COUNT(o.id) > 0
ORDER BY total_spent DESC NULLS LAST;
```

### RIGHT JOIN va FULL OUTER JOIN

```sql
-- RIGHT JOIN (kam ishlatiladi)
SELECT p.name,
       c.name as category_name
FROM products p
RIGHT JOIN categories c ON p.category_id = c.id
WHERE c.is_active = true;

-- FULL OUTER JOIN - barcha ma'lumotlar
SELECT p.name as product_name,
       c.name as category_name,
       CASE
         WHEN p.id IS NULL THEN 'No products in category'
         WHEN c.id IS NULL THEN 'Product without category'
         ELSE 'Matched'
       END as match_status
FROM products p
FULL OUTER JOIN categories c ON p.category_id = c.id;
```

### Self JOIN

```sql
-- Parent-child kategoriyalar
SELECT child.name as child_category,
       parent.name as parent_category
FROM categories child
LEFT JOIN categories parent ON child.parent_id = parent.id
WHERE child.parent_id IS NOT NULL;

-- Bir xil kategoriyada bir xil narxdagi mahsulotlar
SELECT p1.name as product1,
       p2.name as product2,
       p1.price,
       c.name as category_name
FROM products p1
JOIN products p2 ON p1.category_id = p2.category_id
                AND p1.price = p2.price
                AND p1.id < p2.id  -- Dublicates oldini olish
JOIN categories c ON p1.category_id = c.id
WHERE p1.status = 'active' AND p2.status = 'active';
```

## 4. Aggregate Functions va Grouping 📊

### Basic Aggregate Functions

```sql
-- 1. COUNT, SUM, AVG, MIN, MAX
SELECT COUNT(*) as total_products,
       COUNT(CASE WHEN status = 'active' THEN 1 END) as active_products,
       SUM(stock_quantity) as total_stock,
       AVG(price) as average_price,
       MIN(price) as cheapest_price,
       MAX(price) as most_expensive_price,
       STDDEV(price) as price_deviation
FROM products;

-- 2. Kategoriya bo'yicha statistika
SELECT c.name as category_name,
       COUNT(p.id) as products_count,
       SUM(p.stock_quantity) as total_stock,
       ROUND(AVG(p.price), 2) as avg_price,
       MIN(p.price) as min_price,
       MAX(p.price) as max_price
FROM categories c
LEFT JOIN products p ON c.id = p.category_id AND p.status = 'active'
GROUP BY c.id, c.name
HAVING COUNT(p.id) > 0
ORDER BY avg_price DESC;

-- 3. Monthly sales statistikasi
SELECT DATE_TRUNC('month', o.ordered_at) as month,
       COUNT(o.id) as total_orders,
       COUNT(DISTINCT o.user_id) as unique_customers,
       SUM(o.total_amount) as revenue,
       ROUND(AVG(o.total_amount), 2) as avg_order_value
FROM orders o
WHERE o.status IN ('delivered', 'shipped')
  AND o.ordered_at >= NOW() - INTERVAL '1 year'
GROUP BY DATE_TRUNC('month', o.ordered_at)
ORDER BY month DESC;

-- 4. Top selling products
SELECT p.name,
       SUM(oi.quantity) as total_sold,
       SUM(oi.total_price) as total_revenue,
       COUNT(DISTINCT oi.order_id) as orders_count,
       ROUND(AVG(oi.unit_price), 2) as avg_selling_price
FROM products p
JOIN order_items oi ON p.id = oi.product_id
JOIN orders o ON oi.order_id = o.id
WHERE o.status IN ('delivered', 'shipped')
GROUP BY p.id, p.name
HAVING SUM(oi.quantity) > 5
ORDER BY total_revenue DESC
LIMIT 10;
```

### Window Functions

```sql
-- 1. ROW_NUMBER va RANK
SELECT p.name,
       p.price,
       c.name as category_name,
       ROW_NUMBER() OVER (PARTITION BY p.category_id ORDER BY p.price DESC) as price_rank,
       RANK() OVER (ORDER BY p.price DESC) as overall_rank,
       DENSE_RANK() OVER (PARTITION BY p.category_id ORDER BY p.price DESC) as dense_rank
FROM products p
JOIN categories c ON p.category_id = c.id
WHERE p.status = 'active';

-- 2. Running totals va percentages
SELECT o.ordered_at,
       o.total_amount,
       SUM(o.total_amount) OVER (ORDER BY o.ordered_at) as running_total,
       ROUND(
         o.total_amount * 100.0 / SUM(o.total_amount) OVER (), 2
       ) as percentage_of_total
FROM orders o
WHERE o.status IN ('delivered', 'shipped')
ORDER BY o.ordered_at;

-- 3. LAG va LEAD (oldingi va keyingi qiymatlar)
SELECT o.ordered_at,
       o.total_amount,
       LAG(o.total_amount) OVER (ORDER BY o.ordered_at) as prev_order_amount,
       LEAD(o.total_amount) OVER (ORDER BY o.ordered_at) as next_order_amount,
       o.total_amount - LAG(o.total_amount) OVER (ORDER BY o.ordered_at) as diff_from_prev
FROM orders o
WHERE o.user_id = 1 AND o.status IN ('delivered', 'shipped')
ORDER BY o.ordered_at;

-- 4. NTILE (quartiles)
SELECT p.name,
       p.price,
       NTILE(4) OVER (ORDER BY p.price) as price_quartile,
       CASE NTILE(4) OVER (ORDER BY p.price)
         WHEN 1 THEN 'Budget'
         WHEN 2 THEN 'Mid-Range'
         WHEN 3 THEN 'Premium'
         WHEN 4 THEN 'Luxury'
       END as price_category
FROM products p
WHERE p.status = 'active'
ORDER BY p.price;
```

### Common Table Expressions (CTE)

```sql
-- 1. Simple CTE
WITH expensive_products AS (
    SELECT id, name, price, category_id
    FROM products
    WHERE price > 500 AND status = 'active'
),
category_stats AS (
    SELECT category_id,
           COUNT(*) as product_count,
           AVG(price) as avg_price
    FROM expensive_products
    GROUP BY category_id
)
SELECT c.name as category_name,
       cs.product_count,
       ROUND(cs.avg_price, 2) as avg_price,
       STRING_AGG(ep.name, ', ') as products
FROM category_stats cs
JOIN categories c ON cs.category_id = c.id
JOIN expensive_products ep ON cs.category_id = ep.category_id
GROUP BY c.id, c.name, cs.product_count, cs.avg_price
ORDER BY cs.avg_price DESC;

-- 2. Recursive CTE (Category hierarchy)
WITH RECURSIVE category_hierarchy AS (
    -- Base case: parent categories
    SELECT id, name, parent_id, 0 as level, name as path
    FROM categories
    WHERE parent_id IS NULL

    UNION ALL

    -- Recursive case: child categories
    SELECT c.id, c.name, c.parent_id, ch.level + 1,
           ch.path || ' > ' || c.name
    FROM categories c
    JOIN category_hierarchy ch ON c.parent_id = ch.id
)
SELECT level, path, name
FROM category_hierarchy
ORDER BY path;

-- 3. Multiple CTEs bilan complex analytics
WITH monthly_sales AS (
    SELECT DATE_TRUNC('month', o.ordered_at) as month,
           SUM(o.total_amount) as revenue
    FROM orders o
    WHERE o.status IN ('delivered', 'shipped')
      AND o.ordered_at >= NOW() - INTERVAL '12 months'
    GROUP BY DATE_TRUNC('month', o.ordered_at)
),
sales_with_growth AS (
    SELECT month,
           revenue,
           LAG(revenue) OVER (ORDER BY month) as prev_revenue,
           ROUND(
             ((revenue - LAG(revenue) OVER (ORDER BY month)) /
              LAG(revenue) OVER (ORDER BY month) * 100), 2
           ) as growth_rate
    FROM monthly_sales
)
SELECT month,
       revenue,
       prev_revenue,
       growth_rate,
       CASE
         WHEN growth_rate > 10 THEN 'High Growth'
         WHEN growth_rate > 0 THEN 'Positive Growth'
         WHEN growth_rate > -10 THEN 'Slight Decline'
         ELSE 'Significant Decline'
       END as performance
FROM sales_with_growth
ORDER BY month;
```

### Performance Testing

```javascript
// Node.js performance testing
const pool = require('../config/database');

class QueryPerformance {
  static async testQuery(name, query, params = []) {
    const start = Date.now();
    try {
      const result = await pool.query(query, params);
      const duration = Date.now() - start;

      console.log(`${name}: ${duration}ms (${result.rows.length} rows)`);
      return { duration, rows: result.rows.length };
    } catch (error) {
      console.error(`${name} failed:`, error.message);
      throw error;
    }
  }

  static async compareQueries() {
    // Test 1: Simple vs optimized product search
    await this.testQuery(
      'Unoptimized search',
      `SELECT p.*, c.name as category_name
       FROM products p, categories c
       WHERE p.category_id = c.id
         AND p.name ILIKE '%phone%'`
    );

    await this.testQuery(
      'Optimized search',
      `SELECT p.*, c.name as category_name
       FROM products p
       JOIN categories c ON p.category_id = c.id
       WHERE p.status = 'active'
         AND p.name ILIKE $1`,
      ['%phone%']
    );

    // Test 2: Subquery vs JOIN
    await this.testQuery(
      'Subquery approach',
      `SELECT * FROM products p
       WHERE p.id IN (
         SELECT DISTINCT oi.product_id
         FROM order_items oi
         JOIN orders o ON oi.order_id = o.id
         WHERE o.status = 'delivered'
       )`
    );

    await this.testQuery(
      'JOIN approach',
      `SELECT DISTINCT p.*
       FROM products p
       JOIN order_items oi ON p.id = oi.product_id
       JOIN orders o ON oi.order_id = o.id
       WHERE o.status = 'delivered'`
    );
  }
}

// Test ishga tushirish
```

### Inventory Management

```sql
-- 2. Stock analysis va reorder alerts
WITH inventory_analysis AS (
    SELECT
        p.id,
        p.name,
        p.sku,
        c.name as category_name,
        p.stock_quantity,
        p.low_stock_threshold,
        COALESCE(sales.units_sold_30d, 0) as units_sold_30d,
        CASE
            WHEN p.stock_quantity = 0 THEN 'OUT_OF_STOCK'
            WHEN p.stock_quantity <= p.low_stock_threshold THEN 'LOW_STOCK'
            WHEN COALESCE(sales.units_sold_30d, 0) = 0 THEN 'NO_SALES'
            ELSE 'NORMAL'
        END as stock_status,
        CASE
            WHEN COALESCE(sales.units_sold_30d, 0) > 0
            THEN p.stock_quantity::float / (sales.units_sold_30d / 30.0)
            ELSE NULL
        END as days_of_inventory
    FROM products p
    JOIN categories c ON p.category_id = c.id
    LEFT JOIN (
        SELECT
            oi.product_id,
            SUM(oi.quantity) as units_sold_30d
        FROM order_items oi
        JOIN orders o ON oi.order_id = o.id
        WHERE o.status IN ('delivered', 'shipped')
          AND o.ordered_at >= CURRENT_DATE - INTERVAL '30 days'
        GROUP BY oi.product_id
    ) sales ON p.id = sales.product_id
    WHERE p.status = 'active'
)
SELECT
    stock_status,
    COUNT(*) as product_count,
    STRING_AGG(
        name || ' (' || sku || ')',
        ', '
        ORDER BY stock_quantity
    ) FILTER (WHERE stock_status IN ('OUT_OF_STOCK', 'LOW_STOCK')) as products_needing_attention
FROM inventory_analysis
GROUP BY stock_status
ORDER BY
    CASE stock_status
        WHEN 'OUT_OF_STOCK' THEN 1
        WHEN 'LOW_STOCK' THEN 2
        WHEN 'NO_SALES' THEN 3
        ELSE 4
    END;
```

### Customer Lifetime Value (CLV)

```sql
-- 3. Customer analytics va CLV
WITH customer_metrics AS (
    SELECT
        u.id,
        u.username,
        u.email,
        EXTRACT(EPOCH FROM (CURRENT_DATE - u.created_at::date)) / 86400 as days_as_customer,
        COUNT(o.id) as total_orders,
        SUM(o.total_amount) as total_spent,
        AVG(o.total_amount) as avg_order_value,
        MAX(o.ordered_at) as last_order_date,
        EXTRACT(EPOCH FROM (CURRENT_DATE - MAX(o.ordered_at)::date)) / 86400 as days_since_last_order,
        COUNT(o.id)::float / NULLIF(EXTRACT(EPOCH FROM (CURRENT_DATE - MIN(o.ordered_at)::date)) / 86400, 0) as order_frequency_per_day
    FROM users u
    LEFT JOIN orders o ON u.id = o.user_id AND o.status IN ('delivered', 'shipped')
    WHERE u.is_active = true
    GROUP BY u.id, u.username, u.email, u.created_at
    HAVING COUNT(o.id) > 0
),
clv_calculation AS (
    SELECT
        *,
        -- Simple CLV: avg_order_value * order_frequency * customer_lifespan (projected)
        ROUND((avg_order_value * order_frequency_per_day * 365), 2) as projected_annual_value,
        CASE
            WHEN days_since_last_order > 90 THEN 'At Risk'
            WHEN days_since_last_order > 30 THEN 'Inactive'
            WHEN total_spent > 1000 THEN 'VIP'
            WHEN total_orders > 5 THEN 'Loyal'
            ELSE 'Regular'
        END as customer_segment
    FROM customer_metrics
)
SELECT
    customer_segment,
    COUNT(*) as customer_count,
    ROUND(AVG(total_spent), 2) as avg_total_spent,
    ROUND(AVG(avg_order_value), 2) as avg_order_value,
    ROUND(AVG(projected_annual_value), 2) as avg_projected_annual_value,
    ROUND(AVG(days_since_last_order), 2) as avg_days_since_last_order
FROM clv_calculation
GROUP BY customer_segment
ORDER BY avg_projected_annual_value DESC;
```
