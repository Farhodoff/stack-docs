## 1. Migration Fundamentals 🔄

### Migration nima va nega kerak?

Database migration - bu database schema'ni bir versiyadan ikkinchisiga o'tkazish jarayoni.

```sql
-- Migration misoli: users jadvaliga phone column qo'shish
-- VERSION 1: Migration yaratilishidan oldin
CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    username VARCHAR(50) UNIQUE NOT NULL,
    email VARCHAR(255) UNIQUE NOT NULL,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- VERSION 2: Migration qo'llanganidan keyin
ALTER TABLE users ADD COLUMN phone VARCHAR(20);
ALTER TABLE users ADD COLUMN is_verified BOOLEAN DEFAULT false;
```

### Migration'ning afzalliklari:

1. **Version Control** - Schema o'zgarishlari versiyalanadi

```typescript
// Data migration script
// scripts/migrate-age-data.ts
import { PrismaClient } from '@prisma/client'

const prisma = new PrismaClient()

async function migrateAgeData() {
  console.log('🔄 Starting age data migration...')

  const users = await prisma.user.findMany({
    where: {
      age: { not: null },
      ageNew: null
    }
  })

  let migrated = 0
  let errors = 0

  for (const user of users) {
    try {
      const ageValue = parseInt(user.age!)

      if (!isNaN(ageValue) && ageValue > 0 && ageValue < 150) {
        await prisma.user.update({
          where: { id: user.id },
          data: { ageNew: ageValue }
        })
        migrated++
      } else {
        console.warn(`Invalid age for user ${user.id}: ${user.age}`)
        errors++
      }
    } catch (error) {
      console.error(`Error migrating user ${user.id}:`, error)
      errors++
    }
  }

  console.log(`✅ Migration completed: ${migrated} migrated, ${errors} errors`)
}

migrateAgeData()
  .catch(console.error)
  .finally(() => prisma.$disconnect())
```

```typescript
// Data migration script
async function migratePostData() {
  const posts = await prisma.post.findMany()

  for (const post of posts) {
    // Generate slug
    const slug = post.title.toLowerCase()
      .replace(/[^a-z0-9 -]/g, '')
      .replace(/\s+/g, '-')
      .replace(/-+/g, '-')
      .trim()

    // Map old status to new enum
    let newStatus: PostStatus = 'DRAFT'
    switch (post.status?.toLowerCase()) {
      case 'published': newStatus = 'PUBLISHED'; break
      case 'archived': newStatus = 'ARCHIVED'; break
      default: newStatus = 'DRAFT'
    }

    // Generate excerpt from content
    const excerpt = post.content.substring(0, 200) + '...'

    await prisma.post.update({
      where: { id: post.id },
      data: {
        statusNew: newStatus,
        slug: `${slug}-${post.id}`, // Ensure uniqueness
        excerpt
      }
    })
  }
}
```

## 4. Production Migration Strategies 🚀

### Zero-Downtime Migration Patterns

#### Pattern 1: Additive Changes Only

```sql
-- ✅ SAFE: Adding nullable columns
ALTER TABLE users ADD COLUMN phone VARCHAR(20);
ALTER TABLE users ADD COLUMN last_login_at TIMESTAMPTZ;

-- ✅ SAFE: Adding indexes (background)
CREATE INDEX CONCURRENTLY idx_users_last_login
ON users(last_login_at) WHERE last_login_at IS NOT NULL;

-- ✅ SAFE: Adding new tables
CREATE TABLE user_preferences (
    id SERIAL PRIMARY KEY,
    user_id INTEGER REFERENCES users(id),
    theme VARCHAR(20) DEFAULT 'light',
    notifications JSONB DEFAULT '{}'::jsonb
);
```

#### Pattern 2: Backward Compatible Changes

```typescript

// Step 1: Code supports both schemas
export class UserService {

      where: { id },
      select: {

        lastName: true,
```

```typescript
async function validateMigration() {
  try {
    // 1. Validate schema syntax
    await execAsync('npx prisma validate')
    console.log('✅ Schema validation passed')

    const migrationFiles = await getMigrationFiles()

    process.exit(1)
  }
}

async function checkForUnsafeOperations(migrationFile: string) {
  const content = await fs.readFile(migrationFile, 'utf8')

  const unsafeOperations = [
    'DROP TABLE',
    'DROP COLUMN',
    'ALTER COLUMN .* SET NOT NULL',
    'ALTER COLUMN .* TYPE'
  ]

  for (const operation of unsafeOperations) {
    if (new RegExp(operation, 'i').test(content)) {
      console.warn(`⚠️  Potentially unsafe operation detected: ${operation}`)
      console.warn(`   File: ${migrationFile}`)
    }
  }
}
```

### Migration Status Monitoring

```typescript
// Migration health checker
export class MigrationMonitor {
  static async checkMigrationStatus() {
    try {
      const result = await execAsync('npx prisma migrate status --json')
      const status = JSON.parse(result.stdout)

      return {
        database: status.database,
        migrationsTableExists: status.migrationsTableExists,
        pendingMigrations: status.pendingMigrations,
        failedMigrations: status.failedMigrations,
        appliedMigrations: status.appliedMigrations?.length || 0,
        healthy: status.pendingMigrations.length === 0 && status.failedMigrations.length === 0
      }
    } catch (error) {
      console.error('Migration status check failed:', error)
      return {
        healthy: false,
        error: error.message
      }
    }
  }

  static async healthCheck() {
    const migrationStatus = await this.checkMigrationStatus()

    if (!migrationStatus.healthy) {
      // Send alert notification
      await this.sendAlert({
        type: 'MIGRATION_ISSUE',
        message: 'Database migration issues detected',
        details: migrationStatus
      })
    }

    return migrationStatus
  }

  private static async sendAlert(alert: any) {
    // Integration with monitoring system (e.g., Slack, PagerDuty)
    console.log('🚨 ALERT:', alert)
  }
}

// Express middleware for migration health
export function migrationHealthMiddleware() {
  return async (req: Request, res: Response, next: NextFunction) => {
    if (req.path === '/health/migrations') {
      const status = await MigrationMonitor.checkMigrationStatus()

      return res.status(status.healthy ? 200 : 503).json({
        status: status.healthy ? 'healthy' : 'unhealthy',
        details: status
      })
    }

    next()
  }
}
```

### Automatic Rollback System

```typescript
// Automatic rollback on failed deployment
class DeploymentManager {
  async deployWithAutoRollback() {
    const backupId = await this.createBackup()

    try {
      console.log('🚀 Starting deployment...')

      // Deploy migrations
      await this.deployMigrations()

      // Deploy application
      await this.deployApplication()

      // Health check
      await this.performHealthCheck()

      console.log('✅ Deployment successful')

      // Clean up old backups
      await this.cleanupOldBackups()

    } catch (error) {
      console.error('❌ Deployment failed:', error)

      // Automatic rollback
      await this.rollback(backupId)

      throw error
    }
  }

  private async createBackup(): Promise<string> {
    const backupId = `backup-${Date.now()}`
    await execAsync(`pg_dump ${process.env.DATABASE_URL} > ${backupId}.sql`)
    return backupId
  }

  private async rollback(backupId: string) {
    console.log(`🔄 Rolling back to backup ${backupId}...`)

    try {
      // Stop application
      await this.stopApplication()

      // Restore database
      await execAsync(`psql ${process.env.DATABASE_URL} < ${backupId}.sql`)

      // Start application with previous version
      await this.startPreviousApplicationVersion()

      console.log('✅ Rollback completed')

    } catch (rollbackError) {
      console.error('❌ Rollback failed:', rollbackError)
      // Alert ops team for manual intervention
      await this.alertOpsTeam('critical', 'Automated rollback failed')
    }
  }
}
```

```typescript
// Track migration performance
class MigrationProfiler {
  static async profileMigration(migrationName: string) {
    const startTime = Date.now()
    const startMemory = process.memoryUsage()

    try {
      console.log(`📊 Profiling migration: ${migrationName}`)
```

```typescript
// Migration safety analyzer
class MigrationSafetyAnalyzer {
  static async analyzeMigration(migrationPath: string) {
    const migrationSQL = await fs.readFile(migrationPath, 'utf8')

    const analysis = {
      riskLevel: 'LOW' as 'LOW' | 'MEDIUM' | 'HIGH',
      estimatedDowntime: '0s',
      warnings: [] as string[],
      recommendations: [] as string[]
    }

    // Check for high-risk operations
    if (this.containsTableDrop(migrationSQL)) {
      analysis.riskLevel = 'HIGH'
      analysis.warnings.push('Contains DROP TABLE - irreversible operation')
      analysis.recommendations.push('Ensure data is backed up')
    }

    if (this.containsColumnDrop(migrationSQL)) {
      analysis.riskLevel = 'HIGH'
      analysis.warnings.push('Contains DROP COLUMN - data will be lost')
      analysis.recommendations.push('Consider archiving data first')
    }

    if (this.containsNotNullConstraint(migrationSQL)) {
      analysis.riskLevel = 'MEDIUM'
      analysis.warnings.push('Adding NOT NULL constraint may fail on existing data')
      analysis.recommendations.push('Ensure all existing records have values')
    }

    if (this.containsLargeTableChange(migrationSQL)) {
      analysis.estimatedDowntime = '5-15 minutes'
      analysis.warnings.push('Large table modification may cause downtime')
      analysis.recommendations.push('Consider maintenance window')
    }

    return analysis
  }

  private static containsTableDrop(sql: string): boolean {
    return /DROP\s+TABLE/i.test(sql)
  }

  private static containsColumnDrop(sql: string): boolean {
    return /DROP\s+COLUMN/i.test(sql)
  }

  private static containsNotNullConstraint(sql: string): boolean {
    return /ALTER\s+COLUMN.*SET\s+NOT\s+NULL/i.test(sql)
  }

  private static containsLargeTableChange(sql: string): boolean {
    const largeTablePattern = /(users|posts|orders|transactions|logs)/i
    const alterPattern = /ALTER\s+TABLE/i
    return largeTablePattern.test(sql) && alterPattern.test(sql)
  }
}
```

```typescript
    resetOnConflict: true,
    seedAfterMigration: true
  },

  staging: {
    autoApply: false,
    requireApproval: true,
    testWithProductionData: true
  },

  production: {
    autoApply: false,
    requireMultipleApprovals: true,
    maintenanceWindow: true,
    autoRollback: true
  }
}

// Migration orchestrator
class MigrationOrchestrator {
  async executeMigration(environment: string) {
    const config = migrationConfig[environment]

    if (environment === 'production') {
      await this.executeProductionMigration(config)
    } else if (environment === 'staging') {
      await this.executeStagingMigration(config)
    } else {
      await this.executeDevelopmentMigration(config)
    }
  }

  private async executeProductionMigration(config: any) {
    // 1. Pre-flight checks
    await this.performPreFlightChecks()

    // 2. Create backup
    const backupId = await this.createDatabaseBackup()

    // 3. Enable maintenance mode
    await this.enableMaintenanceMode()

    try {
      // 4. Apply migrations
      await this.applyMigrationsWithMonitoring()

      // 5. Run health checks
      await this.runHealthChecks()

      // 6. Disable maintenance mode
      await this.disableMaintenanceMode()

    } catch (error) {
      if (config.autoRollback) {
        await this.rollback(backupId)
      }
      throw error
    }
  }
}
```
