```sql
# postgresql.conf faylida:

# Connection taxaddi
listen_addresses = 'localhost'      # Local development
port = 5432                         # Default port
max_connections = 100               # Connection limit

# Memory buferlar
shared_buffers = 128MB              # Shared memory
work_mem = 4MB                      # Per-query memory
maintenance_work_mem = 64MB         # Maintenance operations

# Logging
log_statement = 'all'               # Log barcha querylar (dev uchun)
log_directory = 'log'              # Log papkasi
log_filename = 'postgresql-%Y-%m-%d_%H%M%S.log'

# Performance
checkpoint_completion_target = 0.9
wal_buffers = 16MB
random_page_cost = 1.1             # SSD optimization
```

### Birinchi Database va Jadval Yaratish

```sql
-- 1. Database yaratish
CREATE DATABASE blog_app
WITH
    OWNER = postgres
    ENCODING = 'UTF8'
    LC_COLLATE = 'en_US.UTF-8'
    LC_CTYPE = 'en_US.UTF-8';

-- 2. Database'ga ulanish
\c blog_app

-- 3. Foydalanuvchilar jadvali yaratish
CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    username VARCHAR(50) UNIQUE NOT NULL,
    email VARCHAR(255) UNIQUE NOT NULL,
    full_name VARCHAR(255) NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    bio TEXT,
    avatar_url VARCHAR(500),
    is_active BOOLEAN DEFAULT true,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW(),

    -- Validatsiya
    CONSTRAINT email_format CHECK (email ~* '^[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Za-z]{2,}$'),
    CONSTRAINT username_length CHECK (char_length(username) >= 3)
);

-- 4. Index qo'shish
CREATE INDEX idx_users_username ON users(username);
CREATE INDEX idx_users_email ON users(email);
CREATE INDEX idx_users_active ON users(is_active) WHERE is_active = true;

-- 5. Sample ma'lumot qo'shish
INSERT INTO users (username, email, full_name, password_hash, bio) VALUES
('john_doe', 'john@example.com', 'John Doe', 'hashed_password_here', 'Software Developer'),
('jane_smith', 'jane@example.com', 'Jane Smith', 'hashed_password_here', 'UI/UX Designer'),
('alice_brown', 'alice@example.com', 'Alice Brown', 'hashed_password_here', 'Product Manager');

-- 6. Ma'lumotni ko'rish
SELECT id, username, email, full_name, created_at FROM users;
```

### Database Backup va Restore

```bash
# Database backup yaratish
pg_dump -U postgres -h localhost blog_app > blog_app_backup.sql

# Specific table backup
pg_dump -U postgres -h localhost -t users blog_app > users_backup.sql

# Database restore qilish
psql -U postgres -h localhost blog_app < blog_app_backup.sql

# CSV format backup
psql -U postgres -d blog_app -c "\copy users TO 'users.csv' DELIMITER ',' CSV HEADER;"
```

### Birinchi Database Query

```javascript
// models/User.js
const pool = require('../config/database');

class User {
  // Barcha foydalanuvchilarni olish
  static async getAll() {
    try {
      const result = await pool.query(`
        SELECT id, username, email, full_name, bio, created_at
        FROM users
        WHERE is_active = true
        ORDER BY created_at DESC
      `);
      return result.rows;
    } catch (error) {
      throw new Error(`Foydalanuvchilarni olishda xato: ${error.message}`);
    }
  }

  // ID bo'yicha foydalanuvchini topish
  static async findById(id) {
    try {
      const result = await pool.query(`
        SELECT id, username, email, full_name, bio, avatar_url, created_at
        FROM users
        WHERE id = $1 AND is_active = true
      `, [id]);

      return result.rows[0] || null;
    } catch (error) {
      throw new Error(`Foydalanuvchi topishda xato: ${error.message}`);
    }
  }

  // Yangi foydalanuvchi yaratish
  static async create(userData) {
    const { username, email, full_name, password_hash, bio } = userData;

    try {
      const result = await pool.query(`
        INSERT INTO users (username, email, full_name, password_hash, bio)
        VALUES ($1, $2, $3, $4, $5)
        RETURNING id, username, email, full_name, created_at
      `, [username, email, full_name, password_hash, bio]);

      return result.rows[0];
    } catch (error) {
      if (error.code === '23505') { // Unique violation
        throw new Error('Bunday username yoki email allaqachon mavjud');
      }
      throw new Error(`Foydalanuvchi yaratishda xato: ${error.message}`);
    }
  }

  // Foydalanuvchini yangilash
  static async updateById(id, updates) {
    const allowedFields = ['full_name', 'bio', 'avatar_url'];
    const fields = Object.keys(updates).filter(field => allowedFields.includes(field));

    if (fields.length === 0) {
      throw new Error('Yangilash uchun maydon topilmadi');
    }

    const setPart = fields.map((field, index) => `${field} = $${index + 1}`).join(', ');
    const values = fields.map(field => updates[field]);
    values.push(id); // ID oxirgi parametr

    try {
      const result = await pool.query(`
        UPDATE users
        SET ${setPart}, updated_at = NOW()
        WHERE id = $${values.length} AND is_active = true
        RETURNING id, username, email, full_name, bio, updated_at
      `, values);

      return result.rows[0] || null;
    } catch (error) {
      throw new Error(`Foydalanuvchi yangilashda xato: ${error.message}`);
    }
  }

  // Foydalanuvchini o'chirish (soft delete)
  static async deleteById(id) {
    try {
      const result = await pool.query(`
        UPDATE users
        SET is_active = false, updated_at = NOW()
        WHERE id = $1 AND is_active = true
        RETURNING id, username
      `, [id]);

      return result.rows[0] || null;
    } catch (error) {
      throw new Error(`Foydalanuvchi o'chirishda xato: ${error.message}`);
    }
  }
}

module.exports = User;
```

### Express API yaratish

```javascript
// server.js
const express = require('express');
const User = require('./models/User');
require('dotenv').config();

const app = express();
const PORT = process.env.PORT || 3000;

app.use(express.json());

// CRUD API endpoints
app.get('/api/users', async (req, res) => {
  try {
    const users = await User.getAll();
    res.json({
      success: true,
      data: users,
      count: users.length
    });
  } catch (error) {
    res.status(500).json({
      success: false,
      error: error.message
    });
  }
});

app.get('/api/users/:id', async (req, res) => {
  try {
    const user = await User.findById(req.params.id);
    if (!user) {
      return res.status(404).json({
        success: false,
        error: 'Foydalanuvchi topilmadi'
      });
    }
    res.json({
      success: true,
      data: user
    });
  } catch (error) {
    res.status(500).json({
      success: false,
      error: error.message
    });
  }
});

app.post('/api/users', async (req, res) => {
  try {
    const user = await User.create(req.body);
    res.status(201).json({
      success: true,
      data: user,
      message: 'Foydalanuvchi muvaffaqiyatli yaratildi'
    });
  } catch (error) {
    res.status(400).json({
      success: false,
      error: error.message
    });
  }
});

app.put('/api/users/:id', async (req, res) => {
  try {
    const user = await User.updateById(req.params.id, req.body);
    if (!user) {
      return res.status(404).json({
        success: false,
        error: 'Foydalanuvchi topilmadi yoki yangilanmadi'
      });
    }
    res.json({
      success: true,
      data: user,
      message: 'Foydalanuvchi muvaffaqiyatli yangilandi'
    });
  } catch (error) {
    res.status(400).json({
      success: false,
      error: error.message
    });
  }
});

app.delete('/api/users/:id', async (req, res) => {
  try {
    const user = await User.deleteById(req.params.id);
    if (!user) {
      return res.status(404).json({
        success: false,
        error: 'Foydalanuvchi topilmadi'
      });
    }
    res.json({
      success: true,
      message: 'Foydalanuvchi muvaffaqiyatli o\'chirildi',
      data: user
    });
  } catch (error) {
    res.status(500).json({
      success: false,
      error: error.message
    });
  }
});

// Health check
app.get('/health', async (req, res) => {
  try {
    const result = await pool.query('SELECT NOW() as server_time');
    res.json({
      status: 'OK',
      database: 'Connected',
      server_time: result.rows[0].server_time
    });
  } catch (error) {
    res.status(503).json({
      status: 'Error',
      database: 'Disconnected',
      error: error.message
    });
  }
});

app.listen(PORT, () => {
  console.log(`🚀 Server ${PORT} portda ishlamoqda`);
  console.log(`🏥 Health check: http://localhost:${PORT}/health`);
  console.log(`👥 Users API: http://localhost:${PORT}/api/users`);
});
```

## 6. Database Security va Best Practices 🔒

### Security Settings

```sql
-- 1. Foydalanuvchi va rolelar yaratish
CREATE ROLE app_user LOGIN PASSWORD 'strong_password_here';
CREATE DATABASE blog_app_prod OWNER app_user;

-- 2. Minimal permissions berish
GRANT CONNECT ON DATABASE blog_app_prod TO app_user;
GRANT USAGE ON SCHEMA public TO app_user;
GRANT SELECT, INSERT, UPDATE, DELETE ON ALL TABLES IN SCHEMA public TO app_user;
GRANT USAGE, SELECT ON ALL SEQUENCES IN SCHEMA public TO app_user;

-- 3. Superuser permissions cheklash
-- Production'da superuser ishlatmang!

-- 4. SSL ulanish talab qilish
-- postgresql.conf da:
ssl = on
ssl_cert_file = 'server.crt'
```

## 7. Performance Monitoring va Optimization 📊

### Basic Performance Queries

```sql
-- 1. Aktiv connectionlarni ko'rish
SELECT pid, usename, application_name, client_addr, state, query_start, query
FROM pg_stat_activity
WHERE state = 'active';

-- 2. Eng sekin querylarni topish
SELECT query,
       calls,
       total_time,
       mean_time,
       stddev_time
FROM pg_stat_statements
ORDER BY mean_time DESC
LIMIT 10;

-- 3. Database hajmi
SELECT pg_database.datname,
       pg_size_pretty(pg_database_size(pg_database.datname)) AS size
FROM pg_database
ORDER BY pg_database_size(pg_database.datname) DESC;

-- 4. Index usage statistikasi
SELECT indexrelname as index_name,
       idx_tup_read,
       idx_tup_fetch,
       pg_size_pretty(pg_relation_size(indexrelname::regclass)) as index_size
FROM pg_stat_user_indexes
ORDER BY idx_tup_read DESC;
```

### Monitoring Script

```javascript
// utils/monitor.js
const pool = require('../config/database');

class DatabaseMonitor {
  static async getConnectionStats() {
    const result = await pool.query(`
      SELECT count(*) as total_connections,
             count(*) filter (where state = 'active') as active_connections,
             count(*) filter (where state = 'idle') as idle_connections
      FROM pg_stat_activity
      WHERE datname = current_database();
    `);
    return result.rows[0];
  }

  static async getDatabaseSize() {
    const result = await pool.query(`
      SELECT pg_size_pretty(pg_database_size(current_database())) as database_size;
    `);
    return result.rows[0].database_size;
  }

  static async getSlowQueries(limit = 10) {
    const result = await pool.query(`
      SELECT query, calls, total_time, mean_time
      FROM pg_stat_statements
      WHERE query NOT LIKE '%pg_stat%'
      ORDER BY mean_time DESC
      LIMIT $1;
    `, [limit]);
    return result.rows;
  }

  static async healthCheck() {
    try {
      const connections = await this.getConnectionStats();
      const size = await this.getDatabaseSize();

      return {
        status: 'healthy',
        timestamp: new Date(),
        connections,
        database_size: size
      };
    } catch (error) {
      return {
        status: 'unhealthy',
        error: error.message
      };
    }
  }
}

module.exports = DatabaseMonitor;
```
