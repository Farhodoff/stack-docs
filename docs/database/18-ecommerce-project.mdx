## 3. Advanced Business Logic Implementation 💼

### Inventory Management System

```typescript
// services/InventoryService.ts
export class InventoryService {
  // Reserve inventory for cart items
  static async reserveInventory(cartItems: CartItem[]) {
    return await prisma.$transaction(async (tx) => {
      const reservations: InventoryReservation[] = []

      for (const item of cartItems) {
        const product = await tx.product.findUnique({
          where: { id: item.productId },
          include: { variants: true }
        })

        if (!product) {
          throw new Error(`Product ${item.productId} not found`)
        }

        // Check if variant-specific inventory is available
        if (item.variantId) {
          const variant = await tx.productVariant.findUnique({
            where: { id: item.variantId }
          })

          if (!variant || variant.quantity < item.quantity) {
            throw new Error(`Insufficient inventory for variant ${item.variantId}`)
          }

          // Reserve variant inventory
          await tx.productVariant.update({
            where: { id: item.variantId },
            data: { quantity: { decrement: item.quantity } }
          })

          reservations.push({
            type: 'VARIANT',
            variantId: item.variantId,
            quantity: item.quantity
          })

        } else {
          // Check product-level inventory
          if (!product.trackQuantity || product.quantity >= item.quantity) {
            await tx.product.update({
              where: { id: item.productId },
              data: { quantity: { decrement: item.quantity } }
            })

            reservations.push({
              type: 'PRODUCT',
              productId: item.productId,
              quantity: item.quantity
            })
          } else {
            throw new Error(`Insufficient inventory for product ${item.productId}`)
          }
        }
      }

      // Create temporary reservations (expires in 15 minutes)
      const reservation = await tx.inventoryReservation.create({
        data: {
          userId: cartItems[0].userId,
          expiresAt: new Date(Date.now() + 15 * 60 * 1000),
          items: reservations
        }
      })

      return reservation
    })
  }

  // Confirm inventory reservation (convert to actual order)
  static async confirmReservation(reservationId: string, orderId: string) {
    return await prisma.$transaction(async (tx) => {
      const reservation = await tx.inventoryReservation.findUnique({
        where: { id: reservationId }
      })

      if (!reservation) {
        throw new Error('Reservation not found')
      }

      if (reservation.expiresAt < new Date()) {
        throw new Error('Reservation expired')
      }

      // Mark reservation as confirmed
      await tx.inventoryReservation.update({
        where: { id: reservationId },
        data: {
          status: 'CONFIRMED',
          orderId
        }
      })

      return reservation
    })
  }

  // Release expired reservations
  static async releaseExpiredReservations() {
    const expiredReservations = await prisma.inventoryReservation.findMany({
      where: {
        status: 'PENDING',
        expiresAt: { lt: new Date() }
      }
    })

    for (const reservation of expiredReservations) {
      await this.releaseReservation(reservation.id)
    }
  }

  private static async releaseReservation(reservationId: string) {
    return await prisma.$transaction(async (tx) => {
      const reservation = await tx.inventoryReservation.findUnique({
        where: { id: reservationId }
      })

      if (!reservation) return

      // Restore inventory quantities
      for (const item of reservation.items as any[]) {
        if (item.type === 'VARIANT') {
          await tx.productVariant.update({
            where: { id: item.variantId },
            data: { quantity: { increment: item.quantity } }
          })
        } else {
          await tx.product.update({
            where: { id: item.productId },
            data: { quantity: { increment: item.quantity } }
          })
        }
      }

      // Mark reservation as released
      await tx.inventoryReservation.update({
        where: { id: reservationId },
        data: { status: 'RELEASED' }
      })
    })
  }
```

```typescript
// services/PricingService.ts
export class PricingService {
  // Calculate product price with all modifiers
  static async calculatePrice(productId: string, variantId?: string, userId?: string) {
    const product = await prisma.product.findUnique({
      where: { id: productId },
      include: {
        variants: variantId ? {
          where: { id: variantId }
        } : false,
        category: true
      }
    })

    if (!product) throw new Error('Product not found')

    let basePrice = product.price
    let comparePrice = product.comparePrice

    // Use variant price if specified
    if (variantId && product.variants[0]) {
      basePrice = product.variants[0].price || basePrice
      comparePrice = product.variants[0].comparePrice || comparePrice
    }

    // Apply user-specific pricing
    if (userId) {
      const userTier = await this.getUserPricingTier(userId)
      basePrice = this.applyTierDiscount(basePrice, userTier)
    }

    // Apply active promotions
    const activePromotions = await this.getActivePromotions(productId, product.categoryId!)
    const promotionalPrice = this.applyPromotions(basePrice, activePromotions)

    // Bulk pricing (if applicable)
    const bulkPricing = await this.getBulkPricing(productId)

    return {
      basePrice: Number(basePrice),
      comparePrice: comparePrice ? Number(comparePrice) : null,
      finalPrice: Number(promotionalPrice),
      discountPercentage: comparePrice ?
        Math.round(((Number(comparePrice) - Number(promotionalPrice)) / Number(comparePrice)) * 100) : 0,
      savings: comparePrice ? Number(comparePrice) - Number(promotionalPrice) : 0,
      bulkPricing,
      appliedPromotions: activePromotions
    }
  }

  private static async getUserPricingTier(userId: string) {
    const user = await prisma.user.findUnique({
      where: { id: userId },
      include: {
        profile: true,
        orders: {
          where: { status: 'DELIVERED' },
          select: { totalAmount: true }
        }
      }
    })

    if (!user) return 'STANDARD'

    // Calculate total spent
    const totalSpent = user.orders.reduce((sum, order) =>
      sum + Number(order.totalAmount), 0)

    // Determine tier based on spending
    if (totalSpent >= 10000) return 'PLATINUM'
    if (totalSpent >= 5000) return 'GOLD'
    if (totalSpent >= 1000) return 'SILVER'
    return 'STANDARD'
  }

  private static applyTierDiscount(price: Decimal, tier: string): Decimal {
    const discounts = {
      'PLATINUM': 0.15, // 15% discount
      'GOLD': 0.10,     // 10% discount
      'SILVER': 0.05,   // 5% discount
      'STANDARD': 0     // No discount
    }

    const discount = discounts[tier] || 0
    return price.mul(1 - discount)
  }

  private static async getActivePromotions(productId: string, categoryId: string) {
    return await prisma.promotion.findMany({
      where: {
        isActive: true,
        startsAt: { lte: new Date() },
        OR: [
          { expiresAt: null },
          { expiresAt: { gte: new Date() } }
        ],
        OR: [
          { applicableProducts: { has: productId } },
          { applicableCategories: { has: categoryId } },
          { applicationType: 'ALL' }
        ]
      },
      orderBy: { priority: 'desc' }
    })
  }

  private static applyPromotions(price: Decimal, promotions: any[]): Decimal {
    let finalPrice = price

    for (const promotion of promotions) {
      switch (promotion.type) {
        case 'PERCENTAGE':
          finalPrice = finalPrice.mul(1 - promotion.value / 100)
          break
        case 'FIXED_AMOUNT':
          finalPrice = finalPrice.sub(promotion.value)
          break
        case 'BUY_X_GET_Y':
          // Complex promotion logic here
          break
      }

      // Ensure price doesn't go below minimum
      if (promotion.minimumPrice && finalPrice.lt(promotion.minimumPrice)) {
        finalPrice = promotion.minimumPrice
      }
    }

    return finalPrice
  }
}
```

```typescript
export class OrderService {
  // Create order from cart
  static async createOrderFromCart(userId: string, orderData: CreateOrderData) {
    return await prisma.$transaction(async (tx) => {
      // Get user's cart
      const cartItems = await tx.cartItem.findMany({
        where: { userId },
        include: {
          product: true,
          variant: true
        }
      })

      if (cartItems.length === 0) {
        throw new Error('Cart is empty')
      }

      // Validate inventory and calculate prices
      let subtotal = new Decimal(0)
      const orderItems: CreateOrderItemData[] = []

      for (const cartItem of cartItems) {
        // Re-validate pricing and availability
        const pricing = await PricingService.calculatePrice(
          cartItem.productId,
          cartItem.variantId || undefined,
          userId
        )

        // Check inventory again
        await InventoryService.validateAvailability(cartItem)

        const itemTotal = new Decimal(pricing.finalPrice).mul(cartItem.quantity)
        subtotal = subtotal.add(itemTotal)

        orderItems.push({
          productId: cartItem.productId,
          variantId: cartItem.variantId,
          productName: cartItem.product.name,
          productSku: cartItem.variant?.sku || cartItem.product.sku,
          variantName: cartItem.variant?.name,
          quantity: cartItem.quantity,
          unitPrice: new Decimal(pricing.finalPrice),
          totalPrice: itemTotal,
          customOptions: cartItem.customOptions as object
        })
      }

      // Calculate shipping and taxes
      const shippingAmount = await this.calculateShipping(orderData, cartItems)
      const taxAmount = await this.calculateTax(subtotal, orderData.shippingAddress)

      // Apply coupons
      let discountAmount = new Decimal(0)
      if (orderData.couponCode) {
        discountAmount = await this.applyCoupon(orderData.couponCode, subtotal, userId)
      }

      const totalAmount = subtotal.add(shippingAmount).add(taxAmount).sub(discountAmount)

      // Generate order number
      const orderNumber = await this.generateOrderNumber()

      // Create order
      const order = await tx.order.create({
        data: {
          orderNumber,
          userId,
          status: 'PENDING',
          subtotal,
          taxAmount,
          shippingAmount,
          discountAmount,
          totalAmount,
          shippingMethod: orderData.shippingMethod,
          billingAddressId: orderData.billingAddressId,
          shippingAddressId: orderData.shippingAddressId,
          notes: orderData.notes,
          items: {
            create: orderItems
          }
        },
        include: {
          items: true,
          billingAddress: true,
          shippingAddress: true
        }
      })

      // Reserve inventory
      await InventoryService.reserveInventory(cartItems)

      // Clear cart
      await tx.cartItem.deleteMany({
        where: { userId }
      })

      // Create payment intent
      if (orderData.paymentMethod !== 'CASH_ON_DELIVERY') {
        await PaymentService.createPaymentIntent(order.id, totalAmount, orderData.paymentMethod)
      }

      // Send order confirmation
      await NotificationService.sendOrderConfirmation(order.id)

      return order
    })
  }

  // Process payment and fulfill order
  static async processPayment(orderId: string, paymentData: any) {
    return await prisma.$transaction(async (tx) => {
      const order = await tx.order.findUnique({
        where: { id: orderId },
        include: { items: true, user: true }
      })

      if (!order) throw new Error('Order not found')

      // Process payment through payment provider
      const paymentResult = await PaymentService.processPayment(order, paymentData)

      if (paymentResult.success) {
        // Update order status
        await tx.order.update({
          where: { id: orderId },
          data: {
            status: 'CONFIRMED',
            paymentStatus: 'PAID'
          }
        })

        // Confirm inventory reservation
        await InventoryService.confirmReservation(orderId)

        // Create fulfillment tasks
        await FulfillmentService.createFulfillmentTasks(orderId)

        // Award loyalty points
        await LoyaltyService.awardPurchasePoints(order.userId!, order.totalAmount)

        // Send confirmation notifications
        await NotificationService.sendPaymentConfirmation(orderId)

        return { success: true, order }
      } else {
        // Payment failed - release inventory
        await InventoryService.releaseReservation(orderId)

        await tx.order.update({
          where: { id: orderId },
          data: {
            status: 'CANCELLED',
            paymentStatus: 'FAILED'
          }
        })

        throw new Error('Payment processing failed')
      }
    })
  }

  private static async generateOrderNumber(): Promise<string> {
    const today = new Date()
    const datePrefix = today.toISOString().slice(0, 10).replace(/-/g, '')

    // Get today's order count
    const todayOrderCount = await prisma.order.count({
      where: {
        createdAt: {
          gte: new Date(today.setHours(0, 0, 0, 0))
        }
      }
    })

    const orderNumber = `ORD-${datePrefix}-${String(todayOrderCount + 1).padStart(4, '0')}`
    return orderNumber
  }
```

```typescript
    // Category filter
    if (categoryId) {

    if (brandId) {

    if (priceMin !== undefined || priceMax !== undefined) {

    if (attributes && Object.keys(attributes).length > 0) {
      where.attributes = {
        some: {

              attributeGroupId,
              value: { in: values as string[] }
            }))
          }
        }
      }
    }

    // Rating filter
    if (rating) {
      where.reviews = {
        some: {}
      }
      // Will need to calculate average rating in the query

    if (availability === 'in_stock') {
      where.quantity = { gt: 0 }

    // Dynamic sorting
    let orderBy: Prisma.ProductOrderByWithRelationInput[] = []

    switch (sortBy) {
      case 'price_low':
        orderBy = [{ price: 'asc' }]

      case 'price_high':
        orderBy = [{ price: 'desc' }]
        break
      case 'newest':
        orderBy = [{ createdAt: 'desc' }]
        break
      case 'popular':

          [{ isFeatured: 'desc' }, { salesCount: 'desc' }]
        break
    }

    // Execute search with pagination
    const [products, total] = await Promise.all([
      prisma.product.findMany({

          brand: { select: { id: true, name: true } },

            where: { isCover: true },
            select: { url: true, alt: true },

            select: { id: true, price: true, image: true },

  private static async getCategoryTree(categoryId: string): Promise<string[]> {
    const category = await prisma.category.findUnique({
      where: { id: categoryId },
      include: {
        children: {
          include: {
            children: true // 3 levels deep
          }
        }
      }
    })

    if (!category) return [categoryId]

    const categoryIds = [categoryId]

    // Add all descendant categories
    function addDescendants(cat: any) {
      if (cat.children) {
        cat.children.forEach((child: any) => {
          categoryIds.push(child.id)
          addDescendants(child)
        })
      }
    }

    addDescendants(category)
    return categoryIds
  }

          products: { some: where }

        select: {

          name: true,

          _count: { select: { products: true } }

        _min: { price: true },

                some: {
                  product: where
```

```typescript
      })

    } catch (error) {
      console.error('Error fetching product:', error)
      res.status(500).json({
        success: false,
        error: 'Failed to fetch product'
      })
    }
  }

  private static async getRelatedProducts(productId: string, categoryId: string) {
    return await prisma.product.findMany({
      where: {
        id: { not: productId },
        categoryId,
        status: 'PUBLISHED',
        visibility: 'VISIBLE'
      },
      include: {
        images: {
          where: { isCover: true },
          take: 1
        },
        brand: { select: { name: true } },
        _count: {
          select: {
            reviews: { where: { status: 'APPROVED' } }
          }
        }
      },
      take: 8,
      orderBy: [
        { salesCount: 'desc' },
        { viewCount: 'desc' }
      ]
    })
  }

  private static async trackProductView(productId: string, userId?: string, ip?: string, userAgent?: string) {
    try {
      await prisma.productView.create({
        data: {
          productId,
          userId,
          ipAddress: ip,
          userAgent
        }
      })

      // Increment view count
      await prisma.product.update({
        where: { id: productId },
        data: { viewCount: { increment: 1 } }
      })
    } catch (error) {
      // Don't throw error for tracking failures
      console.error('Failed to track product view:', error)
    }
  }

  private static buildCategoryBreadcrumb(category: any): any[] {
    const breadcrumb = [category]
    let current = category.parent

    while (current) {
      breadcrumb.unshift(current)
      current = current.parent
    }

    return breadcrumb.map(cat => ({
      id: cat.id,
      name: cat.name,
      slug: cat.slug
    }))
  }

  private static organizeAttributes(attributes: any[]) {
    const grouped = new Map()

    attributes.forEach(attr => {
      const group = attr.attributeValue.attributeGroup
      if (!grouped.has(group.id)) {
        grouped.set(group.id, {
          id: group.id,
          name: group.name,
          displayName: group.displayName,
          type: group.type,
          values: []
        })
      }

      grouped.get(group.id).values.push({
        id: attr.attributeValue.id,
        value: attr.attributeValue.value,
        displayValue: attr.attributeValue.displayValue,
        colorCode: attr.attributeValue.colorCode,
        image: attr.attributeValue.image
      })
    })

    return Array.from(grouped.values())
  }

  private static organizeVariantOptions(variants: any[]) {
    const options = new Map()

    variants.forEach(variant => {
      variant.variantAttributes.forEach(attr => {
        const group = attr.attributeValue.attributeGroup

        if (!options.has(group.id)) {
          options.set(group.id, {
            id: group.id,
            name: group.name,
            displayName: group.displayName,
            type: group.type,
            values: new Map()
          })
        }

        const groupOptions = options.get(group.id)
        const valueId = attr.attributeValue.id

        if (!groupOptions.values.has(valueId)) {
          groupOptions.values.set(valueId, {
            id: valueId,
            value: attr.attributeValue.value,
            displayValue: attr.attributeValue.displayValue,
            colorCode: attr.attributeValue.colorCode,
            image: attr.attributeValue.image,
            variants: []
          })
        }

        groupOptions.values.get(valueId).variants.push(variant.id)
      })
    })

    // Convert maps to arrays
    return Array.from(options.values()).map(group => ({
      ...group,
      values: Array.from(group.values.values())
    }))
  }
}
```

```typescript
// controllers/CartController.ts
export class CartController {
  // Add item to cart
  static async addToCart(req: Request, res: Response) {
    try {
      const userId = req.user!.id
      const { productId, variantId, quantity = 1, customOptions } = req.body

      // Validate product and variant
      const product = await prisma.product.findUnique({
        where: { id: productId },
        include: {
          variants: variantId ? { where: { id: variantId } } : false
        }
      })

      if (!product) {
        return res.status(404).json({
          success: false,
          error: 'Product not found'
        })
      }

      if (product.status !== 'PUBLISHED') {
        return res.status(400).json({
          success: false,
          error: 'Product is not available'
        })
      }

      // Check inventory availability
      if (variantId) {
        const variant = product.variants[0]
        if (!variant || !variant.isActive || variant.quantity < quantity) {
          return res.status(400).json({
            success: false,
            error: 'Insufficient inventory'
          })
        }
      } else {
        if (product.trackQuantity && product.quantity < quantity) {
          return res.status(400).json({
            success: false,
            error: 'Insufficient inventory'
          })
        }
      }

      // Calculate current price
      const pricing = await PricingService.calculatePrice(productId, variantId, userId)

      // Check if item already exists in cart
      const existingItem = await prisma.cartItem.findUnique({
        where: {
          userId_productId_variantId: {
            userId,
            productId,
            variantId
          }
        }
      })

      let cartItem

      if (existingItem) {
        // Update quantity
        cartItem = await prisma.cartItem.update({
          where: { id: existingItem.id },
          data: {
            quantity: existingItem.quantity + quantity,
            unitPrice: pricing.finalPrice,
            customOptions: customOptions || existingItem.customOptions
          },
          include: {
            product: {
              select: {
                id: true,
                name: true,
                slug: true,
                images: {
                  where: { isCover: true },
                  take: 1,
                  select: { url: true, alt: true }
                }
              }
            },
            variant: {
              select: {
                id: true,
                name: true,
                image: true
              }
            }
          }
        })
      } else {
        // Create new cart item
        cartItem = await prisma.cartItem.create({
          data: {
            userId,
            productId,
            variantId,
            quantity,
            unitPrice: pricing.finalPrice,
            customOptions
          },
          include: {
            product: {
              select: {
                id: true,
                name: true,
                slug: true,
                images: {
                  where: { isCover: true },
                  take: 1,
                  select: { url: true, alt: true }
                }
              }
            },
            variant: {
              select: {
                id: true,
                name: true,
                image: true
              }
            }
          }
        })
      }

      // Get updated cart totals
      const cartSummary = await this.getCartSummary(userId)

      res.status(201).json({
        success: true,
        data: cartItem,
        cart: cartSummary
      })

    } catch (error) {
      console.error('Error adding to cart:', error)
      res.status(500).json({
        success: false,
        error: 'Failed to add item to cart'
      })
    }
  }

  // Get cart contents
  static async getCart(req: Request, res: Response) {
    try {
      const userId = req.user!.id

      const cartItems = await prisma.cartItem.findMany({
        where: { userId },
        include: {
          product: {
            include: {
              images: {
                where: { isCover: true },
                take: 1
              },
              brand: { select: { name: true } }
            }
          },
          variant: {
            include: {
              variantAttributes: {
                include: {
                  attributeValue: {
                    include: {
                      attributeGroup: true
                    }
                  }
                }
              }
            }
          }
        },
        orderBy: { createdAt: 'desc' }
      })

      // Validate pricing and availability for each item
      const validatedItems = await Promise.all(
        cartItems.map(async (item) => {
          const pricing = await PricingService.calculatePrice(
            item.productId,
            item.variantId || undefined,
            userId
          )

          const isAvailable = await this.checkItemAvailability(item)

          const needsUpdate = Number(item.unitPrice) !== pricing.finalPrice

          return {
            ...item,
            currentPrice: pricing.finalPrice,
            originalPrice: Number(item.unitPrice),
            priceChanged: needsUpdate,
            available: isAvailable,
            totalPrice: pricing.finalPrice * item.quantity,
            originalTotal: Number(item.unitPrice) * item.quantity
          }
        })
      )

      const cartSummary = await this.getCartSummary(userId)

      res.json({
        success: true,
        data: {
          items: validatedItems,
          summary: cartSummary
        }
      })

    } catch (error) {
      console.error('Error fetching cart:', error)
      res.status(500).json({
        success: false,
        error: 'Failed to fetch cart'
      })
    }
  }

  // Update cart item
  static async updateCartItem(req: Request, res: Response) {
    try {
      const { id } = req.params
      const { quantity } = req.body
      const userId = req.user!.id

      if (quantity <= 0) {
        return this.removeFromCart(req, res)
      }

      const cartItem = await prisma.cartItem.findFirst({
        where: { id, userId },
        include: { product: true, variant: true }
      })

      if (!cartItem) {
        return res.status(404).json({
          success: false,
          error: 'Cart item not found'
        })
      }

      // Check availability
      const available = await this.checkItemAvailability(cartItem, quantity)
      if (!available) {
        return res.status(400).json({
          success: false,
          error: 'Insufficient inventory'
        })
      }

      // Update item
      const updatedItem = await prisma.cartItem.update({
        where: { id },
        data: { quantity },
        include: {
          product: {
            select: {
              id: true,
              name: true,
              slug: true,
              images: {
                where: { isCover: true },
                take: 1
              }
            }
          },
          variant: {
            select: {
              id: true,
              name: true,
              image: true
            }
          }
        }
      })

      const cartSummary = await this.getCartSummary(userId)

      res.json({
        success: true,
        data: updatedItem,
        cart: cartSummary
      })

    } catch (error) {
      console.error('Error updating cart item:', error)
      res.status(500).json({
        success: false,
        error: 'Failed to update cart item'
      })
    }
  }

  private static async getCartSummary(userId: string) {
    const items = await prisma.cartItem.findMany({
      where: { userId },
      include: { product: true, variant: true }
    })

    let subtotal = 0
    let totalItems = 0

    for (const item of items) {
      const pricing = await PricingService.calculatePrice(
        item.productId,
        item.variantId || undefined,
        userId
      )
      subtotal += pricing.finalPrice * item.quantity
      totalItems += item.quantity
    }

    return {
      itemCount: items.length,
      totalItems,
      subtotal,
      estimatedTotal: subtotal // Would include shipping/tax in real scenario
    }
  }

  private static async checkItemAvailability(cartItem: any, requestedQuantity?: number) {
    const quantity = requestedQuantity || cartItem.quantity

    if (cartItem.variantId) {
      const variant = await prisma.productVariant.findUnique({
        where: { id: cartItem.variantId }
      })
      return variant && variant.isActive && variant.quantity >= quantity
    } else {
      const product = await prisma.product.findUnique({
        where: { id: cartItem.productId }
      })
      return product &&
        product.status === 'PUBLISHED' &&
        (!product.trackQuantity || product.quantity >= quantity)
    }
  }
}
```
