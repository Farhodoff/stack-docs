# JavaScript DOM Manipulation

DOM (Document Object Model) - bu HTML hujjatning JavaScript orqali o'zgartirib bo'ladigan daraxt tuzilmasi. Bu dars orqali web sahifalarni dinamik qilishni o'rganasiz.

## DOM nima?

DOM - bu brauzer tomonidan yaratilgan HTML sahifaning obyekt ko'rinishi. Har bir HTML element JavaScript obyektiga aylanadi.

```html
<!DOCTYPE html>
<html>
<head>
    <title>DOM Misol</title>
</head>
<body>
    <h1 id="sarlavha">Salom, DOM!</h1>
    <p class="matn">Bu birinchi paragraf.</p>
    <button onclick="o'zgartir()">O'zgartir</button>
</body>
</html>
```

```javascript
// Bu HTML DOM daraxtini tasvirlaydi:

//   └── html

//       │   └── title
//       └── body
//           ├── h1 (id="sarlavha")
//           ├── p (class="matn")
//           └── button
```

### 1. Todo List Dasturi

```html
<div class="todo-app">
    <h2>Todo List</h2>
    <div class="input-section">
        <input type="text" id="todo-input" placeholder="Yangi vazifa...">
        <button id="add-btn">Qo'shish</button>
    </div>
    <ul id="todo-list"></ul>
    <div class="stats">
        Jami: <span id="total">0</span> |
        Tugagan: <span id="completed">0</span> |
        Qolgan: <span id="remaining">0</span>
    </div>
</div>
```

```javascript
class TodoApp {
    constructor() {
        this.todos = [];
        this.todoInput = document.getElementById("todo-input");
        this.addBtn = document.getElementById("add-btn");
        this.todoList = document.getElementById("todo-list");
        this.totalSpan = document.getElementById("total");
        this.completedSpan = document.getElementById("completed");
        this.remainingSpan = document.getElementById("remaining");

        this.init();
    }

    init() {
        // Event listenerlar
        this.addBtn.addEventListener("click", () => this.addTodo());
        this.todoInput.addEventListener("keypress", (e) => {
            if (e.key === "Enter") this.addTodo();
        });
    }

    addTodo() {
        let text = this.todoInput.value.trim();
        if (!text) return;

        let todo = {
            id: Date.now(),
            text: text,
            completed: false
        };

        this.todos.push(todo);
        this.renderTodo(todo);
        this.updateStats();

        this.todoInput.value = "";
        this.todoInput.focus();
    }

    renderTodo(todo) {
        let li = document.createElement("li");
        li.className = "todo-item";
        li.dataset.id = todo.id;

        li.innerHTML = `
            <input type="checkbox" class="toggle" ${todo.completed ? "checked" : ""}>
            <span class="text ${todo.completed ? "completed" : ""}">${todo.text}</span>
            <button class="delete-btn">×</button>
        `;

        let checkbox = li.querySelector(".toggle");
        let deleteBtn = li.querySelector(".delete-btn");
        let textSpan = li.querySelector(".text");

        checkbox.addEventListener("change", () => {
            this.toggleTodo(todo.id);
            textSpan.classList.toggle("completed", checkbox.checked);
        });

        deleteBtn.addEventListener("click", () => {
            this.deleteTodo(todo.id);
            li.remove();
        });

        this.todoList.appendChild(li);
    }

    toggleTodo(id) {
        let todo = this.todos.find(t => t.id === id);
        if (todo) {
            todo.completed = !todo.completed;
            this.updateStats();
        }
    }

    deleteTodo(id) {
        this.todos = this.todos.filter(t => t.id !== id);
        this.updateStats();
    }

    updateStats() {
        let total = this.todos.length;
        let completed = this.todos.filter(t => t.completed).length;
        let remaining = total - completed;

        this.totalSpan.textContent = total;
        this.completedSpan.textContent = completed;
        this.remainingSpan.textContent = remaining;
    }
}

// Dasturni ishga tushirish
document.addEventListener("DOMContentLoaded", () => {
    new TodoApp();
});
```

```javascript
class FormValidator {
    constructor(formSelector) {
        this.form = document.querySelector(formSelector);
        this.rules = {
            name: {
                required: true,
                minLength: 2,
                pattern: /^[a-zA-Z\s]+$/
            },
            email: {
                required: true,
                pattern: /^[^\s@]+@[^\s@]+\.[^\s@]+$/
            },
            phone: {
                pattern: /^[\d\+\-\(\)\s]+$/,
                minLength: 10
            },
            password: {
                required: true,
                minLength: 8,
                pattern: /^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)/
            }
        };

        this.init();
    }

    init() {
        // Real-time validation
        Object.keys(this.rules).forEach(fieldName => {
            let field = this.form.querySelector(`[name="${fieldName}"]`);
            if (field) {
                field.addEventListener("input", () => this.validateField(fieldName));
                field.addEventListener("blur", () => this.validateField(fieldName));
            }
        });

        // Form submit validation
        this.form.addEventListener("submit", (e) => this.handleSubmit(e));
    }

    validateField(fieldName) {
        let field = this.form.querySelector(`[name="${fieldName}"]`);
        let errorElement = this.form.querySelector(`#${fieldName}-error`);
        let rule = this.rules[fieldName];
        let value = field.value.trim();

        let errors = [];

        // Required validation
        if (rule.required && !value) {
            errors.push(`${fieldName} majburiy`);
        }

        // Pattern validation
        if (value && rule.pattern && !rule.pattern.test(value)) {
            errors.push(`${fieldName} format noto'g'ri`);
        }

        // Length validation
        if (value && rule.minLength && value.length < rule.minLength) {
            errors.push(`${fieldName} kamida ${rule.minLength} ta belgi bo'lishi kerak`);
        }

        // Show/hide errors
        if (errors.length > 0) {
            errorElement.textContent = errors[0];
            field.classList.add("invalid");
            field.classList.remove("valid");
        } else {
            errorElement.textContent = "";
            field.classList.remove("invalid");
            field.classList.add("valid");
        }

        // Password strength
        if (fieldName === "password") {
            this.updatePasswordStrength(value);
        }

        return errors.length === 0;
    }

    updatePasswordStrength(password) {
        let strengthElement = document.getElementById("strength");
        let strength = 0;
        let feedback = [];

        if (password.length >= 8) strength++;
        else feedback.push("8+ belgi");

        if (/[a-z]/.test(password)) strength++;
        else feedback.push("kichik harf");

        if (/[A-Z]/.test(password)) strength++;
        else feedback.push("katta harf");

        if (/\d/.test(password)) strength++;
        else feedback.push("raqam");

        if (/[!@#$%^&*(),.?":{}|<>]/.test(password)) strength++;
        else feedback.push("maxsus belgi");

        let levels = ["", "Juda zaif", "Zaif", "O'rtacha", "Kuchli", "Juda kuchli"];
        let colors = ["", "#ff4757", "#ff6348", "#ffa502", "#2ed573", "#00d2d3"];

        strengthElement.textContent = levels[strength];
        strengthElement.style.color = colors[strength];

        if (feedback.length > 0) {
            strengthElement.textContent += ` (Kerak: ${feedback.join(", ")})`;
        }
    }

    handleSubmit(e) {
        e.preventDefault();

        let isValid = true;

        // Validate all fields
        Object.keys(this.rules).forEach(fieldName => {
            if (!this.validateField(fieldName)) {
                isValid = false;
            }
        });

        if (isValid) {
            this.submitForm();
        } else {
            this.showFormError("Iltimos, barcha xatolarni tuzating");
        }
    }

    showFormError(message) {
        let existingError = this.form.querySelector(".form-error");
        if (existingError) {
            existingError.remove();
        }

        let errorDiv = document.createElement("div");
        errorDiv.className = "form-error";
        errorDiv.textContent = message;
        errorDiv.style.color = "red";
        errorDiv.style.marginTop = "10px";

        this.form.appendChild(errorDiv);
    }

    submitForm() {
        // Form ma'lumotlarini to'plash
        let formData = new FormData(this.form);
        let data = Object.fromEntries(formData);

        console.log("Form ma'lumotlari:", data);

        // Success message
        let successDiv = document.createElement("div");
        successDiv.className = "success-message";
        successDiv.textContent = "Ma'lumotlar muvaffaqiyatli jo'natildi!";
        successDiv.style.color = "green";
        successDiv.style.marginTop = "10px";

        this.form.appendChild(successDiv);

        // Form ni tozalash
        setTimeout(() => {
            this.form.reset();
            successDiv.remove();
            this.form.querySelectorAll(".valid, .invalid").forEach(el => {
                el.classList.remove("valid", "invalid");
            });
            this.form.querySelectorAll(".error").forEach(el => {
                el.textContent = "";
            });
        }, 3000);
    }
}

// Validatorni ishga tushirish
document.addEventListener("DOMContentLoaded", () => {
    new FormValidator("#user-form");
});
```

```javascript
class ImageGallery {
    constructor(containerId) {
        this.container = document.getElementById(containerId);
        this.modal = document.getElementById("modal");
        this.modalImg = document.getElementById("modal-img");
        this.modalTitle = document.getElementById("modal-title");
        this.modalDescription = document.getElementById("modal-description");

        this.images = [
            { id: 1, url: "nature1.jpg", title: "Tog'lar", description: "Go'zal tog' manzarasi", category: "nature" },
            { id: 2, url: "city1.jpg", title: "Shahar", description: "Kechki shahar ko'rinishi", category: "city" },
            { id: 3, url: "people1.jpg", title: "Oila", description: "Baxtli oila", category: "people" },
            // ... ko'proq rasmlar
        ];

        this.currentFilter = "all";
        this.init();
    }

    init() {
        this.renderGallery();
        this.setupEventListeners();
    }

    setupEventListeners() {
        // Filter tugmalari
        let filterBtns = document.querySelectorAll(".filter-btn");
        filterBtns.forEach(btn => {
            btn.addEventListener("click", () => {
                filterBtns.forEach(b => b.classList.remove("active"));
                btn.classList.add("active");
                this.currentFilter = btn.dataset.filter;
                this.renderGallery();
            });
        });

        // Modal yopish
        let closeBtn = document.querySelector(".close");
        closeBtn.addEventListener("click", () => this.closeModal());

        // Modal tashqarisini bosish
        this.modal.addEventListener("click", (e) => {
            if (e.target === this.modal) {
                this.closeModal();
            }
        });

        // Klaviatura ESC
        document.addEventListener("keydown", (e) => {
            if (e.key === "Escape") {
                this.closeModal();
            }
        });
    }

    renderGallery() {
        this.container.innerHTML = "";

        let filteredImages = this.currentFilter === "all"
            ? this.images
            : this.images.filter(img => img.category === this.currentFilter);

        filteredImages.forEach(image => {
            let imageItem = this.createImageElement(image);
            this.container.appendChild(imageItem);
        });
    }

    createImageElement(image) {
        let item = document.createElement("div");
        item.className = "gallery-item";
        item.dataset.category = image.category;

        item.innerHTML = `
            <img src="${image.url}" alt="${image.title}" loading="lazy">
            <div class="overlay">
                <h4>${image.title}</h4>
                <p>${image.description}</p>
            </div>
        `;

        Item.addEventListener("click", () => this.openModal(image));

        return item;
    }

    openModal(image) {
        this.modal.style.display = "block";
        this.modalImg.src = image.url;
        this.modalImg.alt = image.title;
        this.modalTitle.textContent = image.title;
        this.modalDescription.textContent = image.description;

        // Animation
        setTimeout(() => {
            this.modal.classList.add("show");
        }, 10);
    }

    closeModal() {
        this.modal.classList.remove("show");
        setTimeout(() => {
            this.modal.style.display = "none";
        }, 300);
    }
}
```
