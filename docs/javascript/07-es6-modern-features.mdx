### Tagged Template Literals

```javascript
function highlight(strings, ...values) {
    return strings.reduce((result, string, i) => {
        let value = values[i] ? `<mark>${values[i]}</mark>` : '';
        return result + string + value;
    }, '');
}

let searchTerm = "JavaScript";
let text = "dasturlash";

let highlighted = highlight`${searchTerm} tili ${text} uchun zo'r tanlov`;
console.log(highlighted); // "<mark>JavaScript</mark> tili <mark>dasturlash</mark> uchun zo'r tanlov"

// Practical example: SQL query builder
function sql(strings, ...values) {
    let query = strings.reduce((result, string, i) => {
        let value = values[i];
        if (typeof value === 'string') {
            value = `'${value.replace(/'/g, "''")}'`; // SQL injection prevention
        }
        return result + string + (value || '');
    }, '');

    return query.trim();
}

let userId = 123;
let status = "active";
let query = sql`SELECT * FROM users WHERE id = ${userId} AND status = ${status}`;
console.log(query); // "SELECT * FROM users WHERE id = 123 AND status = 'active'"
```

## Destructuring (Qismlarга Ajratish)

### Array Destructuring

```javascript
let colors = ["qizil", "yashil", "ko'k", "sariq"];

// Eski usul
let first = colors[0];
let second = colors[1];

// ✅ Modern destructuring
let [birinchi, ikkinchi, uchinchi] = colors;
console.log(birinchi); // "qizil"
console.log(ikkinchi); // "yashil"

// Skipping elements
let [, , blue] = colors;
console.log(blue); // "ko'k"

// Rest operator
let [head, ...tail] = colors;
console.log(head); // "qizil"
console.log(tail); // ["yashil", "ko'k", "sariq"]

// Default values
let [a, b, c, d, e = "default"] = colors;
console.log(e); // "sariq" (chunki colors[4] mavjud emas)

// Swapping variables
let x = 1, y = 2;
[x, y] = [y, x];
console.log(x, y); // 2, 1

// Nested arrays
let nested = [[1, 2], [3, 4]];
let [[a1, b1], [a2, b2]] = nested;
console.log(a1, b1, a2, b2); // 1 2 3 4
```

### Arrow Function vs Regular Function

```javascript
// 'this' binding difference
let obj = {
    name: "MyObject",

    // Regular function - 'this' refers to obj
    regularMethod: function() {
        console.log("Regular:", this.name); // "MyObject"

        setTimeout(function() {
            console.log("Timeout regular:", this.name); // undefined (window object)
        }, 100);
    },

    // Arrow function - 'this' is lexical
    arrowMethod: () => {
        console.log("Arrow:", this.name); // undefined ('this' is not obj)
    },

    // Best practice: regular method with arrow callback
    modernMethod: function() {
        console.log("Modern:", this.name); // "MyObject"

        setTimeout(() => {
            console.log("Timeout arrow:", this.name); // "MyObject"
        }, 100);
    }
};

// Array methods with arrows
let numbers = [1, 2, 3, 4, 5];

// Traditional way
let doubled = numbers.map(function(n) {
    return n * 2;
});

// ✅ Modern way
let modern = numbers
    .filter(n => n % 2 === 0)
    .map(n => n * 3)
    .reduce((sum, n) => sum + n, 0);

console.log(modern); // 18 (2*3 + 4*3)
```

    constructor() {
        this.cache = new Map();
    }

            value,
            expires: Date.now() + ttl

### 2. Modern Cache Implementation

```javascript
class SmartCache {
    constructor({ maxSize = 100, ttl = Infinity, onEvict = null } = {}) {
        this.cache = new Map();
        this.maxSize = maxSize;
        this.ttl = ttl;
        this.onEvict = onEvict;
        this.accessTimes = new Map();
    }

    set(key, value, customTtl = this.ttl) {
        // Evict if needed
        if (this.cache.size >= this.maxSize) {
            this.evictLRU();
        }

        let expiry = customTtl === Infinity ? Infinity : Date.now() + customTtl;

        this.cache.set(key, { value, expiry });
        this.accessTimes.set(key, Date.now());

        return this;
    }

    get(key) {
        let item = this.cache.get(key);

        if (!item) return undefined;

        // Check expiry
        if (item.expiry !== Infinity && Date.now() > item.expiry) {
            this.delete(key);
            return undefined;
        }

        // Update access time
        this.accessTimes.set(key, Date.now());

        return item.value;
    }

    has(key) {
        return this.get(key) !== undefined;
    }

    delete(key) {
        let item = this.cache.get(key);

        if (item && this.onEvict) {
            this.onEvict(key, item.value);
        }

        this.cache.delete(key);
        this.accessTimes.delete(key);

        return this;
    }

    evictLRU() {
        // Find least recently used
        let lruKey = [...this.accessTimes.entries()]
            .reduce((lru, [key, time]) =>
                time < lru.time ? { key, time } : lru,
                { key: null, time: Infinity }
            ).key;

        if (lruKey) {
            this.delete(lruKey);
        }
    }

    clear() {
        if (this.onEvict) {
            for (let [key, item] of this.cache) {
                this.onEvict(key, item.value);
            }
        }

        this.cache.clear();
        this.accessTimes.clear();
    }

    // ES6+ features: Iterators and generators
    *entries() {
        for (let [key, item] of this.cache) {
            if (item.expiry === Infinity || Date.now() <= item.expiry) {
                yield [key, item.value];
            }
        }
    }

    *keys() {
        for (let [key] of this.entries()) {
            yield key;
        }
    }

    *values() {
        for (let [, value] of this.entries()) {
            yield value;
        }
    }

    get size() {
        // Count non-expired items
        return [...this.entries()].length;
    }

    // Proxy-based dynamic getters
    createProxy() {
        return new Proxy(this, {
            get(target, property) {
                if (property in target) {
                    return target[property];
                }

                // Dynamic getter for cache keys
                return target.get(property);
            },

            set(target, property, value) {
                if (property in target) {
                    target[property] = value;
                } else {
                    target.set(property, value);
                }
                return true;
            }
        });
    }
}

// Usage
let cache = new SmartCache({
    maxSize: 3,
    ttl: 5000,
    onEvict: (key, value) => console.log(`Evicted: ${key}`)
});

cache.set("user:1", { name: "Ali" });
cache.set("user:2", { name: "Fatima" });

// Proxy usage
let dynamicCache = cache.createProxy();
dynamicCache.user3 = { name: "Ahmad" };
console.log(dynamicCache.user3); // { name: "Ahmad" }

// Iterator usage
for (let [key, value] of cache.entries()) {
    console.log(`${key}:`, value);
}
```

    constructor() {
        this.events = new Map();
        this.onceEvents = new WeakSet();
        this.middleware = [];
    }

    // Add middleware
    use(middleware) {
        this.middleware.push(middleware);
        return this;
    }

    // Subscribe to events
    on(eventName, listener, { once = false, priority = 0 } = {}) {
        if (!this.events.has(eventName)) {
            this.events.set(eventName, []);
        }

        let listeners = this.events.get(eventName);
        let listenerData = { listener, priority, id: Symbol() };

        if (once) {
            this.onceEvents.add(listenerData);
        }

        // Insert by priority
        let insertIndex = listeners.findIndex(l => l.priority < priority);
        if (insertIndex === -1) {
            listeners.push(listenerData);
        } else {
            listeners.splice(insertIndex, 0, listenerData);
        }

        // Return unsubscribe function
        return () => this.off(eventName, listenerData.id);
    }

    // One-time listener
    once(eventName, listener, options = {}) {
        return this.on(eventName, listener, { ...options, once: true });
    }

    // Remove listeners
    off(eventName, listenerOrId) {
        if (!this.events.has(eventName)) return this;

        let listeners = this.events.get(eventName);

        if (typeof listenerOrId === 'symbol') {
            // Remove by ID
            this.events.set(eventName,
                listeners.filter(l => l.id !== listenerOrId)
            );
        } else {
            // Remove by function reference
            this.events.set(eventName,
                listeners.filter(l => l.listener !== listenerOrId)
            );
        }

        return this;
    }

    // Emit events with middleware support
    async emit(eventName, ...args) {
        let eventData = { eventName, args, canceled: false };

        // Run middleware
        for (let middleware of this.middleware) {
            await middleware(eventData);
            if (eventData.canceled) return false;
        }

        if (!this.events.has(eventName)) return true;

        let listeners = [...this.events.get(eventName)];
        let results = [];

        for (let listenerData of listeners) {
            try {
                let result = await listenerData.listener(...eventData.args);
                results.push(result);

                // Remove 'once' listeners
                if (this.onceEvents.has(listenerData)) {
                    this.off(eventName, listenerData.id);
                    this.onceEvents.delete(listenerData);
                }
            } catch (error) {
                console.error(`Event listener error for '${eventName}':`, error);
            }
        }

        return results;
    }

    // Remove all listeners
    removeAllListeners(eventName) {
        if (eventName) {
            this.events.delete(eventName);
        } else {
            this.events.clear();
        }
        return this;
    }

    // Get listener count
    listenerCount(eventName) {
        return this.events.get(eventName)?.length || 0;
    }

    // Get all event names
    eventNames() {
        return [...this.events.keys()];
    }

    // Async iterator for events
    async *waitFor(eventName, timeout = Infinity) {
        let timeoutId;

        while (true) {
            let eventPromise = new Promise(resolve => {
                let unsubscribe = this.once(eventName, (...args) => {
                    resolve({ args, timeout: false });
                });

                if (timeout !== Infinity) {
                    timeoutId = setTimeout(() => {
                        unsubscribe();
                        resolve({ timeout: true });
                    }, timeout);
                }
            });

            let result = await eventPromise;

            if (timeoutId) clearTimeout(timeoutId);

            if (result.timeout) break;

            yield result.args;
        }
    }
}

// Usage example
let emitter = new ModernEventEmitter();

// Add logging middleware
emitter.use(async ({ eventName, args }) => {
    console.log(`Event: ${eventName}`, args);
});

// Regular listener
emitter.on('user:login', (user) => {
    console.log(`User logged in: ${user.name}`);
}, { priority: 10 });

// High priority listener
emitter.on('user:login', (user) => {
    console.log(`Security check for: ${user.name}`);
}, { priority: 20 });

// One-time listener
let unsubscribe = emitter.once('user:logout', (user) => {
    console.log(`User logged out: ${user.name}`);
});

// Emit events
emitter.emit('user:login', { name: 'Ali', id: 1 });

// Async event waiting
async function waitForLogin() {
    for await (let [user] of emitter.waitFor('user:login', 5000)) {
        console.log('Received login event:', user);
        break; // Exit after first event
    }
}

waitForLogin();

// Feature detection
function supportsES6Features() {
    let features = {
        arrows: (() => { try { eval("()=>{}"); return true; } catch(e) { return false; } })(),
        classes: (() => { try { eval("class Test{}"); return true; } catch(e) { return false; } })(),
        destructuring: (() => { try { eval("let [a] = [1]"); return true; } catch(e) { return false; } })(),
        modules: typeof Symbol !== 'undefined',
        promises: typeof Promise !== 'undefined',
        generators: (function*(){}).constructor === Object.getPrototypeOf(function*(){}).constructor
    };

    return features;
}

if (!Array.prototype.includes) {
    Array.prototype.includes = function(searchElement, fromIndex = 0) {
        return this.indexOf(searchElement, fromIndex) !== -1;
    };

    String.prototype.includes = function(search, start = 0) {
        return this.indexOf(search, start) !== -1;
    };

async function loadModernFeatures() {
    let support = supportsES6Features();

    if (!support.promises) {
        await import('./polyfills/promise.js');
