# JavaScript Events va Interaksiya

Events - bu foydalanuvchi yoki brauzer tomonidan sodir bo'ladigan hodisalar. JavaScript orqali bu hodisarlarga javob қaytarish va interaqtiv web sahifalar yaratish mumkin.

## Event nima?

Event - bu web sahifada sodir bo'ladigan hodisa:
- Foydalanuvchi tugmani bosdi (click)
- Tekstni yozdi (keypress)
- Sahifa yuklandi (load)
- Forma jo'natildi (submit)
- Sichqoncha harakatlandi (mouseover)

```javascript
// Event misoli
function tugmaBosildi() {
    alert("Tugma bosildi!");
}

// HTML: <button onclick="tugmaBosildi()">Bosing</button>
```

```javascript
let form = document.getElementById("my-form");
let inputs = form.querySelectorAll("input");

// Form submit
form.addEventListener("submit", (e) => {
    e.preventDefault(); // Sahifa reload bo'lmaslik uchun

    let formData = new FormData(form);
    let data = Object.fromEntries(formData);

    console.log("Form data:", data);

    // Validation
    if (validateForm(data)) {
        console.log("Form valid - serverga jo'natish mumkin");
        // Serverga jo'natish logikasi
    } else {
        console.log("Form invalid");
    }
});

// Form reset
form.addEventListener("reset", () => {
    console.log("Form tozalandi");
});

// Input validation
inputs.forEach(input => {
    input.addEventListener("blur", (e) => {
        validateField(e.target);
    });

    input.addEventListener("focus", (e) => {
        clearFieldError(e.target);
    });
});

function validateField(field) {
    let value = field.value.trim();
    let error = "";

    switch (field.name) {
        case "username":
            if (!value) error = "Username majburiy";
            else if (value.length < 3) error = "Username kamida 3 harf bo'lishi kerak";
            break;

        case "email":
            let emailPattern = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
            if (!value) error = "Email majburiy";
            else if (!emailPattern.test(value)) error = "Email formati noto'g'ri";
            break;

        case "password":
            if (!value) error = "Parol majburiy";
            else if (value.length < 6) error = "Parol kamida 6 harf bo'lishi kerak";
            break;
    }

    showFieldError(field, error);
    return !error;
}

function showFieldError(field, error) {
    let errorElement = field.parentElement.querySelector(".error");
    if (!errorElement) {
        errorElement = document.createElement("div");
        errorElement.className = "error";
        field.parentElement.appendChild(errorElement);
    }

    errorElement.textContent = error;
    field.classList.toggle("invalid", !!error);
}

function clearFieldError(field) {
    field.classList.remove("invalid");
    let errorElement = field.parentElement.querySelector(".error");
    if (errorElement) {
        errorElement.textContent = "";
    }
}
```

### Custom Events

```javascript
// Custom event yaratish
let customEvent = new CustomEvent("myCustomEvent", {
    detail: {
        message: "Bu custom event!",
        timestamp: new Date()
    },
    bubbles: true,
    cancelable: true
});

// Custom eventga listener
document.addEventListener("myCustomEvent", (e) => {
    console.log("Custom event keldi:", e.detail);
});

// Custom eventni dispatch qilish
document.dispatchEvent(customEvent);

// Component-to-component communication misoli
class ComponentA {
    constructor() {
        this.element = document.querySelector("#component-a");
        this.setupEvents();
    }

    setupEvents() {
        this.element.addEventListener("click", () => {
            this.notify("ComponentA clicked");
        });
    }

    notify(message) {
        let event = new CustomEvent("component-message", {
            detail: { sender: "ComponentA", message }
        });
        document.dispatchEvent(event);
    }
}

class ComponentB {
    constructor() {
        this.element = document.querySelector("#component-b");
        this.listen();
    }

    listen() {
        document.addEventListener("component-message", (e) => {
            console.log(`ComponentB Received: ${e.detail.message}`);
            this.element.textContent = `Last message: ${e.detail.message}`;
        });
    }
}
```

### 1. Drag and Drop

```html
<div class="container">
    <div class="draggable-items">
        <div class="item" draggable="true" data-id="1">Element 1</div>
        <div class="item" draggable="true" data-id="2">Element 2</div>
        <div class="item" draggable="true" data-id="3">Element 3</div>
    </div>

    <div class="drop-zones">
        <div class="drop-zone" data-zone="A">Zone A</div>
        <div class="drop-zone" data-zone="B">Zone B</div>
        <div class="drop-zone" data-zone="C">Zone C</div>
    </div>
</div>
```

```javascript
class DragAndDrop {
    constructor() {
        this.draggedElement = null;
        this.init();
    }

    init() {
        // Draggable elementlar
        document.querySelectorAll(".item").forEach(item => {
            item.addEventListener("dragstart", (e) => this.handleDragStart(e));
            item.addEventListener("dragend", (e) => this.handleDragEnd(e));
        });

        // Drop zones
        document.querySelectorAll(".drop-zone").forEach(zone => {
            zone.addEventListener("dragover", (e) => this.handleDragOver(e));
            zone.addEventListener("dragenter", (e) => this.handleDragEnter(e));
            zone.addEventListener("dragleave", (e) => this.handleDragLeave(e));
            zone.addEventListener("drop", (e) => this.handleDrop(e));
        });
    }

    handleDragStart(e) {
        this.draggedElement = e.target;
        e.target.style.opacity = "0.5";

        // Ma'lumot o'rnatish
        e.dataTransfer.setData("text/plain", e.target.dataset.id);
        e.dataTransfer.effectAllowed = "move";

        console.log("Drag start:", e.target.dataset.id);
    }

    handleDragEnd(e) {
        e.target.style.opacity = "1";
        this.draggedElement = null;

        // Barcha highlight'larni olib tashlash
        document.querySelectorAll(".drop-zone").forEach(zone => {
            zone.classList.remove("drag-over");
        });
    }

    handleDragOver(e) {
        e.preventDefault(); // Drop'ni ruxsat berish uchun
        e.dataTransfer.dropEffect = "move";
    }

    handleDragEnter(e) {
        e.target.classList.add("drag-over");
    }

    handleDragLeave(e) {
        // Faqat zonadan chiqishda (ichki elementlarga o'tishda emas)
        if (!e.target.contains(e.relatedTarget)) {
            e.target.classList.remove("drag-over");
        }
    }

    handleDrop(e) {
        e.preventDefault();
        e.target.classList.remove("drag-over");

        if (this.draggedElement) {
            let itemId = e.dataTransfer.getData("text/plain");
            let zoneName = e.target.dataset.zone;

            console.log(`Item ${itemId} dropped in zone ${zoneName}`);

            // Elementni yangi joyga ko'chirish
            e.target.appendChild(this.draggedElement);

            // Animation
            this.draggedElement.style.transform = "scale(1.1)";
            setTimeout(() => {
                this.draggedElement.style.transform = "scale(1)";
            }, 200);

            this.logState();
        }
    }

    logState() {
        let state = {};
        document.querySelectorAll(".drop-zone").forEach(zone => {
            let zoneName = zone.dataset.zone;
            let items = Array.from(zone.querySelectorAll(".item")).map(item =>
                item.dataset.id
            );
            state[zoneName] = items;
        });
        console.log("Current state:", state);
    }
}

// Drag and Drop ni ishga tushirish
document.addEventListener("DOMContentLoaded", () => {
    new DragAndDrop();
});
```

### 2. Real-time Search

```html
<div class="search-container">
    <input type="text" id="search-input" placeholder="Qidiruv...">
    <div id="search-results"></div>
    <div class="search-stats">
        <span id="result-count">0</span> natija topildi
        <span id="search-time"></span>
    </div>
</div>
```

```javascript
class RealTimeSearch {
    constructor() {
        this.searchInput = document.getElementById("search-input");
        this.resultsContainer = document.getElementById("search-results");
        this.resultCount = document.getElementById("result-count");
        this.searchTime = document.getElementById("search-time");

        // Sample data
        this.data = [
            { id: 1, title: "JavaScript Asoslari", category: "Programming", tags: ["js", "web"] },
            { id: 2, title: "React Framework", category: "Frontend", tags: ["react", "ui"] },
            { id: 3, title: "Node.js Server", category: "Backend", tags: ["nodejs", "server"] },
            { id: 4, title: "HTML va CSS", category: "Frontend", tags: ["html", "css"] },
            { id: 5, title: "Database Design", category: "Backend", tags: ["db", "sql"] },
            // ... ko'proq ma'lumot
        ];

        this.init();
    }

    init() {
        let searchTimer;

        this.searchInput.addEventListener("input", (e) => {
            clearTimeout(searchTimer);

            // Debouncing - 300ms kutish
            searchTimer = setTimeout(() => {
                this.performSearch(e.target.value.trim());
            }, 300);
        });

        this.searchInput.addEventListener("focus", () => {
            this.searchInput.parentElement.classList.add("focused");
        });

        this.searchInput.addEventListener("blur", () => {
            this.searchInput.parentElement.classList.remove("focused");
        });

        this.searchInput.addEventListener("keydown", (e) => {
            this.handleKeyboardNavigation(e);
        });
    }

    performSearch(query) {
        let startTime = performance.now();

        if (!query) {
            this.showResults([], 0);
            return;
        }

        // Search algorithm
        let results = this.data.filter(item => {
            let searchText = `${item.title} ${item.category} ${item.tags.join(" ")}`.toLowerCase();
            return searchText.includes(query.toLowerCase());
        });

        // Relevance sorting
        results = results.map(item => {
            let score = 0;

            // Title match = higher score
            if (item.title.toLowerCase().includes(query.toLowerCase())) {
                score += 10;
            }

            // Exact category match
            if (item.category.toLowerCase() === query.toLowerCase()) {
                score += 5;
            }

            // Tag match
            if (item.tags.some(tag => tag.includes(query.toLowerCase()))) {
                score += 3;
            }

            return { ...item, score };
        }).sort((a, b) => b.score - a.score);

        let endTime = performance.now();
        let searchTime = endTime - startTime;

        this.showResults(results, searchTime);
    }

    showResults(results, searchTime) {
        this.resultCount.textContent = results.length;
        this.searchTime.textContent = `(${searchTime.toFixed(1)}ms)`;

        if (results.length === 0) {
            this.resultsContainer.innerHTML = "<div class='no-results'>Hech narsa topilmadi</div>";
            return;
        }

        let html = results.map((item, index) => `
            <div class="search-result ${index === 0 ? 'active' : ''}" data-id="${item.id}">
                <h4>${this.highlightText(item.title, this.searchInput.value)}</h4>
                <span class="category">${item.category}</span>
                <div class="tags">
                    ${item.tags.map(tag => `<span class="tag">${tag}</span>`).join("")}
                </div>
            </div>
        `).join("");

        this.resultsContainer.innerHTML = html;

        // Result click eventi
        this.resultsContainer.querySelectorAll(".search-result").forEach(result => {
            result.addEventListener("click", (e) => {
                let itemId = result.dataset.id;
                this.selectResult(itemId);
            });
        });
    }

    highlightText(text, query) {
        if (!query) return text;

        let regex = new RegExp(`(${query})`, "gi");
        return text.replace(regex, "<mark>$1</mark>");
    }

    handleKeyboardNavigation(e) {
        let results = this.resultsContainer.querySelectorAll(".search-result");
        let activeIndex = Array.from(results).findIndex(r => r.classList.contains("active"));

        switch (e.key) {
            case "ArrowDown":
                e.preventDefault();
                if (activeIndex < results.length - 1) {
                    results[activeIndex]?.classList.remove("active");
                    results[activeIndex + 1]?.classList.add("active");
                }
                break;

            case "ArrowUp":
                e.preventDefault();
                if (activeIndex > 0) {
                    results[activeIndex]?.classList.remove("active");
                    results[activeIndex - 1]?.classList.add("active");
                }
                break;

            case "Enter":
                if (results[activeIndex]) {
                    let itemId = results[activeIndex].dataset.id;
                    this.selectResult(itemId);
                }
                break;

            case "Escape":
                this.searchInput.blur();
                this.resultsContainer.innerHTML = "";
                break;
        }
    }

    selectResult(itemId) {
        let selectedItem = this.data.find(item => item.id == itemId);
        if (selectedItem) {
            console.log("Selected:", selectedItem);
            this.searchInput.value = selectedItem.title;
            this.resultsContainer.innerHTML = "";

            // Custom event dispatch
            let event = new CustomEvent("search-select", {
                detail: { item: selectedItem }
            });
            document.dispatchEvent(event);
        }
    }
}

// Search'ni ishga tushirish
document.addEventListener("DOMContentLoaded", () => {
    let search = new RealTimeSearch();

    // Search select eventini eshitish
    document.addEventListener("search-select", (e) => {
        console.log("Item tanlandi:", e.detail.item);
    });
});
```

```javascript
class SnakeGame {
    constructor() {
        this.canvas = document.getElementById("game-canvas");
        this.ctx = this.canvas.getContext("2d");
        this.startBtn = document.getElementById("start-btn");
        this.pauseBtn = document.getElementById("pause-btn");

        // Game state

        this.snake = [{ x: 200, y: 200 }];

        this.direction = { x: 0, y: 0 };

        this.gameRunning = false;
        this.gameLoop = null;

        this.init();
    }

    init() {
        this.setupEventListeners();
        this.render();
    }

    setupEventListeners() {
        // Game controls
        this.startBtn.addEventListener("click", () => this.startGame());
        this.pauseBtn.addEventListener("click", () => this.pauseGame());

        // Keyboard controls
        document.addEventListener("keydown", (e) => this.handleKeyPress(e));

        // Game focus
        this.canvas.addEventListener("click", () => {
            this.canvas.focus();
        });

        this.canvas.tabIndex = 0; // Focusable qilish

        // Touch controls (mobile uchun)
        let touchStartX = 0;
        let touchStartY = 0;

        this.canvas.addEventListener("touchstart", (e) => {
            e.preventDefault();
            touchStartX = e.touches[0].clientX;
            touchStartY = e.touches[0].clientY;
        });

        this.canvas.addEventListener("touchend", (e) => {
            e.preventDefault();
            let touchEndX = e.changedTouches[0].clientX;
            let touchEndY = e.changedTouches[0].clientY;

            let deltaX = touchEndX - touchStartX;
            let deltaY = touchEndY - touchStartY;

            let threshold = 50;

            if (Math.abs(deltaX) > threshold || Math.abs(deltaY) > threshold) {
                if (Math.abs(deltaX) > Math.abs(deltaY)) {
                    // Horizontal swipe
                    this.changeDirection(deltaX > 0 ? "right" : "left");
                } else {
                    // Vertical swipe
                    this.changeDirection(deltaY > 0 ? "down" : "up");
                }
            }
        });
    }

    handleKeyPress(e) {
        if (!this.gameRunning) return;

        switch (e.key) {
            case "ArrowUp":
            case "w":
            case "W":
                this.changeDirection("up");
                break;
            case "ArrowDown":
            case "s":
            case "S":
                this.changeDirection("down");
                break;
            case "ArrowLeft":
            case "a":
            case "A":
                this.changeDirection("left");
                break;
            case "ArrowRight":
            case "d":
            case "D":
                this.changeDirection("right");
                break;
            case " ":
                e.preventDefault();
                this.pauseGame();
                break;
        }
    }

    changeDirection(newDirection) {
        const directions = {
            up: { x: 0, y: -this.gridSize },
            down: { x: 0, y: this.gridSize },
            left: { x: -this.gridSize, y: 0 },
            right: { x: this.gridSize, y: 0 }
        };

        const newDir = directions[newDirection];

        // Teskari yo'nalishga o'tmaslik
        if (newDir.x !== -this.direction.x || newDir.y !== -this.direction.y) {
            this.direction = newDir;
        }
    }

    startGame() {
        if (this.gameRunning) return;

        this.gameRunning = true;
        this.startBtn.disabled = true;
        this.pauseBtn.disabled = false;

        this.gameLoop = setInterval(() => {
            this.update();
            this.render();
        }, 150); // 150ms = game speed
    }

    pauseGame() {
        this.gameRunning = false;
        this.startBtn.disabled = false;
        this.pauseBtn.disabled = true;

        if (this.gameLoop) {
            clearInterval(this.gameLoop);
            this.gameLoop = null;
        }
    }

    update() {
        // Snake boshi
        const head = { ...this.snake[0] };
        head.x += this.direction.x;
        head.y += this.direction.y;

        // Wall collision
        if (head.x < 0 || head.x >= this.canvas.width ||
            head.y < 0 || head.y >= this.canvas.height) {
            this.gameOver();
            return;
        }

        // Self collision
        if (this.snake.some(segment => segment.x === head.x && segment.y === head.y)) {
            this.gameOver();
            return;
        }

        this.snake.unshift(head);

        // Food check
        if (head.x === this.food.x && head.y === this.food.y) {
            this.score += 10;
            this.scoreElement.textContent = this.score;
            this.food = this.generateFood();

            // Snake grow (don't remove tail)
        } else {
            // Remove tail
            this.snake.pop();
        }
    }

    generateFood() {
        let food;
        do {
            food = {
                x: Math.floor(Math.random() * (this.canvas.width / this.gridSize)) * this.gridSize,
                y: Math.floor(Math.random() * (this.canvas.height / this.gridSize)) * this.gridSize
            };
        } while (this.snake.some(segment => segment.x === food.x && segment.y === food.y));

        return food;
    }

    render() {
        // Clear canvas
        this.ctx.fillStyle = "#000";
        this.ctx.fillRect(0, 0, this.canvas.width, this.canvas.height);

        // Draw snake
        this.ctx.fillStyle = "#0f0";
        this.snake.forEach((segment, index) => {
            if (index === 0) {
                // Head
                this.ctx.fillStyle = "#090";
            } else {
                this.ctx.fillStyle = "#0f0";
            }
            this.ctx.fillRect(segment.x, segment.y, this.gridSize - 2, this.gridSize - 2);
        });

        // Draw food
        this.ctx.fillStyle = "#f00";
        this.ctx.fillRect(this.food.x, this.food.y, this.gridSize - 2, this.gridSize - 2);

        // Draw grid (optional)
        this.ctx.strokeStyle = "#333";
        this.ctx.lineWidth = 1;
        for (let x = 0; x <= this.canvas.width; x += this.gridSize) {
            this.ctx.beginPath();
            this.ctx.moveTo(x, 0);
            this.ctx.lineTo(x, this.canvas.height);
            this.ctx.stroke();
        }
        for (let y = 0; y <= this.canvas.height; y += this.gridSize) {
            this.ctx.beginPath();
            this.ctx.moveTo(0, y);
            this.ctx.lineTo(this.canvas.width, y);
            this.ctx.stroke();
        }
    }

    gameOver() {
        this.pauseGame();

        // Game over animation
        this.ctx.fillStyle = "rgba(255, 0, 0, 0.5)";
        this.ctx.fillRect(0, 0, this.canvas.width, this.canvas.height);

        this.ctx.fillStyle = "#fff";
        this.ctx.font = "32px Arial";
        this.ctx.textAlign = "center";
        this.ctx.fillText("O'YIN TUGADI", this.canvas.width / 2, this.canvas.height / 2 - 20);

        this.ctx.font = "16px Arial";
        this.ctx.fillText(`Score: ${this.score}`, this.canvas.width / 2, this.canvas.height / 2 + 20);

        this.ctx.fillText("Qayta boshlash uchun Start bosing", this.canvas.width / 2, this.canvas.height / 2 + 50);

        // Reset game state
        setTimeout(() => {
            this.resetGame();
        }, 3000);
    }

    resetGame() {
        this.snake = [{ x: 200, y: 200 }];
        this.food = this.generateFood();
        this.direction = { x: 0, y: 0 };
        this.score = 0;
        this.scoreElement.textContent = this.score;
        this.render();
    }
}

// O'yinni ishga tushirish
document.addEventListener("DOMContentLoaded", () => {
    new SnakeGame();
});
```

```javascript
// ❌ Heavy function to'g'ridan-to'g'ri
input.addEventListener("input", heavyFunction);

// ✅ Debouncing
function debounce(func, delay) {
    let timer;
    return function (...args) {
        clearTimeout(timer);
        timer = setTimeout(() => func.apply(this, args), delay);
    };
}

input.addEventListener("input", debounce(heavyFunction, 300));

// ✅ Throttling
function throttle(func, limit) {
    let inThrottle;
    return function (...args) {
        if (!inThrottle) {
            func.apply(this, args);
            inThrottle = true;
            setTimeout(() => inThrottle = false, limit);
        }
    };
}

window.addEventListener("scroll", throttle(onScroll, 100));
```
