### Private Fields va Methods (ES2022)

```javascript
class BankAccount {
    // Private fields
    #balance = 0;
    #accountNumber;
    #pin;

    constructor(accountNumber, pin) {
        this.#accountNumber = accountNumber;
        this.#pin = pin;
    }

    // Private method
    #validatePin(pin) {
        return this.#pin === pin;
    }

    // Public methods
    deposit(amount, pin) {
        if (!this.#validatePin(pin)) {
            throw new Error("Noto'g'ri PIN kod");
        }

        if (amount <= 0) {
            throw new Error("Miqdor musbat bo'lishi kerak");
        }

        this.#balance += amount;
        console.log(`${amount} so'm qo'shildi. Balans: ${this.#balance}`);
    }

    withdraw(amount, pin) {
        if (!this.#validatePin(pin)) {
            throw new Error("Noto'g'ri PIN kod");
        }

        if (amount > this.#balance) {
            throw new Error("Yetarli mablag' yo'q");
        }

        this.#balance -= amount;
        console.log(`${amount} so'm yechildi. Balans: ${this.#balance}`);
    }

    getBalance(pin) {
        if (!this.#validatePin(pin)) {
            throw new Error("Noto'g'ri PIN kod");
        }
        return this.#balance;
    }

    // Static private method
    static #generateAccountNumber() {
        return 'ACC' + Math.random().toString(36).substr(2, 9).toUpperCase();
    }

    static createAccount(pin) {
        let accountNumber = this.#generateAccountNumber();
        return new BankAccount(accountNumber, pin);
    }
}
```

### Inheritance (Meros)

```javascript
// Base class
class Animal {
    constructor(name, species) {
        this.name = name;
        this.species = species;
        this.energy = 100;
    }

    eat(food) {
        this.energy += 20;
        console.log(`${this.name} ${food} yedi. Energiya: ${this.energy}`);
    }

    sleep() {
        this.energy = 100;
        console.log(`${this.name} uxladi. Energiya to'liq tiklandi.`);
    }

        console.log(`${this.name} tovush chiqaradi`);
    }
}

// Derived class
class Dog extends Animal {
    constructor(name, breed) {
        super(name, "Dog"); // Parent constructor call
        this.breed = breed;
        this.tricks = [];
    }

    // Override parent method
    makeSound() {
        console.log(`${this.name} vovullaydi: Vov vov!`);
    }

    // New methods
    bark() {
        this.makeSound(); // Call overridden method
    }

    learnTrick(trick) {
        this.tricks.push(trick);
        console.log(`${this.name} yangi hiyla o'rgandi: ${trick}`);
    }

    performTrick() {
        if (this.tricks.length === 0) {
            console.log(`${this.name} hali hiyla bilmaydi`);
            return;
        }

        let randomTrick = this.tricks[Math.floor(Math.random() * this.tricks.length)];
        console.log(`${this.name} hiyla ko'rsatdi: ${randomTrick}`);
    }
}

class Cat extends Animal {
    constructor(name, color) {
        super(name, "Cat");
        this.color = color;
        this.livesRemaining = 9;
    }

    makeSound() {
        console.log(`${this.name} miyovlaydi: Miyov miyov!`);
    }

    climb() {
        if (this.energy < 30) {
            console.log(`${this.name} charchagan, ko'tarolmaydi`);
            return;
        }

        this.energy -= 30;
        console.log(`${this.name} daraxtga chiqdi! Energiya: ${this.energy}`);
    }
}
```

```javascript
class Logger {
    constructor() {
        if (Logger.instance) {
            return Logger.instance;
        }

        this.logs = [];
        this.level = 'info';
        Logger.instance = this;
    }

    setLevel(level) {
        this.level = level;

    log(level, message, data = null) {
        if (this.shouldLog(level)) {
            let logEntry = {
                timestamp: new Date().toISOString(),
                level,
                message,
                data
            };

            this.logs.push(logEntry);
            console.log(`[${level.toUpperCase()}] ${message}`, data || '');
        }

    shouldLog(level) {
        let levels = { error: 0, warn: 1, info: 2, debug: 3 };
        return levels[level] <= levels[this.level];
    }

        this.log('info', message, data);
    }

    warn(message, data) {
        this.log('warn', message, data);
    }

    error(message, data) {
        this.log('error', message, data);
    }

    debug(message, data) {
        this.log('debug', message, data);
    }
}

// apiClient.js - Module with dependencies
import config from './config.js';
import logger from './logger.js';

class APIClient {
    constructor() {
        this.baseUrl = config.apiUrl;
        this.cache = new Map();
    }

    async request(endpoint, options = {}) {
        let url = `${this.baseUrl}${endpoint}`;
        let cacheKey = `${options.method || 'GET'}:${url}`;

        // Check cache
        if (config.cacheEnabled && options.method !== 'POST') {
            let cached = this.cache.get(cacheKey);
            if (cached && Date.now() - cached.timestamp < 60000) { // 1 minute
                logger.debug('Cache hit', { endpoint, cached: true });
                return cached.data;
            }
        }

        try {
            logger.info('API request', { method: options.method || 'GET', url });

            let response = await fetch(url, {
                headers: {
                    'Content-Type': 'application/json',
                    ...options.headers
                },
                ...options
            });

            if (!response.ok) {
                throw new Error(`HTTP ${response.status}: ${response.statusText}`);
            }

            let data = await response.json();

            // Cache successful GET requests
            if (config.cacheEnabled && (!options.method || options.method === 'GET')) {
                this.cache.set(cacheKey, {
                    data,
                    timestamp: Date.now()
                });
            }

            logger.debug('API response', { endpoint, status: response.status });
            return data;
        } catch (error) {
            logger.error('API error', { endpoint, error: error.message });
            throw error;
        }
    }

    get(endpoint, params = {}) {
        let queryString = new URLSearchParams(params).toString();
        let url = queryString ? `${endpoint}?${queryString}` : endpoint;
        return this.request(url);
    }

    post(endpoint, data) {
        return this.request(endpoint, {
            method: 'POST',
            body: JSON.stringify(data)
        });
    }

    put(endpoint, data) {
        return this.request(endpoint, {
            method: 'PUT',
            body: JSON.stringify(data)
        });
    }

    delete(endpoint) {
        return this.request(endpoint, {
            method: 'DELETE'
        });
    }
}

export default new APIClient();
```

### Observer Pattern

```javascript
// Event emitter pattern
class EventEmitter {
    constructor() {
        this.events = new Map();
    }

    on(eventName, callback) {
        if (!this.events.has(eventName)) {
            this.events.set(eventName, []);
        }
        this.events.get(eventName).push(callback);

        // Return unsubscribe function
        return () => this.off(eventName, callback);
    }

    off(eventName, callback) {
        if (!this.events.has(eventName)) return;

        let callbacks = this.events.get(eventName);
        let index = callbacks.indexOf(callback);
        if (index !== -1) {
            callbacks.splice(index, 1);
        }
    }

    emit(eventName, ...args) {
        if (!this.events.has(eventName)) return;

        this.events.get(eventName).forEach(callback => {
            try {
                callback(...args);
            } catch (error) {
                console.error(`Error in event listener for "${eventName}":`, error);
            }
        });
    }

    once(eventName, callback) {
        let unsubscribe = this.on(eventName, (...args) => {
            unsubscribe();
            callback(...args);
        });
    }
}

// Observable data store
class Store extends EventEmitter {
    constructor(initialState = {}) {
        super();
        this.state = { ...initialState };
    }

    getState() {
        return { ...this.state };
    }

    setState(newState) {
        let oldState = { ...this.state };
        this.state = { ...this.state, ...newState };

        this.emit('stateChanged', this.state, oldState);

        // Emit specific property changes
        for (let key in newState) {
            if (oldState[key] !== newState[key]) {
                this.emit(`${key}Changed`, newState[key], oldState[key]);
            }
        }
    }

    subscribe(callback) {
        return this.on('stateChanged', callback);
    }
}

// Usage
let store = new Store({ count: 0, user: null });

// Subscribe to changes
let unsubscribe = store.subscribe((newState, oldState) => {
    console.log('State changed:', { newState, oldState });
});

store.on('countChanged', (newCount, oldCount) => {
    console.log(`Count: ${oldCount} → ${newCount}`);
});

store.setState({ count: 5 });         // Triggers both callbacks
store.setState({ user: 'Ali' });      // Triggers state change callback
```

### Factory Pattern

```javascript
// Abstract factory for creating different types of users
class UserFactory {
    static createUser(type, data) {
        switch (type) {
            case 'admin':
                return new AdminUser(data);
            case 'moderator':
                return new ModeratorUser(data);
            case 'regular':
                return new RegularUser(data);
            default:
                throw new Error(`Unknown user type: ${type}`);
        }
    }

    static fromJSON(json) {
        let data = typeof json === 'string' ? JSON.parse(json) : json;
        return this.createUser(data.type, data);
    }
}

// Base user class
class User {
    constructor(data) {
        this.id = data.id || Date.now();
        this.name = data.name;
        this.email = data.email;
        this.createdAt = data.createdAt || new Date();
    }

    toJSON() {
        return {
            id: this.id,
            name: this.name,
            email: this.email,
            type: this.constructor.name.replace('User', '').toLowerCase(),
            createdAt: this.createdAt,
            permissions: this.getPermissions()

    getPermissions() {
        return ['read'];

        return this.getPermissions().includes(permission);
    }
}

class RegularUser extends User {

        return ['read', 'comment'];
    }
}
```

### Strategy Pattern

```javascript
// Payment processing with different strategies
class PaymentStrategy {
    process(amount) {
        throw new Error('process() method must be implemented');
    }

    validate() {
        throw new Error('validate() method must be implemented');
    }
}

class CreditCardPayment extends PaymentStrategy {
    constructor(cardNumber, expiryDate, cvv) {
        super();
        this.cardNumber = cardNumber;
        this.expiryDate = expiryDate;
        this.cvv = cvv;
    }

    validate() {
        // Simple validation
        if (!this.cardNumber || this.cardNumber.length < 13) {
            throw new Error('Invalid card number');
        }
        if (!this.cvv || this.cvv.length < 3) {
            throw new Error('Invalid CVV');
        }
        return true;
    }

    async process(amount) {
        this.validate();

        // Simulate API call
        await new Promise(resolve => setTimeout(resolve, 1000));

        console.log(`Processing $${amount} via Credit Card ending in ${this.cardNumber.slice(-4)}`);
        return {
            success: true,
            transactionId: 'cc_' + Date.now(),
            method: 'credit_card'
        };
    }
}

class PayPalPayment extends PaymentStrategy {
    constructor(email, password) {
        super();
        this.email = email;
        this.password = password;
    }

    validate() {
        if (!this.email || !this.email.includes('@')) {
            throw new Error('Invalid PayPal email');
        }
        return true;
    }

    async process(amount) {
        this.validate();

        await new Promise(resolve => setTimeout(resolve, 800));

        console.log(`Processing $${amount} via PayPal account ${this.email}`);
        return {
            success: true,
            transactionId: 'pp_' + Date.now(),
            method: 'paypal'
        };
    }
}

class BitcoinPayment extends PaymentStrategy {
    constructor(walletAddress) {
        super();
        this.walletAddress = walletAddress;
    }

    validate() {
        if (!this.walletAddress || this.walletAddress.length < 20) {
            throw new Error('Invalid Bitcoin wallet address');
        }
        return true;
    }

    async process(amount) {
        this.validate();

        await new Promise(resolve => setTimeout(resolve, 2000)); // Bitcoin is slower

        console.log(`Processing $${amount} via Bitcoin to ${this.walletAddress}`);
        return {
            success: true,
            transactionId: 'btc_' + Date.now(),
            method: 'bitcoin'
        };
    }
}

// Payment processor
class PaymentProcessor {
    constructor() {
        this.strategies = new Map();
        this.registerStrategy('credit_card', CreditCardPayment);
        this.registerStrategy('paypal', PayPalPayment);
        this.registerStrategy('bitcoin', BitcoinPayment);
    }

    registerStrategy(name, strategyClass) {
        this.strategies.set(name, strategyClass);
    }

    createPaymentMethod(type, ...args) {
        let StrategyClass = this.strategies.get(type);
        if (!StrategyClass) {
            throw new Error(`Unknown payment method: ${type}`);
        }
        return new StrategyClass(...args);
    }

    async processPayment(amount, paymentMethod) {
        try {
            let result = await paymentMethod.process(amount);
            console.log('Payment successful:', result);
            return result;
        } catch (error) {
            console.error('Payment failed:', error.message);
            throw error;
        }
    }
}
```

```javascript

    }

        return this.coffee.description();
    }
}

// Concrete decorators

    cost() {
        return this.coffee.cost() + 0.5;
    }

    description() {
        return this.coffee.description() + ', milk';
    }
}

class SugarDecorator extends CoffeeDecorator {
    cost() {
        return this.coffee.cost() + 0.2;
    }

    description() {
        return this.coffee.description() + ', sugar';
    }
}

class WhipCreamDecorator extends CoffeeDecorator {
    cost() {
        return this.coffee.cost() + 1.0;
    }

    description() {
        return this.coffee.description() + ', whip cream';
    }
}

// Function decorator pattern for JavaScript
function withLogging(target, propertyName, descriptor) {
    let originalMethod = descriptor.value;

    descriptor.value = function(...args) {
        console.log(`Calling ${propertyName} with:`, args);
        let result = originalMethod.apply(this, args);
        console.log(`${propertyName} returned:`, result);
        return result;
    };

    return descriptor;
}

function withTimer(target, propertyName, descriptor) {
    let originalMethod = descriptor.value;

    descriptor.value = function(...args) {
        let start = performance.now();
        let result = originalMethod.apply(this, args);
        let end = performance.now();
        console.log(`${propertyName} took ${(end - start).toFixed(2)}ms`);
        return result;
    };

    return descriptor;
}

// Usage with coffee
let coffee = new SimpleCoffee();
coffee = new MilkDecorator(coffee);
coffee = new SugarDecorator(coffee);
coffee = new WhipCreamDecorator(coffee);

console.log(`${coffee.description()} costs $${coffee.cost()}`);
// "Simple coffee, milk, sugar, whip cream costs $3.7"

// Class with decorators (requires transpiler like Babel for current support)
class Calculator {
    @withLogging
    @withTimer
    add(a, b) {
        return a + b;
    }

    @withLogging
    multiply(a, b) {
        // Simulate slow operation
        let result = 0;
        for (let i = 0; i < 1000000; i++) {
            result = a * b;
        }
        return result;
    }
}

// Manual decoration without decorator syntax
function decorateMethod(obj, methodName, decorators) {
    let originalMethod = obj[methodName];

    decorators.reverse().forEach(decorator => {
        let descriptor = { value: originalMethod };
        originalMethod = decorator(obj, methodName, descriptor).value;
    });

    obj[methodName] = originalMethod;
}

let calc = new Calculator();
decorateMethod(calc, 'add', [withTimer, withLogging]);
decorateMethod(calc, 'multiply', [withLogging]);

calc.add(5, 3);        // Logs timing and arguments
calc.multiply(4, 6);   // Logs arguments and result
```

```javascript
class SmartThermostat {
    // Private fields
    #temperature = 20;
    #targetTemperature = 22;
    #isHeating = false;
    #isCooling = false;
    #schedule = new Map();

    constructor(initialTemp = 20) {
        this.#temperature = initialTemp;
        this.#targetTemperature = initialTemp;
        this.#startMainLoop();
    }

    // Getters
    get temperature() {
        return this.#temperature;
    }

    get targetTemperature() {
        return this.#targetTemperature;
    }

    get isHeating() {
        return this.#isHeating;
    }

    get isCooling() {
        return this.#isCooling;
    }

    get status() {
        if (this.#isHeating) return 'heating';
        if (this.#isCooling) return 'cooling';
        return 'idle';
    }

    // Setters with validation
    set targetTemperature(temp) {
        if (typeof temp !== 'number') {
            throw new Error('Temperature must be a number');
        }
        if (temp < 10 || temp > 30) {
            throw new Error('Temperature must be between 10-30°C');
        }

        console.log(`Target temperature set to ${temp}°C`);
        this.#targetTemperature = temp;
        this.#updateSystem();
    }

    // Private methods
    #updateSystem() {
        let diff = this.#targetTemperature - this.#temperature;

        if (Math.abs(diff) < 0.5) {
            this.#isHeating = false;
            this.#isCooling = false;
            console.log('Temperature reached target');
        } else if (diff > 0) {
            this.#isHeating = true;
            this.#isCooling = false;
            console.log('Heating system activated');
        } else {
            this.#isHeating = false;
            this.#isCooling = true;
            console.log('Cooling system activated');
        }
    }

    #startMainLoop() {
        setInterval(() => {
            if (this.#isHeating) {
                this.#temperature += 0.1;
            } else if (this.#isCooling) {
                this.#temperature -= 0.1;
            }

            this.#updateSystem();
            this.#checkSchedule();
        }, 1000);
    }

    #checkSchedule() {
        let now = new Date();
        let timeKey = `${now.getHours()}:${now.getMinutes().toString().padStart(2, '0')}`;

        if (this.#schedule.has(timeKey)) {
            let scheduledTemp = this.#schedule.get(timeKey);
            console.log(`Scheduled temperature change: ${scheduledTemp}°C`);
            this.targetTemperature = scheduledTemp;
        }
    }

    // Public methods
    setSchedule(time, temperature) {
        this.#schedule.set(time, temperature);
        console.log(`Scheduled: ${temperature}°C at ${time}`);
    }

    getSchedule() {
        return new Map(this.#schedule); // Return copy
    }

    clearSchedule() {
        this.#schedule.clear();
        console.log('Schedule cleared');
    }

    getReport() {
        return {
            current: this.#temperature.toFixed(1),
            target: this.#targetTemperature,
            status: this.status,
            schedule: Object.fromEntries(this.#schedule)
        };
    }
}

thermostat.setSchedule('22:00', 20); // Night time
thermostat.setSchedule('08:00', 22); // Morning

console.log(thermostat.getReport());
```

```javascript
class Product {
    constructor(id, name, price, category) {
        this.id = id;
        this.name = name;
        this.price = price;
        this.category = category;
        this.reviews = [];
    }

    addReview(rating, comment, reviewerName) {
        this.reviews.push({
            rating,
            comment,
            reviewerName,

        });

        };
    }

// Shopping cart
class ShoppingCart {
    constructor() {
        this.items = [];
        this.discounts = [];
    }

    addItem(product, quantity = 1) {
        let existingItem = this.items.find(item => item.product.id === product.id);

        if (existingItem) {
            existingItem.quantity += quantity;
        } else {
            this.items.push({ product, quantity });
        }

        console.log(`Added ${quantity}x ${product.name} to cart`);
    }

    removeItem(productId, quantity = null) {
        let itemIndex = this.items.findIndex(item => item.product.id === productId);

        if (itemIndex === -1) return;

        if (quantity === null) {
            // Remove entire item
            this.items.splice(itemIndex, 1);
        } else {
            // Remove specific quantity
            this.items[itemIndex].quantity -= quantity;
            if (this.items[itemIndex].quantity <= 0) {
                this.items.splice(itemIndex, 1);
            }
        }
    }

    getTotal() {
        let subtotal = this.items.reduce((total, item) => {
            return total + (item.product.price * item.quantity);
        }, 0);

        let discountAmount = this.discounts.reduce((total, discount) => {
            return total + discount.calculate(subtotal);
        }, 0);

        return Math.max(0, subtotal - discountAmount);
    }

    applyDiscount(discount) {
        this.discounts.push(discount);
    }

    clear() {
        this.items = [];
        this.discounts = [];
    }

    getSummary() {
        return {
            items: this.items.map(item => ({
                product: item.product.getDetails(),
                quantity: item.quantity,
                subtotal: item.product.price * item.quantity
            })),
            total: this.getTotal(),
            itemCount: this.items.length
        };
    }
}

// Discount strategies
class PercentageDiscount {
    constructor(percentage) {
        this.percentage = percentage;
    }

    calculate(subtotal) {
        return subtotal * (this.percentage / 100);
    }
}

class FixedDiscount {
    constructor(amount) {
        this.amount = amount;
    }

    calculate(subtotal) {
        return Math.min(this.amount, subtotal);
    }
}

// Order management
class Order {
    static #nextOrderId = 1;

    constructor(cart, customer) {
        this.id = Order.#nextOrderId++;
        this.customer = customer;
        this.items = [...cart.items];
        this.total = cart.getTotal();
        this.status = 'pending';
        this.createdAt = new Date();
        this.shippingAddress = null;
    }

    setShippingAddress(address) {
        this.shippingAddress = address;
    }

    updateStatus(newStatus) {
        this.status = newStatus;
        console.log(`Order ${this.id} status updated to: ${newStatus}`);
    }

    getTrackingInfo() {
        return {
            orderId: this.id,
            status: this.status,
            createdAt: this.createdAt,
            estimatedDelivery: this.#calculateEstimatedDelivery()
        };
    }

    #calculateEstimatedDelivery() {
        let deliveryDate = new Date(this.createdAt);

        // Physical products take longer
        let hasPhysical = this.items.some(item =>
            item.product instanceof PhysicalProduct
        );

        if (hasPhysical) {
            deliveryDate.setDate(deliveryDate.getDate() + 5); // 5 days
        } else {
            deliveryDate = this.createdAt; // Digital products available immediately
        }

        return deliveryDate;
    }
}
```
