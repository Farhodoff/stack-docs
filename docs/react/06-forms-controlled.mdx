# Formalar va Controlled Components

React'da formalar HTML formalaridan farqli ishlaydi. React'da form elementlari **controlled components** bo'lib, ularning qiymati React state orqali boshqariladi.

## 1. Controlled vs Uncontrolled Components

### Controlled Components:
- Form elementlarining qiymati React state'da saqlanadi
- `value` prop orqali qiymat o'rnatiladi
- `onChange` event orqali o'zgarishlar kuzatiladi
- React form ma'lumotlarini butunlay boshqaradi

### Uncontrolled Components:
- Form elementlari o'z ichki DOM state'ini saqlaydi
- `ref` orqali qiymatga murojaat qilinadi
- HTML form'lariga o'xshash ishlaydi

### Text input:

```jsx
import React, { useState } from 'react';

function BasicForm() {
  const [formData, setFormData] = useState({
    name: '',
    email: '',
    phone: ''
  });

  const handleInputChange = (event) => {
    const { name, value } = event.target;
    setFormData(prev => ({
      ...prev,
      [name]: value
    }));
  };

  const handleSubmit = (event) => {
    event.preventDefault();
    console.log('Form yuborildi:', formData);
    alert(`Salom, ${formData.name}! Email: ${formData.email}`);

  return (
    <form onSubmit={handleSubmit}>
      <h2>Asosiy Forma</h2>

      <div>
        <label htmlFor="name">Ismi:</label>
        <input
          type="text"
          id="name"
          name="name"
          value={formData.name}
          onChange={handleInputChange}
          placeholder="To'liq ismingiz"
          required
        />
      </div>

      <div>
        <label htmlFor="email">Email:</label>
        <input
          type="email"
          id="email"
          name="email"
          value={formData.email}
          onChange={handleInputChange}
          placeholder="email@example.com"
          required
        />
      </div>

      <div>
        <label htmlFor="phone">Telefon:</label>
        <input
          type="tel"
          id="phone"
          name="phone"
          value={formData.phone}
          onChange={handleInputChange}

      </div>
```

## 4. Form Validation

Client-side form validation implementing:

```jsx
function ValidationForm() {
  const [formData, setFormData] = useState({
    username: '',
    email: '',
    password: '',
    confirmPassword: '',
    age: '',
    website: '',
    bio: ''
  });

  const [errors, setErrors] = useState({});
  const [touched, setTouched] = useState({});
  const [isSubmitting, setIsSubmitting] = useState(false);

  // Validation rules
  const validationRules = {
    username: {
      required: true,
      minLength: 3,
      maxLength: 20,
      pattern: /^[a-zA-Z0-9_]+$/,
      message: "Username kamida 3 ta belgi bo'lishi va faqat harflar, raqamlar va pastki chiziq bo'lishi kerak"
    },
    email: {
      required: true,
      pattern: /^[^\s@]+@[^\s@]+\.[^\s@]+$/,
      message: "Haqiqiy email manzilini kiriting"
    },
    password: {
      required: true,
      minLength: 8,
      pattern: /^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)(?=.*[@$!%*?&])[A-Za-z\d@$!%*?&]/,
      message: "Parol kamida 8 ta belgi, kichik va katta harf, raqam va maxsus belgi bo'lishi kerak"
    },
    confirmPassword: {
      required: true,
      mustMatch: 'password',
      message: "Parollar bir xil bo'lishi kerak"
    },
    age: {
      required: true,
      min: 18,
      max: 120,
      message: "Yosh 18 dan 120 gacha bo'lishi kerak"
    },
    website: {
      pattern: /^https?:\/\/.+/,
      message: "Haqiqiy URL kiriting (http:// yoki https:// bilan)"
    }
  };

  // Validate field
  const validateField = (name, value) => {
    const rule = validationRules[name];
    if (!rule) return '';

    if (rule.required && (!value || value.trim() === '')) {
      return `${name} majburiy`;
    }

    if (value && rule.minLength && value.length < rule.minLength) {
      return `${name} kamida ${rule.minLength} ta belgi bo'lishi kerak`;
    }

    if (value && rule.maxLength && value.length > rule.maxLength) {
      return `${name} ko'pi bilan ${rule.maxLength} ta belgi bo'lishi kerak`;
    }

    if (value && rule.min && parseFloat(value) < rule.min) {
      return `${name} kamida ${rule.min} bo'lishi kerak`;
    }

    if (value && rule.max && parseFloat(value) > rule.max) {
      return `${name} ko'pi bilan ${rule.max} bo'lishi kerak`;
    }

    if (value && rule.pattern && !rule.pattern.test(value)) {
      return rule.message || `${name} formati noto'g'ri`;
    }

    if (rule.mustMatch && value !== formData[rule.mustMatch]) {
      return rule.message || `${name} mos kelmaydi`;
    }

    return '';
  };

  // Handle input change
  const handleChange = (event) => {
    const { name, value } = event.target;

    setFormData(prev => ({
      ...prev,
      [name]: value
    }));

    // Real-time validation
    if (touched[name]) {
      const error = validateField(name, value);
      setErrors(prev => ({
        ...prev,
        [name]: error
      }));
    }
  };

  // Handle blur (field touched)
  const handleBlur = (event) => {
    const { name, value } = event.target;

    setTouched(prev => ({
      ...prev,
      [name]: true
    }));

    const error = validateField(name, value);
    setErrors(prev => ({
      ...prev,
      [name]: error
    }));
  };

  // Validate all fields
  const validateAll = () => {
    const newErrors = {};
    Object.keys(validationRules).forEach(field => {
      const error = validateField(field, formData[field]);
      if (error) {
        newErrors[field] = error;
      }
    });
    return newErrors;
  };

  // Handle submit
  const handleSubmit = async (event) => {
    event.preventDefault();

    setIsSubmitting(true);

    // Mark all fields as touched
    const allFields = Object.keys(validationRules);
    setTouched(allFields.reduce((acc, field) => ({ ...acc, [field]: true }), {}));

    // Validate all fields
    const validationErrors = validateAll();
    setErrors(validationErrors);

    // If there are errors, stop submission
    if (Object.keys(validationErrors).some(key => validationErrors[key])) {
      setIsSubmitting(false);
      return;
    }

    try {
      // Simulate API call
      await new Promise(resolve => setTimeout(resolve, 2000));

      alert('Forma muvaffaqiyatli yuborildi!');

      // Reset form
      setFormData({
        username: '',
        email: '',
        password: '',
        confirmPassword: '',
        age: '',
        website: '',
        bio: ''
      });
      setErrors({});
      setTouched({});

    } catch (error) {
      alert('Xatolik yuz berdi. Qaytadan urinib ko\'ring.');
    } finally {
      setIsSubmitting(false);
    }
  };

  // Check if form is valid
  const isFormValid = Object.keys(validationRules).every(field =>
    !validateField(field, formData[field])
  ) && Object.keys(validationRules).every(field =>
    formData[field] && formData[field].trim() !== ''
  );

  return (
    <form onSubmit={handleSubmit} style={{ maxWidth: '500px', margin: '0 auto' }}>
      <h2>Validation bilan Forma</h2>

      {/* Username */}
      <div className="form-group">
        <label htmlFor="username">Username *</label>
        <input
          type="text"
          id="username"
          name="username"
          value={formData.username}
          onChange={handleChange}
          onBlur={handleBlur}
          className={errors.username ? 'error' : ''}
          placeholder="myusername123"
        />
        {touched.username && errors.username && (
          <small className="error-message">{errors.username}</small>
        )}
      </div>

      {/* Email */}
      <div className="form-group">
        <label htmlFor="email">Email *</label>
        <input
          type="email"
          id="email"
          name="email"
          value={formData.email}
          onChange={handleChange}
          onBlur={handleBlur}
          className={errors.email ? 'error' : ''}
          placeholder="email@example.com"
        />
        {touched.email && errors.email && (
          <small className="error-message">{errors.email}</small>
        )}
      </div>

      {/* Password */}
      <div className="form-group">
        <label htmlFor="password">Parol *</label>
        <input
          type="password"
          id="password"
          name="password"
          value={formData.password}
          onChange={handleChange}
          onBlur={handleBlur}
          className={errors.password ? 'error' : ''}
          placeholder="Kuchli parol"
        />
        {touched.password && errors.password && (
          <small className="error-message">{errors.password}</small>
        )}
        <small>Kamida 8 belgi, katta va kichik harf, raqam va maxsus belgi</small>
      </div>

      {/* Confirm Password */}
      <div className="form-group">
        <label htmlFor="confirmPassword">Parolni tasdiqlash *</label>
        <input
          type="password"
          id="confirmPassword"
          name="confirmPassword"
          value={formData.confirmPassword}
          onChange={handleChange}
          onBlur={handleBlur}
          className={errors.confirmPassword ? 'error' : ''}
          placeholder="Parolni qayta kiriting"
        />
        {touched.confirmPassword && errors.confirmPassword && (
          <small className="error-message">{errors.confirmPassword}</small>
        )}
      </div>

      {/* Age */}
      <div className="form-group">
        <label htmlFor="age">Yosh *</label>
        <input
          type="number"
          id="age"
          name="age"
          value={formData.age}
          onChange={handleChange}
          onBlur={handleBlur}
          className={errors.age ? 'error' : ''}
          min="18"
          max="120"
        />
        {touched.age && errors.age && (
          <small className="error-message">{errors.age}</small>
        )}
      </div>

      {/* Website */}
      <div className="form-group">
        <label htmlFor="website">Veb-sayt</label>
        <input
          type="url"
          id="website"
          name="website"
          value={formData.website}
          onChange={handleChange}
          onBlur={handleBlur}
          className={errors.website ? 'error' : ''}
          placeholder="https://mywebsite.com"
        />
        {touched.website && errors.website && (
          <small className="error-message">{errors.website}</small>
        )}
      </div>

      {/* Bio */}
      <div className="form-group">
        <label htmlFor="bio">Bio</label>
        <textarea
          id="bio"
          name="bio"
          value={formData.bio}
          onChange={handleChange}
          onBlur={handleBlur}
          rows="4"
          placeholder="O'zingiz haqingizda qisqacha..."
        />
        <small>{formData.bio.length}/500 belgi</small>
      </div>

      {/* Submit Button */}
      <button
        type="submit"
        disabled={!isFormValid || isSubmitting}
        className={isFormValid && !isSubmitting ? 'valid' : 'invalid'}
      >
        {isSubmitting ? 'Yuborilmoqda...' : 'Ro\'yxatdan o\'tish'}
      </button>

      {/* Form Status */}
      <div style={{ marginTop: '20px', padding: '10px', backgroundColor: '#f8f9fa' }}>
        <h4>Forma holati:</h4>
        <p>Valid: {isFormValid ? '✅ Ha' : '❌ Yo\'q'}</p>
        <p>Yuborilmoqda: {isSubmitting ? '🔄 Ha' : '⏸ Yo\'q'}</p>

        <details>
          <summary>Xatoliklar</summary>
          <pre>{JSON.stringify(errors, null, 2)}</pre>
        </details>
      </div>

      <style jsx>{`
        .form-group {
          margin-bottom: 20px;
        }

        .form-group label {
          display: block;
          margin-bottom: 5px;
```

```jsx
            <div style={{ display: 'flex', gap: '10px' }}>
              <input
                type="text"
                placeholder="Malaka nomi"
                value={skill.skill}
                onChange={(e) => updateSkill(index, 'skill', e.target.value)}
                style={{ flex: 1 }}
              />
              <select
                value={skill.experience}
                onChange={(e) => updateSkill(index, 'experience', e.target.value)}
                style={{ flex: 1 }}
              >
                <option value="">Tajriba darajasi</option>
                <option value="beginner">Boshlang'ich</option>
                <option value="intermediate">O'rta</option>
                <option value="advanced">Yuqori</option>
                <option value="expert">Mutaxassis</option>
              </select>
              <button
                type="button"
                onClick={() => removeSkill(index)}
                disabled={form.skills.length === 1}
                style={{ backgroundColor: '#dc3545', color: 'white', border: 'none', padding: '5px 10px' }}
              >
                O'chirish
              </button>
            </div>
          </div>
        ))}
        <button type="button" onClick={addSkill}>
          ➕ Yangi malaka qo'shish
        </button>
      </fieldset>

      {/* Projects */}
      <fieldset>
        <legend>Loyihalar</legend>
        {form.projects.map((project, index) => (
          <div key={index} style={{ border: '1px solid #ddd', padding: '15px', marginBottom: '10px' }}>
            <div>
              <input
                type="text"
                placeholder="Loyiha nomi"
                value={project.title}
                onChange={(e) => updateProject(index, 'title', e.target.value)}
                style={{ width: '100%', marginBottom: '10px' }}
              />
            </div>
            <div>
              <textarea
                placeholder="Loyiha tavsifi"
                value={project.description}
                onChange={(e) => updateProject(index, 'description', e.target.value)}
                rows="3"
                style={{ width: '100%', marginBottom: '10px' }}
              />
            </div>
            <div style={{ display: 'flex', gap: '10px' }}>
              <input
                type="text"
                placeholder="Ishlatilgan texnologiyalar (vergul bilan ajrating)"
                value={project.technologies}
                onChange={(e) => updateProject(index, 'technologies', e.target.value)}
                style={{ flex: 1 }}
              />
              <button
                type="button"
                onClick={() => removeProject(index)}
                disabled={form.projects.length === 1}
                style={{ backgroundColor: '#dc3545', color: 'white', border: 'none', padding: '5px 10px' }}
              >
                O'chirish
              </button>
            </div>
          </div>
        ))}
        <button type="button" onClick={addProject}>
          ➕ Yangi loyiha qo'shish
        </button>
      </fieldset>

      {/* Education */}
      <fieldset>
        <legend>Ta'lim</legend>
        {form.education.map((edu, index) => (
          <div key={index} style={{ border: '1px solid #ddd', padding: '10px', marginBottom: '10px' }}>
            <div style={{ display: 'flex', gap: '10px' }}>
              <input
                type="text"
                placeholder="Ilmiy daraja / Mutaxassislik"
                value={edu.degree}
                onChange={(e) => updateEducation(index, 'degree', e.target.value)}
                style={{ flex: 2 }}
              />
              <input
                type="text"
                placeholder="Ta'lim muassasasi"
                value={edu.institution}
                onChange={(e) => updateEducation(index, 'institution', e.target.value)}
                style={{ flex: 2 }}
              />
              <input
                type="number"
                placeholder="Yil"
                value={edu.year}
                onChange={(e) => updateEducation(index, 'year', e.target.value)}
                min="1950"
                max={new Date().getFullYear()}
                style={{ flex: 1 }}
              />
              <button
                type="button"
                onClick={() => removeEducation(index)}
                disabled={form.education.length === 1}
                style={{ backgroundColor: '#dc3545', color: 'white', border: 'none', padding: '5px 10px' }}
              >
                O'chirish
              </button>
            </div>
          </div>
        ))}
        <button type="button" onClick={addEducation}>
          ➕ Yangi ta'lim qo'shish
        </button>
      </fieldset>

      <button
        type="submit"
        style={{
          width: '100%',
          padding: '15px',
          backgroundColor: '#007bff',
          color: 'white',
          border: 'none',
          borderRadius: '4px',
          fontSize: '18px',
          cursor: 'pointer'
        }}
      >
        CV ni saqlash
      </button>

      {/* Preview */}
      <details style={{ marginTop: '20px' }}>
        <summary>CV Preview</summary>
        <div style={{ backgroundColor: '#f8f9fa', padding: '20px', borderRadius: '8px' }}>
          <h3>{form.name || 'Ism kiritilmagan'}</h3>

          <h4>Malakalar:</h4>
          <ul>
            {form.skills.map((skill, index) => (
              <li key={index}>
                {skill.skill} - {skill.experience}
              </li>
            ))}
          </ul>

          <h4>Loyihalar:</h4>
          {form.projects.map((project, index) => (
            <div key={index}>
              <h5>{project.title}</h5>
              <p>{project.description}</p>
              <p><strong>Texnologiyalar:</strong> {project.technologies}</p>
            </div>
          ))}

          <h4>Ta'lim:</h4>
          <ul>
            {form.education.map((edu, index) => (
              <li key={index}>
                {edu.degree} - {edu.institution} ({edu.year})
              </li>
            ))}
          </ul>
        </div>
      </details>
    </form>
  );
}
```

3. **Accessibility** - label, aria-* attributelar ishlatish
4. **Error handling** - aniq va foydali xatolik xabarlari
5. **UX** - loading states, disable vs busy states

### 🎯 Validation strategiyalari:
1. **Client-side** - tez feedback uchun
2. **Server-side** - xavfsizlik uchun
3. **Progressive enhancement** - HTML5 validation + JS
4. **Custom rules** - business logic uchun

### 📚 Keyingi mavzular:
1. **Lists va Keys** - optimized list rendering
2. **Context API** - form state ni global boshqarish
3. **Custom hooks** - form logic'ni qayta ishlatish
4. **Third-party libraries** - Formik, React Hook Form
