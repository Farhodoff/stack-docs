# Custom Hooks va Logic Qayta Ishlatish

Custom hooks - React'da logic'ni qayta ishlatishning eng kuchli usuli. Ular komponentlar o'rtasida stateful logic'ni ulashish imkonini beradi.

## 1. Custom Hook Nima?

Custom hook - odatiy JavaScript funksiya bo'lib:
- **"use"** bilan boshlanadi (masalan: `useCounter`, `useApi`)
- **Boshqa hook'larni chaqirishi mumkin** (useState, useEffect, va h.k.)
- **Stateful logic'ni qayta ishlatish** imkonini beradi
- **Komponent emas** - JSX qaytarmaydi

### Oddiy misol:

```jsx
import { useState } from 'react';

// Custom hook yaratish
function useCounter(initialValue = 0) {
  const [count, setCount] = useState(initialValue);

  const increment = () => setCount(prev => prev + 1);
  const decrement = () => setCount(prev => prev - 1);
  const reset = () => setCount(initialValue);

  return {
    count,
    increment,
    decrement,
    reset
  };
}

// Custom hook'dan foydalanish
function CounterComponent() {
  const { count, increment, decrement, reset } = useCounter(10);

  return (
    <div>
      <h2>Hisoblagich: {count}</h2>
      <button onClick={increment}>+</button>
      <button onClick={decrement}>-</button>
      <button onClick={reset}>Reset</button>
    </div>
  );
}

function AnotherCounterComponent() {
```

```jsx
import { useState, useEffect } from 'react';

function useLocalStorage(key, initialValue) {
  // State'ni localStorage'dan olish yoki initial value qo'yish
  const [storedValue, setStoredValue] = useState(() => {
    try {
      const item = window.localStorage.getItem(key);
      return item ? JSON.parse(item) : initialValue;
    } catch (error) {
      console.error(`localStorage'dan ${key} olib bo'lmadi:`, error);
      return initialValue;
    }
  });

  // Value'ni o'zgartirganda localStorage'ga saqlash
  const setValue = (value) => {
    try {
      // Value function yoki oddiy qiymat bo'lishi mumkin
      const valueToStore = value instanceof Function ? value(storedValue) : value;
      setStoredValue(valueToStore);

      // localStorage'ga saqlash
      if (typeof window !== 'undefined') {
        window.localStorage.setItem(key, JSON.stringify(valueToStore));
      }
    } catch (error) {
      console.error(`localStorage'ga ${key} saqlab bo'lmadi:`, error);
    }
  };

  // localStorage o'zgarishlarini kuzatish (boshqa tab'lardan)
  useEffect(() => {
    const handleStorageChange = (e) => {
      if (e.key === key && e.newValue !== null) {
        try {
          setStoredValue(JSON.parse(e.newValue));
        } catch (error) {
          console.error('Storage change parse error:', error);
        }
      }
    };

    window.addEventListener('storage', handleStorageChange);

    return () => {
      window.removeEventListener('storage', handleStorageChange);
    };
  }, [key]);

  return [storedValue, setValue];
}
```

```jsx
import { useState, useEffect } from 'react';

function useFetch(url, options = {}) {
  const [data, setData] = useState(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    let abortController = new AbortController();

    const fetchData = async () => {
      try {
        setLoading(true);
        setError(null);

        const response = await fetch(url, {
          ...options,
          signal: abortController.signal
        });

        if (!response.ok) {
          throw new Error(`HTTP error! status: ${response.status}`);
        }

        const result = await response.json();

        if (!abortController.signal.aborted) {
          setData(result);
        }
      } catch (error) {
        if (error.name !== 'AbortError' && !abortController.signal.aborted) {
          setError(error.message);
        }
      } finally {
        if (!abortController.signal.aborted) {
          setLoading(false);
        }
      }
    };

    fetchData();

    return () => {
      abortController.abort();
    };
  }, [url, JSON.stringify(options)]);

  const refetch = () => {
    setLoading(true);
    setError(null);
    // URL'ni o'zgartirib, useEffect'ni qayta ishga tushirish
    // Real implementatsiyada requestId'ni increment qilish mumkin
  };

  return { data, loading, error, refetch };
}

// Advanced API hook with caching
function useApi(url, options = {}) {
  const [data, setData] = useState(null);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState(null);
  const [cache] = useState(new Map());

  const fetchData = async (fetchUrl = url, fetchOptions = options) => {
    // Cache kaliti yaratish
    const cacheKey = `${fetchUrl}:${JSON.stringify(fetchOptions)}`;

    // Cache'da mavjudligini tekshirish
    if (cache.has(cacheKey)) {
      const cachedData = cache.get(cacheKey);
      const isExpired = Date.now() - cachedData.timestamp > 300000; // 5 minut

      if (!isExpired) {
        setData(cachedData.data);
        setLoading(false);
        return cachedData.data;
      }
    }

    setLoading(true);
    setError(null);

    try {
      const response = await fetch(fetchUrl, fetchOptions);

      if (!response.ok) {
        throw new Error(`HTTP error! status: ${response.status}`);
      }

      const result = await response.json();

      // Cache'ga saqlash
      cache.set(cacheKey, {
        data: result,
        timestamp: Date.now()
      });

      setData(result);
      return result;
    } catch (err) {
      setError(err.message);
      throw err;
    } finally {
      setLoading(false);
    }
  };

  // Post, Put, Delete methodlari
  const post = (postUrl, body) => {
    return fetchData(postUrl, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        ...options.headers
      },
      body: JSON.stringify(body)
    });
  };

  const put = (putUrl, body) => {
    return fetchData(putUrl, {
      method: 'PUT',
      headers: {
        'Content-Type': 'application/json',
        ...options.headers
      },
      body: JSON.stringify(body)
    });
  };

  const del = (deleteUrl) => {
    return fetchData(deleteUrl, {
      method: 'DELETE',
      ...options
    });
  };

  useEffect(() => {
    if (url) {
      fetchData();
    }
  }, [url]);

  return {
    data,
    loading,
    error,
    refetch: () => fetchData(),
    post,
    put,
    delete: del
  };
}
```

```jsx
import { useState } from 'react';

function useForm(initialValues, validationRules = {}) {
  const [values, setValues] = useState(initialValues);
  const [errors, setErrors] = useState({});
  const [touched, setTouched] = useState({});

  // Field value'ni o'zgartirish
  const setValue = (name, value) => {
    setValues(prev => ({
      ...prev,
      [name]: value
    }));

    // Real-time validation
    if (touched[name]) {
      validateField(name, value);
    }
  };

  // Input change handler
  const handleChange = (event) => {
    const { name, value, type, checked } = event.target;
    const fieldValue = type === 'checkbox' ? checked : value;
    setValue(name, fieldValue);
  };

  // Field blur handler
  const handleBlur = (event) => {
    const { name } = event.target;
    setTouched(prev => ({ ...prev, [name]: true }));
    validateField(name, values[name]);
  };

  // Bitta field'ni validation qilish
  const validateField = (name, value) => {
    const rule = validationRules[name];
    let error = '';

    if (rule) {
      if (rule.required && (!value || value.toString().trim() === '')) {
        error = rule.requiredMessage || `${name} majburiy`;
      } else if (value && rule.minLength && value.toString().length < rule.minLength) {
        error = rule.minLengthMessage || `${name} kamida ${rule.minLength} belgi bo'lishi kerak`;
      } else if (value && rule.pattern && !rule.pattern.test(value)) {
        error = rule.patternMessage || `${name} formati noto'g'ri`;
      } else if (rule.custom && typeof rule.custom === 'function') {
        error = rule.custom(value, values) || '';
      }
    }

    setErrors(prev => ({
      ...prev,
      [name]: error
    }));

    return error === '';
  };

  // Barcha field'larni validation qilish
  const validateAll = () => {
    const newErrors = {};
    let isValid = true;

    Object.keys(validationRules).forEach(fieldName => {
      const fieldValue = values[fieldName];
      const isFieldValid = validateField(fieldName, fieldValue);

      if (!isFieldValid) {
        isValid = false;
      }
    });

    // Barcha field'larni touched deb belgilash
    setTouched(
      Object.keys(validationRules).reduce((acc, field) => ({
        ...acc,
        [field]: true
      }), {})
    );

    return isValid;
  };

  // Form'ni reset qilish
  const reset = () => {
    setValues(initialValues);
    setErrors({});
    setTouched({});
  };

  // Form submit handler
  const handleSubmit = (submitCallback) => (event) => {
    event.preventDefault();

    if (validateAll()) {
      submitCallback(values);
    }
  };

  // Form holati
  const isValid = Object.keys(errors).every(key => !errors[key]);
  const isDirty = Object.keys(values).some(key => values[key] !== initialValues[key]);

  return {
    values,
    errors,
    touched,
    setValue,
    handleChange,
    handleBlur,
    handleSubmit,
    validateField,
    validateAll,
    reset,
    isValid,
    isDirty
  };
}
```

```jsx
import { useState, useEffect, useRef } from 'react';

// Interval hook
function useInterval(callback, delay) {
  const savedCallback = useRef();

  useEffect(() => {
    savedCallback.current = callback;
  }, [callback]);

  useEffect(() => {
    if (delay !== null) {
      const id = setInterval(() => savedCallback.current(), delay);
      return () => clearInterval(id);
    }
  }, [delay]);
}

// Countdown hook
function useCountdown(targetDate) {
  const [timeLeft, setTimeLeft] = useState(calculateTimeLeft(targetDate));

  useInterval(() => {
    setTimeLeft(calculateTimeLeft(targetDate));
  }, 1000);

  function calculateTimeLeft(target) {
    const difference = +new Date(target) - +new Date();
    let timeLeft = {};

    if (difference > 0) {
      timeLeft = {
        days: Math.floor(difference / (1000 * 60 * 60 * 24)),
        hours: Math.floor((difference / (1000 * 60 * 60)) % 24),
        minutes: Math.floor((difference / 1000 / 60) % 60),
        seconds: Math.floor((difference / 1000) % 60)
      };
    }

    return timeLeft;
  }

  const isFinished = Object.keys(timeLeft).length === 0;

  return { timeLeft, isFinished };
}

// Stopwatch hook
function useStopwatch() {
  const [time, setTime] = useState(0);
  const [isRunning, setIsRunning] = useState(false);

  useInterval(
    () => {
      setTime(time => time + 10);
    },
    isRunning ? 10 : null
  );

  const start = () => setIsRunning(true);
  const stop = () => setIsRunning(false);
  const reset = () => {
    setTime(0);
    setIsRunning(false);
  };

  const formatTime = (totalMs) => {
    const minutes = Math.floor(totalMs / 60000);
    const seconds = Math.floor((totalMs % 60000) / 1000);
    const milliseconds = Math.floor((totalMs % 1000) / 10);

    return `${minutes.toString().padStart(2, '0')}:${seconds.toString().padStart(2, '0')}.${milliseconds.toString().padStart(2, '0')}`;
  };

  return {
    time: formatTime(time),
    isRunning,
    start,
    stop,
    reset
  };
}
```

```jsx
// Online/Offline detector
function useOnlineStatus() {
  const [isOnline, setIsOnline] = useState(
    typeof navigator !== 'undefined' ? navigator.onLine : true
  );

  useEffect(() => {
    const handleOnline = () => setIsOnline(true);
    const handleOffline = () => setIsOnline(false);

    window.addEventListener('online', handleOnline);
    window.addEventListener('offline', handleOffline);

    return () => {
      window.removeEventListener('online', handleOnline);
      window.removeEventListener('offline', handleOffline);
    };
  }, []);

  return isOnline;
}

// Window size tracker
function useWindowSize() {
  const [windowSize, setWindowSize] = useState({
    width: typeof window !== 'undefined' ? window.innerWidth : 0,
    height: typeof window !== 'undefined' ? window.innerHeight : 0,
  });

  useEffect(() => {
    function handleResize() {
      setWindowSize({
        width: window.innerWidth,
        height: window.innerHeight,
      });
    }

    window.addEventListener('resize', handleResize);
    return () => window.removeEventListener('resize', handleResize);
  }, []);

  return windowSize;
}

// Mouse position tracker
function useMousePosition() {
  const [mousePosition, setMousePosition] = useState({ x: 0, y: 0 });

  useEffect(() => {
    const updateMousePosition = (e) => {
      setMousePosition({ x: e.clientX, y: e.clientY });
    };

    window.addEventListener('mousemove', updateMousePosition);
    return () => window.removeEventListener('mousemove', updateMousePosition);
  }, []);

  return mousePosition;
}

// Previous value tracker
function usePrevious(value) {
  const ref = useRef();

  useEffect(() => {
    ref.current = value;
  });

  return ref.current;
}

// Debounced value
function useDebounce(value, delay) {
  const [debouncedValue, setDebouncedValue] = useState(value);

  useEffect(() => {
    const handler = setTimeout(() => {
      setDebouncedValue(value);
    }, delay);

    return () => {
      clearTimeout(handler);
    };
  }, [value, delay]);

  return debouncedValue;
}
```

## Xulosa va Best Practices

### ✅ Custom hooks best practices:
1. **"use" prefixi** - har doim "use" bilan boshlash
2. **Single responsibility** - bitta vazifalı hook'lar yaratish
3. **Parameters** - hook'larni parametrize qilish
4. **Return object** - named return qilish (destructuring uchun)
5. **Error handling** - xatoliklarni to'g'ri handle qilish

### 🎯 Custom hook'lar uchun ideal case'lar:
1. **API calls** - data fetching patterns
2. **Form management** - validation va state
3. **Event listeners** - DOM event'larni boshqarish
4. **Storage** - localStorage, sessionStorage
5. **Timers** - interval, timeout, countdown

### 📚 Keyingi mavzular:
1. **Testing custom hooks** - hook'larni test qilish
2. **Performance optimization** - useMemo, useCallback
3. **React Router** - navigation hooks
4. **State management** - Redux, Zustand hooks

Custom hooks - React'da code reusability va maintainability'ni ta'minlashning eng samarali usuli!
