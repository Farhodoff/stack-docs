# React Router va SPA Navigation

React Router - React ilovalarida navigation va routing'ni boshqarish uchun eng mashhur library. U Single Page Application (SPA) yaratishda markaziy rol o'ynaydi.

## 1. SPA va Routing Asoslari

### Multi-Page vs Single Page Application:

**Traditional Multi-Page App:**
- Har bir sahifa uchun alohida HTML fayl
- Sahifa o'zgarishida server'ga so'rov
- To'liq sahifa qayta yuklanish

**Single Page App (SPA):**
- Bitta HTML fayl
- JavaScript orqali kontent o'zgartirish
- Faqat kerakli ma'lumotlar yuklanish

### React Router o'rnatish:

```bash
npm install react-router-dom
```

### Asosiy setup:


## 3. URL Parameters va Dynamic Routes

```jsx
import { useParams, useSearchParams, Link } from 'react-router-dom';

// Products list
function ProductsList() {
  const products = [
    { id: 1, name: 'iPhone 15', category: 'phone', price: 999 },
    { id: 2, name: 'MacBook Pro', category: 'laptop', price: 1999 },
    { id: 3, name: 'AirPods Pro', category: 'audio', price: 249 },
    { id: 4, name: 'iPad Air', category: 'tablet', price: 599 }
  ];

  const [searchParams, setSearchParams] = useSearchParams();
  const category = searchParams.get('category') || 'all';
  const search = searchParams.get('search') || '';

  // Filter products
  const filteredProducts = products.filter(product => {
    const matchesCategory = category === 'all' || product.category === category;
    const matchesSearch = product.name.toLowerCase().includes(search.toLowerCase());
    return matchesCategory && matchesSearch;
  });

  const updateFilter = (key, value) => {
    const newParams = new URLSearchParams(searchParams);
    if (value) {
      newParams.set(key, value);
    } else {
      newParams.delete(key);
    }
    setSearchParams(newParams);
  };

  return (
    <div style={{ padding: '20px' }}>
      <h1>📦 Mahsulotlar</h1>

      {/* Filters */}
      <div style={{
        display: 'flex',
        gap: '15px',
        marginBottom: '20px',
        padding: '15px',
        backgroundColor: '#f8f9fa',
        borderRadius: '8px'
      }}>
        <div>
          <label>Qidiruv: </label>
          <input
            type="text"
            value={search}
            onChange={(e) => updateFilter('search', e.target.value)}
            placeholder="Mahsulot nomi..."
          />
        </div>

        <div>
          <label>Kategoriya: </label>
          <select
            value={category}
            onChange={(e) => updateFilter('category', e.target.value)}
          >
            <option value="all">Barchasi</option>
            <option value="phone">Telefon</option>
            <option value="laptop">Noutbuk</option>
            <option value="audio">Audio</option>
            <option value="tablet">Planshet</option>
          </select>
        </div>

        <button
          onClick={() => setSearchParams({})}
          style={{
            backgroundColor: '#6c757d',
            color: 'white',
            border: 'none',
            padding: '8px 12px',
            borderRadius: '4px',
            cursor: 'pointer'

              </Link>
            </h3>
            <p>Kategoriya: {product.category}</p>
            <p style={{ fontSize: '20px', color: '#28a745', fontWeight: 'bold' }}>
```

```jsx
  const logout = () => {
    localStorage.removeItem('isLoggedIn');
    localStorage.removeItem('userRole');
  };

  return {
    isLoggedIn,
    userRole,
    login,
    logout
  };
}

// Protected Route component
function ProtectedRoute({ children, requiredRole = null }) {
  const { isLoggedIn, userRole } = useAuth();
  const location = useLocation();

  // Login qilinmaganida login sahifasiga yo'naltirish
  if (!isLoggedIn) {
    return (
      <Navigate
        to="/login"
        state={{ from: location }}
        replace
      />
    );
  }

  // Role tekshirish
  if (requiredRole && userRole !== requiredRole) {
    return (
      <div style={{ padding: '20px', textAlign: 'center' }}>
        <h1>🚫 Ruxsat yo'q</h1>
        <p>Bu sahifaga kirish uchun "{requiredRole}" roli kerak.</p>
        <p>Sizning rolingiz: "{userRole}"</p>
        <Link to="/">Bosh sahifaga qaytish</Link>
      </div>
    );
  }

  return children;
}

// Public Only Route (login qilinganida boshqa sahifaga yo'naltirish)
function PublicOnlyRoute({ children }) {
  const { isLoggedIn } = useAuth();

  if (isLoggedIn) {
    return <Navigate to="/dashboard" replace />;
  }

  return children;
}

// Main App with protected routes
function App() {
  return (
    <BrowserRouter>
      <nav style={{ padding: '20px', backgroundColor: '#f8f9fa' }}>
        <AuthNavigation />
      </nav>

      <Routes>
        {/* Public routes */}
        <Route path="/" element={<HomePage />} />
        <Route path="/about" element={<AboutPage />} />

        {/* Public only (login sahifa) */}
        <Route
          path="/login"
          element={
            <PublicOnlyRoute>
              <LoginPage />
            </PublicOnlyRoute>
          }
        />

        {/* Protected routes */}
        <Route
          path="/dashboard/*"
          element={
            <ProtectedRoute>
              <DashboardRoutes />
            </ProtectedRoute>
          }
        />

        <Route
          path="/profile"
          element={
            <ProtectedRoute>
              <ProfilePage />
            </ProtectedRoute>
          }
        />

        {/* Admin only routes */}
        <Route
          path="/admin/*"
          element={
            <ProtectedRoute requiredRole="admin">
              <AdminRoutes />
            </ProtectedRoute>
          }
        />

        {/* 404 */}
        <Route path="*" element={<NotFoundPage />} />
      </Routes>
    </BrowserRouter>
  );
}

function AuthNavigation() {
```

```jsx
// Error Boundary Component
function ErrorBoundary() {
  const error = useRouteError();

  if (isRouteErrorResponse(error)) {
    return (
      <div style={{ padding: '40px', textAlign: 'center' }}>
        <h1>{error.status}</h1>
        <h2>{error.statusText}</h2>
        {error.data?.message && <p>{error.data.message}</p>}
      </div>
    );
  }

  return (
    <div style={{ padding: '40px', textAlign: 'center' }}>
      <h1>⚠️ Kutilmagan xatolik</h1>
      <p>{error.message}</p>
      <button onClick={() => window.location.reload()}>
        Sahifani yangilash
      </button>
    </div>
  );
```
