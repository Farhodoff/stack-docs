### Server'ga Swagger qo'shish:

```javascript
// app.js
const express = require('express');
const { swaggerUi, specs } = require('./swagger');

const app = express();

// Swagger UI
app.use('/api-docs', swaggerUi.serve, swaggerUi.setup(specs, {
  customCss: '.swagger-ui .topbar { display: none }',
  customSiteTitle: "Blog Platform API Documentation"
}));

// Routes
app.use('/api/auth', require('./routes/auth'));
app.use('/api/blogs', require('./routes/blogs'));

console.log('API Documentation: http://localhost:3000/api-docs');
```

### Test utilities setup:

```javascript
// tests/setup.js
const { app } = require('../app');
const request = require('supertest');

// Test database setup
const testUsers = [];
const testBlogs = [];

// Helper functions
const createTestUser = (userData = {}) => {
  const defaultUser = {
    id: String(testUsers.length + 1),
    name: 'Test User',
    email: 'test@example.com',
    role: 'user',
    createdAt: new Date().toISOString()
  };

  const newUser = { ...defaultUser, ...userData };
  testUsers.push(newUser);
  return newUser;
};

const createTestBlog = (blogData = {}, authorId = '1') => {
  const defaultBlog = {
    id: String(testBlogs.length + 1),
    title: 'Test Blog',
    content: 'Test content that is long enough for validation requirements',
    excerpt: 'Test excerpt',
    authorId,
    status: 'published',
    category: 'technology',
    tags: ['test'],
    createdAt: new Date().toISOString()
  };

  const newBlog = { ...defaultBlog, ...blogData };
  testBlogs.push(newBlog);
  return newBlog;
};

const generateToken = (userId) => {
  const jwt = require('jsonwebtoken');
  return jwt.sign({ userId }, process.env.JWT_SECRET || 'test-secret', {
    expiresIn: '1h'
  });
};

// Reset database before each test
beforeEach(() => {
  testUsers.length = 0;
  testBlogs.length = 0;
});

module.exports = {
  app,
  request,
  createTestUser,
  createTestBlog,
  generateToken
};
```

### Blog API tests:

```javascript
// tests/blogs.test.js
const { request, app, createTestUser, createTestBlog, generateToken } = require('./setup');

describe('Blog API', () => {
  describe('GET /api/blogs', () => {
    it('should return empty array when no blogs exist', async () => {
      const response = await request(app)
        .get('/api/blogs')
        .expect(200);

      expect(response.body).toEqual({
        success: true,
        data: [],
        pagination: {
          page: 1,
          limit: 10,
          total: 0,
          totalPages: 0,
          hasNext: false,
          hasPrev: false
        }
      });
    });

    it('should return paginated blogs', async () => {
      // Setup test data
      const testUser = createTestUser();
      createTestBlog({ title: 'First Blog' }, testUser.id);
      createTestBlog({ title: 'Second Blog' }, testUser.id);

      const response = await request(app)
        .get('/api/blogs?limit=1&page=1')
        .expect(200);

      expect(response.body.success).toBe(true);
      expect(response.body.data).toHaveLength(1);
      expect(response.body.pagination.total).toBe(2);
      expect(response.body.pagination.hasNext).toBe(true);
    });

    it('should filter blogs by category', async () => {
      const testUser = createTestUser();
      createTestBlog({ category: 'technology' }, testUser.id);
      createTestBlog({ category: 'lifestyle' }, testUser.id);

      const response = await request(app)
        .get('/api/blogs?category=technology')
        .expect(200);

      expect(response.body.data).toHaveLength(1);
      expect(response.body.data[0].category).toBe('technology');
    });

    it('should search blogs by title and content', async () => {
      const testUser = createTestUser();
      createTestBlog({
        title: 'GraphQL API Tutorial',
        content: 'Learn GraphQL with this comprehensive guide'
      }, testUser.id);
      createTestBlog({
        title: 'REST API Guide',
        content: 'Traditional REST API development'
      }, testUser.id);

      const response = await request(app)
        .get('/api/blogs?search=GraphQL')
        .expect(200);

      expect(response.body.data).toHaveLength(1);
      expect(response.body.data[0].title).toContain('GraphQL');
    });

    it('should validate query parameters', async () => {
      const response = await request(app)
        .get('/api/blogs?page=invalid')
        .expect(400);

      expect(response.body.success).toBe(false);
      expect(response.body.error.code).toBe('QUERY_VALIDATION_ERROR');
    });
  });

  describe('POST /api/blogs', () => {
    it('should create blog with valid data and auth', async () => {
      const testUser = createTestUser();
      const token = generateToken(testUser.id);

      const blogData = {
        title: 'New Blog Post',
        content: 'This is a comprehensive blog post content that meets the minimum length requirement',
        category: 'technology',
        tags: ['nodejs', 'api']
      };

      const response = await request(app)
        .post('/api/blogs')
        .set('Authorization', `Bearer ${token}`)
        .send(blogData)
        .expect(201);

      expect(response.body.success).toBe(true);
      expect(response.body.data).toMatchObject({
        title: blogData.title,
        content: blogData.content,
        category: blogData.category,
        tags: blogData.tags
      });
      expect(response.body.data.id).toBeDefined();
      expect(response.body.data.createdAt).toBeDefined();
    });

    it('should reject blog creation without authentication', async () => {
      const blogData = {
        title: 'Unauthorized Blog',
        content: 'This should fail because no auth token provided'
      };

      const response = await request(app)
        .post('/api/blogs')
        .send(blogData)
        .expect(401);

      expect(response.body.success).toBe(false);
      expect(response.body.error.code).toBe('TOKEN_MISSING');
    });

    it('should validate required fields', async () => {
      const testUser = createTestUser();
      const token = generateToken(testUser.id);

      // Missing title
      const response = await request(app)
        .post('/api/blogs')
        .set('Authorization', `Bearer ${token}`)
        .send({
          content: 'Content without title'
        })
        .expect(400);

      expect(response.body.success).toBe(false);
      expect(response.body.error.code).toBe('VALIDATION_ERROR');
      expect(response.body.error.details[0].field).toBe('title');
    });

    it('should validate field lengths', async () => {
      const testUser = createTestUser();
      const token = generateToken(testUser.id);

      // Title too short
      const response = await request(app)
        .post('/api/blogs')
        .set('Authorization', `Bearer ${token}`)
        .send({
          title: 'AB', // Less than 3 characters
          content: 'Content that meets minimum length requirements for blog posts'
        })
        .expect(400);

      expect(response.body.error.details[0].message).toContain('kamida 3');
    });

    it('should sanitize HTML content', async () => {
      const testUser = createTestUser();
      const token = generateToken(testUser.id);

      const blogData = {
        title: 'XSS Test Blog',
        content: 'Normal content <script>alert("XSS")</script> with malicious code that should be sanitized'
      };

      const response = await request(app)
        .post('/api/blogs')
        .set('Authorization', `Bearer ${token}`)
        .send(blogData)
        .expect(201);

      expect(response.body.data.content).not.toContain('<script>');
      expect(response.body.data.content).toContain('Normal content');
    });
  });

  describe('PUT /api/blogs/:id', () => {
    it('should update own blog', async () => {
      const testUser = createTestUser();
      const testBlog = createTestBlog({}, testUser.id);
      const token = generateToken(testUser.id);

      const updates = {
        title: 'Updated Blog Title',
        content: 'Updated content that meets all validation requirements'
      };

      const response = await request(app)
        .put(`/api/blogs/${testBlog.id}`)
        .set('Authorization', `Bearer ${token}`)
        .send(updates)
        .expect(200);

      expect(response.body.data.title).toBe(updates.title);
      expect(response.body.data.content).toBe(updates.content);
      expect(response.body.data.updatedAt).toBeDefined();
    });

    it('should not update others blog', async () => {
      const author = createTestUser({ id: '1' });
      const otherUser = createTestUser({ id: '2', email: 'other@example.com' });
      const testBlog = createTestBlog({}, author.id);
      const token = generateToken(otherUser.id);

      const response = await request(app)
        .put(`/api/blogs/${testBlog.id}`)
        .set('Authorization', `Bearer ${token}`)
        .send({ title: 'Hacked Title' })
        .expect(403);

      expect(response.body.error.code).toBe('INSUFFICIENT_PERMISSIONS');
    });
  });

  describe('DELETE /api/blogs/:id', () => {
    it('should delete own blog', async () => {
      const testUser = createTestUser();
      const testBlog = createTestBlog({}, testUser.id);
      const token = generateToken(testUser.id);

      const response = await request(app)
        .delete(`/api/blogs/${testBlog.id}`)
        .set('Authorization', `Bearer ${token}`)
        .expect(200);

      expect(response.body.success).toBe(true);

      // Verify blog is deleted
      await request(app)
        .get(`/api/blogs/${testBlog.id}`)
        .expect(404);
    });

    it('should allow admin to delete any blog', async () => {
      const author = createTestUser({ id: '1' });
      const admin = createTestUser({ id: '2', role: 'admin', email: 'admin@example.com' });
      const testBlog = createTestBlog({}, author.id);
      const adminToken = generateToken(admin.id);

      const response = await request(app)
        .delete(`/api/blogs/${testBlog.id}`)
        .set('Authorization', `Bearer ${adminToken}`)
        .expect(200);

      expect(response.body.success).toBe(true);
```
