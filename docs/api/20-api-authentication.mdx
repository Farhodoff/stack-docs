## Authentication vs Authorization

### Tushunchalar farqi:

#### 🔑 **Authentication** (Autentifikatsiya):
> "Siz kim ekansiz?" - foydalanuvchi identifikatsiyasi

#### 🛡️ **Authorization** (Avtorizatsiya):
> "Siz nimani qila olasiz?" - ruxsatlarni tekshirish

### Real hayot misoli:

```
🏢 Ofis binosi:

1️⃣ Authentication: ID karta bilan kirish
   "Bu Alibek Tursunov ekanligini tasdiqlayman"

2️⃣ Authorization: Qaysi xonalarga kirishga ruxsat
   "Alibek developer, shuning uchun IT xonasiga kirishi mumkin,
    lekin CEO xonasiga kirolmaydi"
```

### Web dasturlashda:

```javascript
// Authentication - Login qilish
POST /api/auth/login
{
  "email": "user@example.com",
  "password": "password123"
}
// Response: JWT token yoki session

// Authorization - Himoyalangan resurs
GET /api/admin/users
Headers: Authorization: Bearer <jwt_token>
// Faqat admin foydalanuvchilarga ruxsat
```

## JWT (JSON Web Tokens)

### JWT nima?

**JWT** - ma'lumotlarni xavfsiz uzatish uchun kompakt, o'zi-o'zini tekshiruvchi token formatidi.

### JWT struktura:

```
eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJzdWIiOiIxMjM0NTY3ODkwIiwibmFtZSI6IkpvaG4gRG9lIiwiaWF0IjoxNTE2MjM5MDIyfQ.SflKxwRJSMeKKF2QT4fwpMeJf36POk6yJV_adQssw5c

Header.Payload.Signature
```

#### 1. **Header:** (Sarlavha)
```json
{
  "alg": "HS256",  // Hashing algoritmi
  "typ": "JWT"     // Token turi
}
```

#### 2. **Payload:** (Ma'lumotlar)
```json
{
  "sub": "1234567890",    // Subject (user ID)
  "name": "Alibek Tursunov",
  "email": "alibek@example.com",
  "role": "admin",
  "iat": 1516239022,      // Issued at (yaratilgan vaqt)
  "exp": 1516325422       // Expires (amal qilish muddati)
}
```

#### 3. **Signature:** (Imzo)
```javascript
HMACSHA256(
  base64UrlEncode(header) + "." +
  base64UrlEncode(payload),
  secret_key
)
```

### JWT'ning afzalliklari:

- ✅ **Stateless** - server xotirasida saqlanmaydi
- ✅ **Scalable** - microservices uchun ideal
- ✅ **Cross-platform** - har qanday tildan ishlatsa bo'ladi
- ✅ **Self-contained** - token ichida barcha ma'lumot
- ✅ **Fast** - database query'siz tekshirish

### JWT bilan authentication:

```javascript
// npm install jsonwebtoken bcryptjs
const jwt = require('jsonwebtoken');
const bcrypt = require('bcryptjs');

// Fake user database
const users = [
  {
    id: 1,
    email: "alibek@example.com",
    password: "$2a$10$...", // hashed password
    name: "Alibek Tursunov",
    role: "admin"
  }
];

const JWT_SECRET = process.env.JWT_SECRET || "your-secret-key";

// Login endpoint
app.post('/api/auth/login', async (req, res) => {
  try {
    const { email, password } = req.body;

    // Input validation
    if (!email || !password) {
      return res.status(400).json({
        success: false,
        error: {
          code: 'MISSING_CREDENTIALS',
          message: 'Email va password majburiy'
        }
      });
    }

    // Find user
    const user = users.find(u => u.email === email);
    if (!user) {
      return res.status(401).json({
        success: false,
        error: {
          code: 'INVALID_CREDENTIALS',
          message: 'Email yoki password noto\'g\'ri'
        }
      });
    }

    // Check password
    const isPasswordValid = await bcrypt.compare(password, user.password);
    if (!isPasswordValid) {
      return res.status(401).json({
        success: false,
        error: {
          code: 'INVALID_CREDENTIALS',
          message: 'Email yoki password noto\'g\'ri'
        }
      });
    }

    // Generate JWT token
    const payload = {
      id: user.id,
      email: user.email,
      name: user.name,
      role: user.role
    };

    const token = jwt.sign(payload, JWT_SECRET, {
      expiresIn: '24h'  // 24 soatlik token
    });

    // Send response
    res.status(200).json({
      success: true,
      data: {
        user: {
          id: user.id,
          email: user.email,
          name: user.name,
          role: user.role
        },
        token: token
      },
      message: 'Muvaffaqiyatli login qildingiz'
    });

  } catch (error) {
    console.error('Login error:', error);
    res.status(500).json({
      success: false,
      error: {
        code: 'LOGIN_FAILED',
        message: 'Login qilishda xatolik yuz berdi'
      }
    });
  }
});

// Register endpoint
app.post('/api/auth/register', async (req, res) => {
  try {
    const { name, email, password } = req.body;

    // Validation
    if (!name || !email || !password) {
      return res.status(400).json({
        success: false,
        error: {
          code: 'MISSING_FIELDS',
          message: 'Barcha maydonlar majburiy'
        }
      });
    }

    if (password.length < 6) {
      return res.status(400).json({
        success: false,
        error: {
          code: 'WEAK_PASSWORD',
          message: 'Password kamida 6 ta belgidan iborat bo\'lishi kerak'
        }
      });
    }

    // Check if user exists
    const existingUser = users.find(u => u.email === email);
    if (existingUser) {
      return res.status(409).json({
        success: false,
        error: {
          code: 'USER_EXISTS',
          message: 'Bu email bilan foydalanuvchi allaqachon mavjud'
        }
      });
    }

    // Hash password
    const saltRounds = 10;
    const hashedPassword = await bcrypt.hash(password, saltRounds);

    // Create new user
    const newUser = {
      id: users.length + 1,
      name: name,
      email: email,
      password: hashedPassword,
      role: 'user',
      createdAt: new Date().toISOString()
    };

    users.push(newUser);

    // Generate token
    const payload = {
      id: newUser.id,
      email: newUser.email,
      name: newUser.name,
      role: newUser.role
    };

    const token = jwt.sign(payload, JWT_SECRET, { expiresIn: '24h' });

    // Response
    res.status(201).json({
      success: true,
      data: {
        user: {
          id: newUser.id,
          email: newUser.email,
          name: newUser.name,
          role: newUser.role
        },
        token: token
      },
      message: 'Muvaffaqiyatli ro\'yxatdan o\'tdingiz'
    });

  } catch (error) {
    console.error('Registration error:', error);
    res.status(500).json({
      success: false,
      error: {
        code: 'REGISTRATION_FAILED',
        message: 'Ro\'yxatdan o\'tishda xatolik yuz berdi'
      }
    });
  }
});
```

## JWT Middleware - Token tekshirish

### Authentication middleware:

```javascript
// middleware/auth.js
const jwt = require('jsonwebtoken');

const JWT_SECRET = process.env.JWT_SECRET || "your-secret-key";

// JWT token tekshirish middleware
const authenticateToken = (req, res, next) => {
  // Authorization header'dan token olish
  const authHeader = req.headers['authorization'];
  const token = authHeader && authHeader.split(' ')[1]; // Bearer TOKEN

  if (!token) {
    return res.status(401).json({
      success: false,
      error: {
        code: 'TOKEN_MISSING',
        message: 'Access token talab qilinadi'
      }
    });
  }

  // Token tekshirish
  jwt.verify(token, JWT_SECRET, (err, user) => {
    if (err) {
      // Token yaroqsiz yoki muddati o'tgan
      let error = {};
      if (err.name === 'TokenExpiredError') {
        error = {
          code: 'TOKEN_EXPIRED',
          message: 'Token muddati o\'tgan, qayta login qiling'
        };
      } else {
        error = {
          code: 'TOKEN_INVALID',
          message: 'Yaroqsiz token'
        };
      }

      return res.status(403).json({
        success: false,
        error: error
      });
    }

    // Token to'g'ri - user ma'lumotlarini request'ga qo'shish
    req.user = user;
    next();
  });
};

// Role tekshirish middleware
const requireRole = (requiredRole) => {
  return (req, res, next) => {
    if (!req.user) {
      return res.status(401).json({
        success: false,
        error: {
          code: 'UNAUTHORIZED',
          message: 'Authentication talab qilinadi'
        }
      });
    }

    if (req.user.role !== requiredRole) {
      return res.status(403).json({
        success: false,
        error: {
          code: 'INSUFFICIENT_PERMISSIONS',
          message: `Bu amalni bajarish uchun ${requiredRole} roli kerak`
        }
      });
    }

    next();
  };
};

// Multiple role tekshirish
const requireRoles = (allowedRoles) => {
  return (req, res, next) => {
    if (!req.user) {
      return res.status(401).json({
        success: false,
        error: {
          code: 'UNAUTHORIZED',
          message: 'Authentication talab qilinadi'
        }
      });
    }

    if (!allowedRoles.includes(req.user.role)) {
      return res.status(403).json({
        success: false,
        error: {
          code: 'INSUFFICIENT_PERMISSIONS',
          message: `Ruxsatli rollar: ${allowedRoles.join(', ')}`
        }
      });
    }

    next();
  };
};

module.exports = {
  authenticateToken,
  requireRole,
  requireRoles
};
```

### Middleware'dan foydalanish:

```javascript
// routes/blogs.js
const express = require('express');
const { authenticateToken, requireRole, requireRoles } = require('../middleware/auth');

const router = express.Router();

// Public route - authentication kerak emas

  // Barcha bloglarni olish
});

// Protected route - login kerak

  // Faqat login qilgan foydalanuvchilar blog yarata oladi

  const newBlog = {
    title: req.body.title,
    content: req.body.content,

// Admin only route

  // Faqat admin foydalanuvchilar blog o'chira oladi
});

// Multiple roles
router.put('/:id', authenticateToken, requireRoles(['admin', 'moderator']), (req, res) => {
  // Admin va moderatorlar blog tahrirlashi mumkin
});
```

## Sessions va Cookies

#### Session-based authentication:

```javascript
// npm install express-session
const session = require('express-session');

// Session configuration
app.use(session({
  secret: process.env.SESSION_SECRET || 'your-session-secret',
  resave: false,
  saveUninitialized: false,
  cookie: {
    maxAge: 24 * 60 * 60 * 1000, // 24 soat
    httpOnly: true,  // XSS himoyasi
    secure: false    // HTTPS'da true qiling
  }
}));

// Session login
app.post('/api/auth/session-login', async (req, res) => {
  try {
    const { email, password } = req.body;

    // User tekshirish (code above)
    const user = users.find(u => u.email === email);
    const isValid = await bcrypt.compare(password, user.password);

    if (user && isValid) {
      // Session'da user ma'lumotlarini saqlash
      req.session.user = {
        id: user.id,
        email: user.email,
        name: user.name,
        role: user.role
      };

      res.status(200).json({
        success: true,
        data: req.session.user,
        message: 'Session login muvaffaqiyatli'
      });
    } else {
      res.status(401).json({
        success: false,
        error: { message: 'Invalid credentials' }
      });
    }
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

// Session middleware
const requireSession = (req, res, next) => {
  if (!req.session.user) {
    return res.status(401).json({
      success: false,
      error: {
        code: 'SESSION_REQUIRED',
        message: 'Login talab qilinadi'
      }
    });
  }
  next();
};

// Session logout
app.post('/api/auth/logout', requireSession, (req, res) => {
  req.session.destroy((err) => {
    if (err) {
      return res.status(500).json({
        success: false,
        error: { message: 'Logout failed' }
      });
    }

    res.clearCookie('connect.sid'); // Session cookie o'chirish
    res.status(200).json({
      success: true,
      message: 'Muvaffaqiyatli logout qildingiz'
    });
  });
});
```

### JWT vs Sessions:

| Xususiyat | JWT | Sessions |
|-----------|-----|----------|
| **Storage** | Client-side | Server-side |
| **Scalability** | Yuqori | Past |
| **Security** | Token exposure riski | Server kontrolli |
| **Cost** | Arzon | Database storage kerak |
| **Logout** | Qiyin | Oson |

### Google OAuth2 setup:

```javascript
// npm install passport passport-google-oauth20
const passport = require('passport');
const GoogleStrategy = require('passport-google-oauth20').Strategy;

// Google Strategy configuration
passport.use(new GoogleStrategy({
  clientID: process.env.GOOGLE_CLIENT_ID,
  clientSecret: process.env.GOOGLE_CLIENT_SECRET,
  callbackURL: "/api/auth/google/callback"
}, async (accessToken, refreshToken, profile, done) => {
  try {
    // Check if user exists
    let user = users.find(u => u.googleId === profile.id);

    if (user) {
      // User exists
      return done(null, user);
    } else {
      // Create new user
      const newUser = {
        id: users.length + 1,
        googleId: profile.id,
        name: profile.displayName,
        email: profile.emails[0].value,
        avatar: profile.photos[0].value,
        role: 'user'
      };

      users.push(newUser);
      return done(null, newUser);
    }
  } catch (error) {
    return done(error, null);
  }
}));

// Passport middleware
app.use(passport.initialize());

// Google auth routes
app.get('/api/auth/google',
  passport.authenticate('google', { scope: ['profile', 'email'] })
);

app.get('/api/auth/google/callback',
  passport.authenticate('google', { failureRedirect: '/login' }),
  (req, res) => {
    // Successful authentication
    const token = jwt.sign({
      id: req.user.id,
      email: req.user.email,
      name: req.user.name,
      role: req.user.role
    }, JWT_SECRET, { expiresIn: '24h' });

    // Redirect with token
    res.redirect(`${process.env.CLIENT_URL}/dashboard?token=${token}`);
  }
);
```

## API Security Best Practices

### 1. **Password Security:**

```javascript
const bcrypt = require('bcryptjs');
const zxcvbn = require('zxcvbn'); // password strength

// Password hashing
const hashPassword = async (password) => {
  const saltRounds = 12; // production'da 12-14 tavsiya etiladi
  return await bcrypt.hash(password, saltRounds);
};

// Password strength validation
const validatePasswordStrength = (password) => {
  const result = zxcvbn(password);

  if (result.score < 3) {
    return {
      valid: false,
      message: 'Password juda zaif. Quyidagi tavsiyalarga amal qiling:',
      suggestions: result.feedback.suggestions
    };
  }

  return { valid: true };
};

// Password validation middleware
const validatePassword = (req, res, next) => {
  const { password } = req.body;

  if (!password) {
    return res.status(400).json({
      error: { message: 'Password talab qilinadi' }
    });
  }

  // Minimum length
  if (password.length < 8) {
    return res.status(400).json({
      error: { message: 'Password kamida 8 ta belgidan iborat bo\'lishi kerak' }
    });
  }

  // Complexity check
  const hasUpper = /[A-Z]/.test(password);
  const hasLower = /[a-z]/.test(password);
  const hasNumber = /\d/.test(password);
  const hasSpecial = /[!@#$%^&*(),.?":{}|<>]/.test(password);

  if (!(hasUpper && hasLower && hasNumber && hasSpecial)) {
    return res.status(400).json({
      error: {
        message: 'Password katta harf, kichik harf, raqam va maxsus belgini o\'z ichiga olishi kerak'
      }
    });
  }

  next();
};
```

### 2. **Rate Limiting:**

```javascript
// npm install express-rate-limit
const rateLimit = require('express-rate-limit');

// Global rate limiting
const globalLimiter = rateLimit({
  windowMs: 15 * 60 * 1000, // 15 daqiqa
  max: 100, // maksimal 100 ta request
  message: {
    error: {
      code: 'TOO_MANY_REQUESTS',
      message: 'Juda ko\'p request yuborildi, biroz kuting'
    }
  }
});

// Login rate limiting (stricter)
const loginLimiter = rateLimit({
  windowMs: 15 * 60 * 1000, // 15 daqiqa
  max: 5, // maksimal 5 ta login urinish
  skipSuccessfulRequests: true,
  message: {
    error: {
      code: 'LOGIN_ATTEMPTS_EXCEEDED',
      message: 'Juda ko\'p login urinish. 15 daqiqadan keyin qayta urinib ko\'ring'
    }
  }
});

// Apply rate limiting
app.use(globalLimiter);
app.use('/api/auth/login', loginLimiter);
```

### 3. **Input Validation:**

```javascript
// npm install validator
const validator = require('validator');

const validateInput = (req, res, next) => {
  const { email, name } = req.body;

  // Email validation
  if (email && !validator.isEmail(email)) {
    return res.status(400).json({
      error: {
        code: 'INVALID_EMAIL',
        message: 'To\'g\'ri email formatini kiriting'
      }
    });
  }

  // Name validation
  if (name) {
    if (!validator.isLength(name, { min: 2, max: 50 })) {
      return res.status(400).json({
        error: {
          code: 'INVALID_NAME',
          message: 'Ism 2-50 ta belgidan iborat bo\'lishi kerak'
        }
      });
    }

    // XSS prevention
    if (!validator.isAlphanumeric(name.replace(/\s/g, ''))) {
      return res.status(400).json({
        error: {
          code: 'INVALID_NAME_FORMAT',
          message: 'Ismda faqat harflar, raqamlar va bo\'shliq bo\'lishi mumkin'
        }
      });
    }
  }

  next();
};
```

## Complete Secure API Example

### Secure Blog API:

```javascript
// server.js - To'liq secure API
const express = require('express');
const bcrypt = require('bcryptjs');
const jwt = require('jsonwebtoken');
const rateLimit = require('express-rate-limit');
const helmet = require('helmet');
const cors = require('cors');
const validator = require('validator');

const app = express();

// Security middleware
app.use(helmet());
app.use(cors({
  origin: process.env.CLIENT_URL,
  credentials: true
}));

app.use(express.json({ limit: '1mb' })); // Body size limit
app.use(express.urlencoded({ extended: true, limit: '1mb' }));

// Rate limiting
const globalLimiter = rateLimit({
  windowMs: 15 * 60 * 1000,
  max: 100,
  message: { error: 'Too many requests' }
});

const authLimiter = rateLimit({
  windowMs: 15 * 60 * 1000,
  max: 5,
  message: { error: 'Too many auth attempts' }
});

app.use(globalLimiter);
app.use('/api/auth', authLimiter);

// Routes
const authRoutes = require('./routes/auth');
const blogRoutes = require('./routes/blogs');

app.use('/api/auth', authRoutes);
app.use('/api/blogs', blogRoutes);

// Global error handler
app.use((err, req, res, next) => {
  console.error(err.stack);

  // JWT errors
  if (err.name === 'JsonWebTokenError') {
    return res.status(401).json({
      error: { code: 'INVALID_TOKEN', message: 'Token yaroqsiz' }
    });
  }

  if (err.name === 'TokenExpiredError') {
    return res.status(401).json({
      error: { code: 'TOKEN_EXPIRED', message: 'Token muddati tugagan' }
    });
  }

  // Default error
  res.status(500).json({
    error: {
      code: 'INTERNAL_SERVER_ERROR',
      message: 'Server xatosi yuz berdi'
    }
  });
});

const PORT = process.env.PORT || 3000;
app.listen(PORT, () => {
  console.log(`🔒 Secure API server running on port ${PORT}`);
});
```
